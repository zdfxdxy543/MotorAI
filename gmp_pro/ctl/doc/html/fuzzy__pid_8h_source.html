102GMP_STATIC_INLINE ctrl_gt ctl_step_fuzzy_pid(ctl_fuzzy_pid_t* fp, ctrl_gt error)
103{
104 // 1. Calculate change in error (EC)
105 ctrl_gt error_change = error - fp->last_error;
106
107 // 2. Quantize E and EC for LUT input
108 ctrl_gt e_quantized = ctl_mul(error, float2ctrl(fp->e_q_factor));
109 ctrl_gt ec_quantized = ctl_mul(error_change, float2ctrl(fp->ec_q_factor));
110
111 // 3. Look up PID parameter adjustments from fuzzy rule tables
112 ctrl_gt delta_kp = ctl_step_interpolate_lut2d(&fp->d_kp_lut, e_quantized, ec_quantized);
113 ctrl_gt delta_ki = ctl_step_interpolate_lut2d(&fp->d_ki_lut, e_quantized, ec_quantized);
114 ctrl_gt delta_kd = ctl_step_interpolate_lut2d(&fp->d_kd_lut, e_quantized, ec_quantized);
115
116 // 4. Calculate the new, tuned PID parameters
117 parameter_gt tuned_kp = fp->base_kp + ctrl2float(delta_kp);
118 parameter_gt tuned_ti = fp->base_ti + ctrl2float(delta_ki);
119 parameter_gt tuned_td = fp->base_td + ctrl2float(delta_kd);
120
121 // Ensure parameters remain physically meaningful
122 if (tuned_kp < 0)
123 tuned_kp = 0;
124 if (tuned_ti < 1e-9f)
125 tuned_ti = 1e-9f; // Avoid division by zero
126 if (tuned_td < 0)
127 tuned_td = 0;
128
129 // 5. Update the underlying PID controller's coefficients
130 parameter_gt T = 1.0f / fp->fs;
131 fp->pid.kp = float2ctrl(tuned_kp);
132 fp->pid.ki = float2ctrl(tuned_kp * T / tuned_ti);
133 fp->pid.kd = float2ctrl(tuned_kp * tuned_td / T);
134
135 // 6. Execute the standard PID step function
136 ctrl_gt output = ctl_step_pid_par(&fp->pid, error);
137
138 // 7. Update state for the next iteration
139 fp->last_error = error;
140
141 return output;
142}
