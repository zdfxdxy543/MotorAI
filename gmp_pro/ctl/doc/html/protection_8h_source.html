75GMP_STATIC_INLINE fast_gt ctl_step_protection_monitor(ctl_protection_monitor_t* mon)
76{
77 uint32_t i;
78
79 // Do not check if a fault is already latched
80 if (mon->fault_index != -1)
81 {
82 return 0; // No *new* fault
83 }
84
85 for (i = 0; i < mon->num_items; ++i)
86 {
87 const ctl_protection_item_t* item = &mon->item_set[i];
88 if ((*item->source > item->supremum) || (*item->source < item->infimum))
89 {
90 mon->fault_index = (int32_t)i;
91 return 1; // New fault detected
92 }
93 }
94
95 return 0; // No fault
96}
