66GMP_STATIC_INLINE ctrl_gt ctl_step_vir_imp(vir_imp_t* imp, ctrl_gt input_current)
67{
68 // 1. Resistive Part: V = I * R
69 ctrl_gt v_r = ctl_mul(input_current, imp->gain_R);
70
71 // 2. Inductive Part: V = L * di/dt
72 // The biquad is configured as a band-limited differentiator with gain L.
73 ctrl_gt v_l = ctl_step_biquad_filter(&imp->diff_filter, input_current);
74
75 // 3. Total Impedance Drop
76 imp->out = v_r + v_l;
