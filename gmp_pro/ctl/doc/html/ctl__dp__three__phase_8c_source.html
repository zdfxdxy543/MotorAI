125void ctl_init_ddsrf_pll_auto_tune(ddsrf_pll_t* pll, parameter_gt f_base, parameter_gt f_ctrl, parameter_gt voltage_mag,
126 parameter_gt bandwidth_hz)
127{
128 // 1. Calculate Loop Gains (Same as SRF-PLL)
129 parameter_gt omega_n = CTL_PARAM_CONST_2PI * bandwidth_hz;
130 parameter_gt loop_gain_constant = CTL_PARAM_CONST_2PI * voltage_mag * f_base;
131
132 if (loop_gain_constant < 0.0001f)
133 loop_gain_constant = 0.0001f;
134
135 parameter_gt damping = 0.707f;
136 parameter_gt kp = (2.0f * damping * omega_n) / loop_gain_constant;
137 parameter_gt ki = (omega_n * omega_n) / loop_gain_constant;
138
139 // 2. Set Decoupling Bandwidth
140 // Optimal is usually freq_grid / sqrt(2)
141 parameter_gt decoupling_fc = f_base / 1.414f;
142
143 // 3. Initialize
144 ctl_init_ddsrf_pll(pll, f_base, kp, ki, f_ctrl, decoupling_fc);
145}

293void ctl_update_gfl_inv_coeff(gfl_inv_ctrl_t* inv, gfl_inv_ctrl_init_t* init)
294{
295 int i = 0;
296
297 for (i = 0; i < 3; ++i)
298 {
299 ctl_init_filter_iir1_lpf(&inv->filter_iabc[i], init->fs, init->current_adc_fc);
300 ctl_init_filter_iir1_lpf(&inv->filter_uabc[i], init->fs, init->voltage_adc_fc);
301 }
302
303 parameter_gt kp_dq =
304 init->grid_filter_L * CTL_PARAM_CONST_2PI * init->current_loop_bw * init->i_base / init->v_base;
305 parameter_gt ki_dq = kp_dq * CTL_PARAM_CONST_2PI * init->current_loop_zero;
306
307 ctl_init_pid(&inv->pid_idq[phase_d], kp_dq, ki_dq, 0, init->fs);
308 ctl_init_pid(&inv->pid_idq[phase_q], kp_dq, ki_dq, 0, init->fs);
309
310 ctl_init_lead_form3(&inv->lead_compensator[phase_d], init->current_phase_lag, init->current_loop_bw, init->fs);
311 ctl_init_lead_form3(&inv->lead_compensator[phase_q], init->current_phase_lag, init->current_loop_bw, init->fs);
312
313#ifdef USING_DSOGI_PLL
314 ctl_init_dsogi_pll(&inv->pll, init->freq_base, init->fs, init->v_grid, init->pll_bw, init->k_pll_sogi);
315#else
316 ctl_init_srf_pll_auto_tune(&inv->pll, init->freq_base, init->fs, init->v_grid, init->pll_bw, 0.707f);
317#endif // USING_DSOGI_PLL
318
319 ctl_init_ramp_generator_via_freq(&inv->rg, init->fs, init->freq_base, 1, 0);
320
321 // Only L filter grid connector
322 if (init->grid_filter_C < 1e-9f)
323 {
324 // close active damping
325 inv->flag_enable_active_damping = 0;
326 inv->coef_ff_damping = 0;
327 }
328 // LC filter grid connector
329 else
330 {
331#if GFL_CAPACITOR_CURRENT_CALCULATE_MODE == 1 || GFL_CAPACITOR_CURRENT_CALCULATE_MODE == 2
332 // active damping voltage = damping_gain * iCdq
333 inv->coef_ff_damping = float2ctrl(init->active_damping_resister * init->i_base / init->v_base);
334#elif GFL_CAPACITOR_CURRENT_CALCULATE_MODE == 3
335 // active damping voltage = damping_gain * (udq - udq_last)
336 inv->coef_ff_damping = float2ctrl(init->active_damping_resister * init->grid_filter_C * init->fs);
337#elif GFL_CAPACITOR_CURRENT_CALCULATE_MODE == 4
338 // TODO FIX HERE
339#endif // GFL_CAPACITOR_CURRENT_CALCULATE_MODE
340
341 ctl_init_biquad_bpf(&inv->filter_damping[phase_d], init->fs, init->active_damping_center_freq,
342 init->active_damping_filter_q);
343 ctl_init_biquad_bpf(&inv->filter_damping[phase_q], init->fs, init->active_damping_center_freq,
344 init->active_damping_filter_q);
345 }
346
347 // decoupling feed-forward
348 inv->coef_ff_decouple = CTL_PARAM_CONST_2PI * init->grid_filter_L * init->freq_base * init->i_base / init->v_base;
349}

433void ctl_auto_tuning_neg_inv(inv_neg_ctrl_init_t* neg_init, const gfl_inv_ctrl_init_t* _gfl_init)
434{
435 gmp_base_assert(neg_init);
436 gmp_base_assert(_gfl_init);
437
438 // 1. Copy Basic System Parameters
439 neg_init->fs = _gfl_init->fs;
440 neg_init->freq_base = _gfl_init->freq_base;
441
442 // 2. Tuning Separation Filter (Biquad LPF)
443 // Goal: Attenuate 2*freq_base (100Hz) ripple significantly.
444 // Strategy: Set fc at approx freq_base / 2.5 (e.g., 20Hz for 50Hz grid).
445 // This offers a trade-off between ripple attenuation and delay.
446 neg_init->seq_filter_fc = _gfl_init->freq_base / 2.5f;
447 //neg_init->seq_filter_q = 0.707f; // Butterworth response
448 neg_init->seq_filter_q = 5.0f; // Butterworth response
449
450 // 3. Tuning Current Loop (Inner Loop)
451 // Constraint: Bandwidth must be lower than filter fc to maintain stability margin.
452 // Strategy: Set BW at filter_fc / 2, it's too small.
453 //parameter_gt neg_current_bw = neg_init->seq_filter_fc / 2.0f;
454
455 // Set BW at fs/50
456 parameter_gt neg_current_bw = neg_init->fs / 200.0f;
457
458 // Set BW at 50Hz
459 //parameter_gt neg_current_bw = 50.0f;
460
461 // Calculate PI gains based on Plant Model: V = L * di/dt
462 // Kp_pu = (2*pi*BW * L) * (I_base / V_base)
463 // Note: This Kp is calculated conceptually here, but physically in update_coeff.
464 // We store the tuning target implicitly via the update function logic,
465 // but here we define the limits and integration speeds.
466
467 // For auto-tuning function, we usually just set reasonable defaults if the struct
468 // expects raw Kp/Ki. However, looking at ctl_update_gfl_inv_coeff, it calculates Kp inside.
469 // BUT inv_neg_ctrl_init_t stores raw Kp/Ki. So we calculate them here.
470
471 parameter_gt L_val = _gfl_init->grid_filter_L;
472 parameter_gt kp_c_si = CTL_PARAM_CONST_2PI * neg_current_bw * L_val;
473
474 // Convert to P.U.
475 neg_init->kp_current = kp_c_si * _gfl_init->i_base / _gfl_init->v_base;
476
477 // Ki: Place zero at low frequency (e.g., BW / 10) to eliminate steady state error
478 // Ki = Kp * 2*pi * f_zero
479 neg_init->ki_current = CTL_PARAM_CONST_2PI * (neg_current_bw / 10.0f);
480
481 // Default Limits
482 neg_init->limit_current_out = 0.8f; // 0.8 p.u. voltage compensation limit
483
484 // 4. Tuning Voltage Loop (Outer Loop) - Optional
485 // Constraint: Must be significantly slower than current loop.
486 // Strategy: Set BW at Current_BW / 5.
487 parameter_gt neg_voltage_bw = neg_current_bw / 5.0f;
488
489 // Plant Model: I = C * dv/dt (Capacitor dominates)
490 // If no C (L filter), voltage control is invalid, but we calculate anyway.
491 parameter_gt C_val = (_gfl_init->grid_filter_C > 1e-9f) ? _gfl_init->grid_filter_C : 1e-5f; // prevent zero
492
493 parameter_gt kp_v_si = CTL_PARAM_CONST_2PI * neg_voltage_bw * C_val;
494
495 // Convert to P.U. (Note: Output is Current Ref, Input is Voltage)
496 // Gain units: A/V. P.U. conversion factor is V_base / I_base.
497 neg_init->kp_voltage = kp_v_si * _gfl_init->v_base / _gfl_init->i_base;
498
499 // Ki
500 neg_init->ki_voltage = CTL_PARAM_CONST_2PI * (neg_voltage_bw / 5.0f);
501
502 neg_init->limit_voltage_out = 0.5f; // 0.5 p.u. current injection limit
503}

584 parameter_gt f_6th = init->freq_base * 6.0f;
585 parameter_gt f_12th = init->freq_base * 12.0f;
586
587 // 2. Init QR Controllers (Using Pre-warping!)
588 // D-Axis

590 ctl_init_qr_controller_prewarped(&hcm->qr_d_12th, init->kr_12th, f_12th, init->bw_12th, init->fs);
591

593 ctl_init_qr_controller_prewarped(&hcm->qr_q_6th, init->kr_6th, f_6th, init->bw_6th, init->fs);
