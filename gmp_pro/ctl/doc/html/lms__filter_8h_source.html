103GMP_STATIC_INLINE ctrl_gt ctl_step_lms_filter(ctl_lms_filter_t* lms, ctrl_gt input, ctrl_gt desired)
104{
105 uint32_t i;
106
107 // --- Key Point Analysis 2: Update Input Buffer ---
108 // A circular buffer is used to efficiently store the most recent 'order' input samples.
109 // This avoids large memory shifts in every step.
110 lms->buffer[lms->buffer_index] = input;
111
112 // --- Key Point Analysis 3: Calculate Filter Output ---
113 // The output y(n) is the dot product of the weights vector W(n) and the input vector X(n).
114 // y(n) = W^T * X(n)
115 lms->output = 0.0f;
116 uint32_t j = lms->buffer_index;
117 for (i = 0; i < lms->order; i++)
118 {
119 lms->output += lms->weights[i] * lms->buffer[j];
120 if (j == 0)
121 {
122 j = lms->order - 1;
123 }
124 else
125 {
126 j--;
127 }
128 }
129
130 // --- Key Point Analysis 4: Calculate Error Signal ---
131 // The error e(n) is the difference between the desired signal and the filter's output.
132 // This error signal is what drives the adaptation process.
133 // e(n) = d(n) - y(n)
134 lms->error = desired - lms->output;
135
136 // --- Key Point Analysis 5: Update Filter Weights ---
137 // This is the core of the LMS algorithm. The weights are adjusted in the direction
138 // that minimizes the error. The step-size 'mu' controls the speed and stability
139 // of this convergence. A larger mu leads to faster adaptation but can cause instability.
140 // W(n+1) = W(n) + mu * e(n) * X(n)
141 j = lms->buffer_index;
142 for (i = 0; i < lms->order; i++)
143 {
144 lms->weights[i] += lms->mu * lms->error * lms->buffer[j];
145 if (j == 0)
146 {
147 j = lms->order - 1;
148 }
149 else
150 {
151 j--;
152 }
153 }
154
155 // Advance the circular buffer index for the next sample
156 lms->buffer_index++;
157 if (lms->buffer_index >= lms->order)
158 {
159 lms->buffer_index = 0;
160 }
161
162 return lms->error;
163}
