253void ctl_init_sogi(ctl_sogi_t* sogi, parameter_gt gain, parameter_gt freq_r, parameter_gt damp, parameter_gt fs)
254{
255 // Calculate the sampling period
256 parameter_gt Ts = 1.0f / fs;
257 // Calculate the resonant angular frequency
258 parameter_gt omega_r = 2.0f * CTL_PARAM_CONST_PI * freq_r;
259
260 // Set the controller parameters
261 sogi->gain = float2ctrl(gain);
262 sogi->k_damp = float2ctrl(damp);
263 // Pre-calculate the resonant frequency gain for the discrete implementation
264 sogi->k_r = float2ctrl(omega_r * Ts);
265
266 // Clear all state variables to ensure a clean start
267 ctl_clear_sogi(sogi);
268}

275void ctl_init_s_function(ctl_s_function_t* obj, parameter_gt gain, parameter_gt f_z1, parameter_gt f_z2,
276 parameter_gt f_p1, parameter_gt f_p2, parameter_gt fs)
277{
278 // Convert frequencies (Hz) to angular frequencies (rad/s)
279 parameter_gt wz1 = CTL_PARAM_CONST_2PI * f_z1;
280 parameter_gt wz2 = CTL_PARAM_CONST_2PI * f_z2;
281 parameter_gt wp1 = CTL_PARAM_CONST_2PI * f_p1;
282 parameter_gt wp2 = CTL_PARAM_CONST_2PI * f_p2;
283
284 // From H(s) = K * [(s/wz1+1)(s/wz2+1)] / [(s/wp1+1)(s/wp2+1)],
285 // we get the polynomial form H(s) = K * (B2*s^2 + B1*s + B0) / (A2*s^2 + A1*s + A0)
286 // Note: This form is normalized so that the constant term is 1.
287 parameter_gt B2 = 1.0f / (wz1 * wz2);
288 parameter_gt B1 = (wz1 + wz2) / (wz1 * wz2);
289 parameter_gt B0 = 1.0f;
290
291 parameter_gt A2 = 1.0f / (wp1 * wp2);
292 parameter_gt A1 = (wp1 + wp2) / (wp1 * wp2);
293 parameter_gt A0 = 1.0f;
294
295 // Discretize using the Bilinear Transform: s = 2/T * (1-z^-1)/(1+z^-1)
296 parameter_gt T = 1.0f / fs;
297 parameter_gt T_sq = T * T;
298 parameter_gt two_over_T = 2.0f / T;
299 parameter_gt four_over_T_sq = 4.0f / T_sq;
300
301 // Common denominator for Z-domain coefficients
302 parameter_gt den = A2 * four_over_T_sq + A1 * two_over_T + A0;
303 parameter_gt inv_den = 1.0f / den;
304
305 // Calculate Z-domain coefficients for H(z) = (b0+b1*z^-1+b2*z^-2)/(1+a1*z^-1+a2*z^-2)
306 obj->b0 = float2ctrl(gain * (B2 * four_over_T_sq + B1 * two_over_T + B0) * inv_den);
307 obj->b1 = float2ctrl(gain * (-2.0f * B2 * four_over_T_sq + 2.0f * B0) * inv_den);
308 obj->b2 = float2ctrl(gain * (B2 * four_over_T_sq - B1 * two_over_T + B0) * inv_den);
309
310 obj->a1 = float2ctrl((-2.0f * A2 * four_over_T_sq + 2.0f * A0) * inv_den);
311 obj->a2 = float2ctrl((A2 * four_over_T_sq - A1 * two_over_T + A0) * inv_den);
312
313 // Clear initial states
314 ctl_clear_s_function(obj);
315}
