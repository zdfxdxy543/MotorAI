105GMP_STATIC_INLINE ctrl_gt ctl_step_spfc_ctrl(spfc_t* pfc)
106{
107 if (!pfc->flag_enable)
108 {
109 pfc->boost_modulation = 0; // Turn off PWM if disabled
110 return pfc->boost_modulation;
111 }
112
113 // Read sensor values from interfaces
114 ctrl_gt rectifier_voltage = pfc->adc_rect_u->value;
115 ctrl_gt rectifier_current = pfc->adc_rect_i->value;
116 ctrl_gt dc_voltage = pfc->adc_dc_u->value;
117
118 // Outer DC Bus voltage controller: calculates the required current amplitude.
119 pfc->current_set = ctl_step_pid_ser(&pfc->voltage_ctrl, pfc->voltage_set - dc_voltage);
120
121 // Generate the instantaneous current reference by shaping the amplitude with the rectified voltage waveform.
122 pfc->current_ref = ctl_mul(pfc->current_set, rectifier_voltage);
123
124 // Inner Current controller: forces the inductor current to follow the reference.
125 pfc->boost_modulation = ctl_step_pid_ser(&pfc->current_ctrl, pfc->current_ref - rectifier_current);
126
127 // Return the calculated modulation duty cycle
128 return pfc->boost_modulation;
129}
