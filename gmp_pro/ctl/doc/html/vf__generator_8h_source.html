208GMP_STATIC_INLINE ctrl_gt ctl_step_slope_f_pu(ctl_slope_f_pu_controller* ctrl)
209{
210 // 1. Ramp the Frequency in PU domain
211 // Input: Target PU (e.g., 0.5 for half speed)
212 // Output: Current PU (e.g., ramps 0.1, 0.11, 0.12...)
213 ctrl->current_freq_pu = ctl_step_slope_limiter(&ctrl->freq_slope, ctrl->target_freq_pu);
214
215 // 2. Convert PU Frequency to Ramp Generator Slope (Angle Increment)
216 // Slope = Current_PU * (Base_Hz / ISR_Hz)
217 ctrl_gt rg_slope = ctl_mul(ctrl->current_freq_pu, ctrl->ratio_freq_pu_to_step);
218
219 // 3. Update Ramp Generator
220 ctl_set_ramp_generator_slope(&ctrl->rg, rg_slope);
221
222 // 4. Step Ramp Generator to get Angle
223 ctrl->enc.elec_position = ctl_step_ramp_generator(&ctrl->rg);
224 ctrl->enc.position = ctrl->enc.elec_position;
225
226 return ctrl->enc.elec_position;
227}

339GMP_STATIC_INLINE ctrl_gt ctl_step_const_vf(ctl_const_vf_controller* ctrl)
340{
341 // Step to the next frequency according to the slope limit
342 ctrl->current_freq = ctl_step_slope_limiter(&ctrl->freq_slope, ctrl->target_frequency);
343
344 // Calculate the absolute value of the frequency for voltage calculation
345 ctrl_gt freq_abs = (ctrl->current_freq >= 0) ? ctrl->current_freq : -ctrl->current_freq;
346
347 // Calculate target voltage magnitude based on the V/F profile
348 if (freq_abs > ctrl->freq_deadband)
349 {
350 ctrl->target_voltage = ctl_step_saturation(&ctrl->volt_sat, ctrl->v_bias + ctl_mul(ctrl->v_over_f, freq_abs));
351 }
352 else // In dead-band
353 {
354 ctrl->target_voltage = 0;
355 }
356
357 // Update the ramp generator's slope (i.e., its frequency)
358 ctl_set_ramp_generator_slope(&ctrl->rg, ctrl->current_freq);
359
360 // Generate the next angle based on the new frequency
361 ctrl->enc.elec_position = ctl_step_ramp_generator(&ctrl->rg);
362 ctrl->enc.position = ctrl->enc.elec_position;
363
364 // Return the calculated voltage magnitude
365 return ctrl->target_voltage;
366}
