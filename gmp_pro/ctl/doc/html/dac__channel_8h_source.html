80GMP_STATIC_INLINE dac_gt ctl_step_dac_channel(dac_channel_t* dac)
81{
82 // Apply gain and bias to the raw control value
83 ctrl_gt temp_val = ctl_add(ctl_mul(dac->raw, dac->gain), dac->bias);
84
85#if defined CTRL_GT_IS_FIXED
86 // For fixed-point, shift the result to match the DAC resolution
87 dac->value = (dac_gt)(temp_val >> (dac->iqn - dac->resolution));
88#elif defined CTRL_GT_IS_FLOAT
89 // For float, scale the per-unit value by the maximum DAC value
90 dac->value = (dac_gt)(temp_val * (float)((1 << dac->resolution) - 1));
91#else
92#error "The system does not specify ctrl_gt is float or fixed. You should define CTRL_GT_IS_FLOAT or CTRL_GT_IS_FIXED."
93#endif // CTRL_GT_IS_XXX
94
95 return dac->value;
96}
