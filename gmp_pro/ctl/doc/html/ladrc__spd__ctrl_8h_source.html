109GMP_STATIC_INLINE ctrl_gt ctl_step_ladrc_speed_pu(ctl_ladrc_speed_pu_t* ladrc, ctrl_gt r_pu, ctrl_gt y_pu)
110{
111 // --- State Feedback & Disturbance Compensation ---
112 // Calculate the preliminary control law based on the estimated states from the *previous* time step.
113 ctrl_gt u0 = ladrc->wc * (r_pu - ladrc->z1);
114
115 // Compensate for the total disturbance and apply the system gain.
116 if (fabsf(ladrc->b0) > 1e-9f)
117 {
118 ladrc->u_out_pu = (u0 - ladrc->z2) / ladrc->b0;
119 }
120 else
121 {
122 ladrc->u_out_pu = 0.0f;
123 }
124
125 // --- Linear Extended State Observer (LESO) ---
126 // Update the observer for the *next* time step using the output from *this* step.
127 ctrl_gt error = ladrc->z1 - y_pu;
128 ctrl_gt beta1 = 2.0f * ladrc->wo;
129 ctrl_gt beta2 = ladrc->wo * ladrc->wo;
130
131 ladrc->z1 = ladrc->z1 + ladrc->h * (ladrc->z2 - beta1 * error + ladrc->b0 * ladrc->u_out_pu);
132 ladrc->z2 = ladrc->z2 + ladrc->h * (-beta2 * error);
133
134 return ladrc->u_out_pu;
135}
