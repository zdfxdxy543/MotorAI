115GMP_STATIC_INLINE void ctl_step_vel_pos_ctrl(ctl_vel_pos_controller_t* ctrl)
116{
117 // Position Control Loop
118 if (ctrl->flag_enable_position_ctrl)
119 {
120 if (ctl_step_divider(&ctrl->div_position))
121 {
122 // Calculate position error
123 int32_t rev_error = ctrl->target_revs - ctrl->pos_if->revolutions;
124 ctrl_gt ang_error = ctrl->target_angle - ctrl->pos_if->position;
125
126 // Total position error
127 ctrl_gt pos_error = (ctrl_gt)rev_error + ang_error;
128
129 // Update position controller
130 ctrl->target_velocity = ctl_step_pid_par(&ctrl->pos_ctrl, pos_error);
131 }
132 }
133
134 // Velocity Control Loop
135 if (ctrl->flag_enable_velocity_ctrl)
136 {
137 if (ctl_step_divider(&ctrl->div_velocity))
138 {
139 // Calculate velocity error
140 ctrl_gt spd_feedback = ctrl->spd_if->speed;
141 ctrl_gt spd_error = ctrl->target_velocity - spd_feedback;
142
143 // Update velocity controller
144 ctrl->cur_output = ctl_step_pid_par(&ctrl->vel_ctrl, spd_error);
145 }
146 }
147}
