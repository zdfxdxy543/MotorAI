95GMP_STATIC_INLINE ctrl_gt ctl_step_adc_channel(adc_channel_t* adc, adc_gt raw)
96{
97 adc->raw = raw;
98
99#if defined CTRL_GT_IS_FIXED
100 // For fixed-point number, transfer ADC data to the specified IQn format
101 ctrl_gt raw_data = adc->raw << (adc->iqn - adc->resolution);
102#elif defined CTRL_GT_IS_FLOAT
103 // For floating-point number, transfer ADC data to per-unit (p.u.)
104 ctrl_gt raw_data = (ctrl_gt)adc->raw / (1 << adc->resolution);
105#else
106#error "The system does not specify ctrl_gt is float or fixed. You should define CTRL_GT_IS_FLOAT or CTRL_GT_IS_FIXED."
107#endif // CTRL_GT_IS_XXX
108
109 // Remove bias
110 ctrl_gt raw_without_bias = raw_data - adc->bias;
111
112 // Apply gain
113 adc->control_port.value = ctl_mul(raw_without_bias, adc->gain);
114
115 return adc->control_port.value;
116}

198{
199 ctrl_gt raw_data;

215
216 raw_without_bias = raw_data - adc_obj->bias[i];
