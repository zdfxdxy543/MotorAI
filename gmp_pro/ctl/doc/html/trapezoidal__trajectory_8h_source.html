102GMP_STATIC_INLINE ctrl_gt ctl_step_trap_planner(ctl_trap_planner_t* planner)
103{
104 ctrl_gt dist_to_go = planner->target_pos - planner->current_pos;
105 ctrl_gt accel = 0.0f;
106
107 // Check if we are within the target deadband
108 if (fabsf(dist_to_go) < planner->pos_deadband)
109 {
110 planner->current_vel = 0.0f;
111 return planner->current_pos;
112 }
113
114 // Determine the direction of motion
115 ctrl_gt direction = (dist_to_go > 0.0f) ? 1.0f : -1.0f;
116
117 // --- Core Logic: Decide whether to accelerate or decelerate ---
118
119 // 1. Calculate the distance required to stop from the current velocity
120 ctrl_gt brake_dist = (planner->current_vel * planner->current_vel) / (2.0f * planner->max_accel);
121
122 // 2. Check if we are within the braking distance
123 // and also moving towards the target (velocity has same sign as dist_to_go)
124 int moving_towards_target = !(signbit(planner->current_vel) ^ signbit(dist_to_go));
125
126 if (moving_towards_target && (fabsf(dist_to_go) <= brake_dist))
127 {
128 // Inside braking zone: must decelerate
129 accel = -direction * planner->max_accel;
130 }
131 else
132 {
133 // Outside braking zone: accelerate towards target
134 accel = direction * planner->max_accel;
135 }
136
137 // --- Integration and Saturation ---
138
139 // 3. Integrate acceleration to get new velocity
140 planner->current_vel += accel;
141
142 // 4. Saturate velocity at the configured limit
143 if (fabsf(planner->current_vel) > planner->max_vel)
144 {
145 planner->current_vel = direction * planner->max_vel;
146 }
147
148 // 5. Integrate velocity to get new position
149 planner->current_pos += planner->current_vel;
150
151 return planner->current_pos;
152}
