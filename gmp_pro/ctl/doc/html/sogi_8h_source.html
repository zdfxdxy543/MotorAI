100GMP_STATIC_INLINE ctrl_gt ctl_step_sogi(ctl_sogi_t* sogi, ctrl_gt input)
101{
102 // err = u - y_d
103 ctrl_gt err = input - sogi->d_integrate;
104 // ref = k * err
105 sogi->integrate_reference = ctl_mul(sogi->k_damp, err);
106 // y_d(n) = y_d(n-1) + wr*T * (ref - y_q(n-1))
107 sogi->d_integrate += ctl_mul(sogi->k_r, sogi->integrate_reference - sogi->q_integrate);
108 // y_q(n) = y_q(n-1) + wr*T * y_d(n)
109 sogi->q_integrate += ctl_mul(sogi->k_r, sogi->d_integrate);
110
111 return ctl_mul(sogi->d_integrate, sogi->gain);
112}
