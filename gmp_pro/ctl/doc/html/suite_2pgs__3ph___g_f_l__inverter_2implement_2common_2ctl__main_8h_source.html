74{
75 // ADC calibrator routine
76 if (flag_enable_adc_calibrator)
77 {
78 if (index_adc_calibrator == 13)
79 ctl_step_adc_calibrator(&adc_calibrator, idc.control_port.value);
80 else if (index_adc_calibrator == 12)
81 ctl_step_adc_calibrator(&adc_calibrator, udc.control_port.value);
82 else if (index_adc_calibrator <= 11 && index_adc_calibrator >= 9)
83 ctl_step_adc_calibrator(&adc_calibrator, uuvw.control_port.value.dat[index_adc_calibrator - 9]);
84 else if (index_adc_calibrator <= 8 && index_adc_calibrator >= 6)
85 ctl_step_adc_calibrator(&adc_calibrator, vabc.control_port.value.dat[index_adc_calibrator - 6]);
86 else if (index_adc_calibrator <= 5 && index_adc_calibrator >= 3)
87 ctl_step_adc_calibrator(&adc_calibrator, iabc.control_port.value.dat[index_adc_calibrator - 3]);
88 else if (index_adc_calibrator <= 2)
89 ctl_step_adc_calibrator(&adc_calibrator, iuvw.control_port.value.dat[index_adc_calibrator]);
90 }
91
92 // normal controller routine
93 else
94 {
95 ctl_step_triangle_wave_generator(&tri_wave);
96 inv_ctrl.idq_set.dat[phase_d] = tri_wave.output;
97
98 // run controller body
99 ctl_step_gfl_inv_ctrl(&inv_ctrl);
100 ctl_step_neg_inv_ctrl(&neg_current_ctrl);
101
102 ctl_step_gfl_pq(&pq_ctrl);
103
104 if (pq_ctrl.flag_enable)
105 {
106 ctl_set_gfl_inv_current(&inv_ctrl, pq_ctrl.idq_set_out.dat[phase_d], pq_ctrl.idq_set_out.dat[phase_q]);
107 }
108
109 // mix all output
110 spwm.vab0_out.dat[phase_A] = inv_ctrl.vab0_out.dat[phase_A] + neg_current_ctrl.vab_out.dat[phase_A];
111 spwm.vab0_out.dat[phase_B] = inv_ctrl.vab0_out.dat[phase_B] + neg_current_ctrl.vab_out.dat[phase_B];
112 spwm.vab0_out.dat[phase_0] = inv_ctrl.vab0_out.dat[phase_0];
113
114 // modulation
115#if defined USING_NPC_MODULATOR
116 ctl_step_npc_modulator(&spwm);
117#else
118 ctl_step_spwm_modulator(&spwm);
119#endif // USING_NPC_MODULATOR
120 }
