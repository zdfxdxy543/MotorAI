107GMP_STATIC_INLINE void ctl_step_spwm_modulator(spwm_modulator_t* mod)
108{
109 gmp_base_assert(mod);
110 gmp_base_assert(mod->iuvw);
111 gmp_base_assert(mod->pwm_out);
112
113 // Temp variable to handle signed calculations safely
114 pwm_gt pwm_value;
115 int i;
116
117 // Modulation, SPWM
118 ctl_ct_iclarke(&mod->vab0_out, &mod->vabc_out);
119
120 // --- Calculate Raw PWM Values ---
121 // Convert voltage command (-1.0 to 1.0) to duty cycle (0 to 1.0), then to raw PWM value.
122 for (i = 0; i < 3; ++i)
123 {
124#if PWM_MODULATOR_USING_NEGATIVE_LOGIC == 1
125 ctrl_gt pwm_pu = ctl_div2(ctl_add(float2ctrl(1.0f), -mod->vabc_out.dat[i]));
126#else // Positive logic
127 ctrl_gt pwm_pu = ctl_div2(ctl_add(float2ctrl(1.0f), mod->vabc_out.dat[i]));
128#endif // PWM_MODULATOR_USING_NEGATIVE_LOGIC
129
130 // Convert voltage command (-1.0 to 1.0) to duty cycle (0 to 1.0), then to raw PWM value.
131 pwm_value = pwm_mul(pwm_pu, mod->pwm_full_scale);
132
133 //
134 // NOTE: pwm_gt must be a signed int number.
135 // Or dead-time compensator would get a wrong result, when low modulation ratio.
136 //
137
138 // dead-time compensation
139 if (mod->flag_enable_deadband_compensator)
140 {
141 // ---------------------------------------------------------
142 // ZONE 1: Strong Positive Current -> Apply Positive Comp
143 // ---------------------------------------------------------
144 if (mod->iuvw->dat[i] > mod->current_deadband + mod->current_hysteresis_band)
145 {
146 pwm_value += mod->pwm_deadband_comp_val;
147 mod->last_current_dir[i] = 1;
148 }
149
150 // ---------------------------------------------------------
151 // ZONE 2: Strong Negative Current -> Apply Negative Comp
152 // ---------------------------------------------------------
153 else if (mod->iuvw->dat[i] < (-mod->current_deadband - mod->current_hysteresis_band))
154 {
155 if (pwm_value >= mod->pwm_deadband_comp_val)
156 {
157 pwm_value -= mod->pwm_deadband_comp_val;
158 }
159 else
160 {
161 pwm_value = 0; // avoid unsigned number underflow.
162 }
163
164 mod->last_current_dir[i] = -1;
165 }
166
167 // ---------------------------------------------------------
168 // ZONE 3: Inside Deadband (Parasitic Cap Zone) -> No Comp
169 // ---------------------------------------------------------
170 else if ((mod->iuvw->dat[i] > -mod->current_deadband) && (mod->iuvw->dat[i] < mod->current_deadband))
171 {
172 mod->last_current_dir[i] = 0;
173 }
174
175 // ---------------------------------------------------------
176 // ZONE 4: Hysteresis Band -> Keep Last State
177 // ---------------------------------------------------------
178 else
179 {
180 // Current is in the gray area:
181 // [Deadband, Deadband + Hysteresis] OR [-Deadband - Hysteresis, -Deadband]
182
183 if (mod->last_current_dir[i] == 1)
184 {
185 pwm_value += mod->pwm_deadband_comp_val;
186 }
187 else if (mod->last_current_dir[i] == -1)
188 {
189 if (pwm_value >= mod->pwm_deadband_comp_val)
190 {
191 pwm_value -= mod->pwm_deadband_comp_val;
192 }
193 else
194 {
195 pwm_value = 0; // avoid unsigned number underflow.
196 }
197 }
198 else
199 {
200 // last_current_dir == 0
201 // We came from the dead band zone, so we stay 0 (no comp)
202 // until we cross the outer hysteresis limit.
203 }
204 }
205 }
206
207 // Saturate final PWM values to ensure they are within valid range.
208 pwm_value = pwm_sat(pwm_value, mod->pwm_full_scale, 0);
209
210 // write back correct number
211 mod->pwm_out[i] = pwm_value;
212 }
213}

341// --- NPC Output Channel Mapping Macros ---
342// 物理含义：0, 2, 4 对应外管(S1/S4)，1, 3, 5 对应内管(S2/S3)
343// 实际上S1和S3构成互补关系，S2和S4构成互补关系。
344// 假设硬件连接顺序为：A相[S1, S2], B相[S1, S2], C相[S1, S2]
345// 注意：这里的定义取决于您的具体硬件和PWM分组，以下为一种常见定义
346#define NPC_IDX_PHASE_A_OUTER 0 // A相 S1 (Top) / S4 (Bot) - 取决于极性，此处指"外侧斩波管"
347#define NPC_IDX_PHASE_A_INNER 1 // A相 S2 (Inner Top) / S3 (Inner Bot)
348#define NPC_IDX_PHASE_B_OUTER 2
349#define NPC_IDX_PHASE_B_INNER 3
350#define NPC_IDX_PHASE_C_OUTER 4
351#define NPC_IDX_PHASE_C_INNER 5
352
353// 或者使用更通用的计算宏
354#define NPC_OUTER_IDX(phase) ((phase) * 2)
355#define NPC_INNER_IDX(phase) ((phase) * 2 + 1)

442GMP_STATIC_INLINE int get_current_zone(ctrl_gt i_sample, ctrl_gt deadband, ctrl_gt hysteresis, signed short* last_dir)
443{
444 int current_dir = *last_dir; // Default to keep history
445
446 // Thresholds
447 ctrl_gt upper_trip_point = deadband + hysteresis;
448 ctrl_gt lower_trip_point = -deadband - hysteresis;
449 ctrl_gt upper_reset_point = deadband;
450 ctrl_gt lower_reset_point = -deadband;
451
452 // --- ZONE 1: Strong Positive ---
453 if (i_sample > upper_trip_point)
454 {
455 current_dir = 1;
456 }
457 // --- ZONE 2: Strong Negative ---
458 else if (i_sample < lower_trip_point)
459 {
460 current_dir = -1;
461 }
462 // --- ZONE 3: Inside Deadband (Strict Zero) ---
463 else if ((i_sample > lower_reset_point) && (i_sample < upper_reset_point))
464 {
465 current_dir = 0;
466 }
467 // --- ZONE 4: Hysteresis Area (Gray Zone) ---
468 else
469 {
470 // Logic:
471 // If we were 1, stay 1 until we drop below deadband.
472 // If we were -1, stay -1 until we rise above -deadband.
473 // If we were 0, stay 0 until we cross trip points (handled by Zone1/2 checks).
474
475 // This 'else' block implicitly maintains 'last_dir' value
476 // which implements the hysteresis hold.
477 }
478
479 // Update state memory
480 *last_dir = (signed short)current_dir;
481 return current_dir;
482}

488GMP_STATIC_INLINE void ctl_step_npc_modulator(npc_modulator_t* mod)
489{
490 gmp_base_assert(mod);
491
492 int i;
493
494 // Modulation, SPWM
495 ctl_ct_iclarke(&mod->vab0_out, &mod->vabc_out);
496
497 for (i = 0; i < 3; ++i)
498 {
499 // 注意，在DSP中，如果A接上桥，B接下桥，以变流器输出方向为正
500 // 如果不取反输入正电压测量得到的是负电压，为了保证符号统一，需要使用负调制。
501#if PWM_MODULATOR_USING_NEGATIVE_LOGIC == 1
502 ctrl_gt v_ref = -mod->vabc_out.dat[i];
503#else
504 ctrl_gt v_ref = mod->vabc_out.dat[i];
505#endif // PWM_MODULATOR_USING_NEGATIVE_LOGIC
506
507 pwm_gt cmp_outer = 0;
508 pwm_gt cmp_inner = 0;
509
510 // -------------------------------------------------------------
511 // Step 1: Base Modulation (分段映射)
512 // -------------------------------------------------------------
513 if (v_ref >= 0)
514 {
515 // --- Positive Half (P <-> O) ---
516 // S1 (Outer) toggles, S2 (Inner) is ON
517
518 // Calculate S1 Duty
519 cmp_outer = pwm_mul(v_ref, mod->pwm_full_scale);
520
521 // S2 kept ON (100%)
522 cmp_inner = mod->pwm_full_scale;
523 }
524 else
525 {
526 // --- Negative Half (O <-> N) ---
527 // S1 (Outer) is OFF, S2 (Inner) toggles
528 // Note: S2 duty logic is inverted relative to magnitude for N generation
529
530 // S1 kept OFF
531 cmp_outer = 0;
532
533 // Calculate S2 Duty ( 1.0 - |v_ref| )
534 //ctrl_gt v_abs = ctl_abs(v_ref);
535 cmp_inner = pwm_mul(float2ctrl(1.0f) + v_ref, mod->pwm_full_scale);
536 }
537
538 // ============================================================
539 // B. 死区补偿逻辑 (Using get_current_zone)
540 // ============================================================
541 if (mod->flag_enable_deadband_compensator)
542 {
543 // 调用封装好的函数，传入当前相电流、参数和状态记忆
544 int dir = get_current_zone(mod->iuvw->dat[i], mod->current_deadband, mod->current_hysteresis_band,
545 &mod->last_current_dir[i]);
546
547 if (dir == 1) // Strong Positive (Out)
548 {
549 // Current flowing out -> Device drop reduces voltage -> Compensate by adding duty
550 if (v_ref >= 0)
551 cmp_outer += mod->pwm_deadband_comp_val;
552 else
553 cmp_inner += mod->pwm_deadband_comp_val;
554 }
555 else if (dir == -1) // Strong Negative (In)
556 {
557 // Current flowing in -> Device drop increases voltage -> Compensate by reducing duty
558 if (v_ref >= 0)
559 cmp_outer -= mod->pwm_deadband_comp_val;
560 else
561 cmp_inner -= mod->pwm_deadband_comp_val;
562 }
563 // dir == 0: Inside Deadband -> No compensation
564 }
565
566 // -------------------------------------------------------------
567 // Step 3: Bootstrap Protection (Saturation / Max Duty Limit)
568 // 这一步必须在死区补偿之后、最终赋值之前进行
569 // -------------------------------------------------------------
570 // if (mod->flag_enable_bootstrap_protection)
571 // {
572 // // 限制 Outer 管 (S1) 的最大占空比，确保有时间充电
573 // if (cmp_outer > mod->pwm_max_duty_limit)
574 // {
575 // cmp_outer = mod->pwm_max_duty_limit;
576 // }
577
578 // // 限制 Inner 管 (S2) 的最大占空比
579 // // 注意：在负半周，S2 负责产生 O 状态。
580 // // 如果 S2 也是浮地驱动(取决于硬件)，也需要限制。
581 // // 通常 S2 也是需要限制的，防止 Inner 驱动过热或掉电。
582 // if (cmp_inner > mod->pwm_max_duty_limit)
583 // {
584 // cmp_inner = mod->pwm_max_duty_limit;
585 // }
586
587 // // 可选：最小脉宽限制 (防止窄脉冲干扰)
588 // // if (cmp_outer < MIN_PULSE) cmp_outer = 0;
589 // }
590
591 // -------------------------------------------------------------
592 // Step 4: Write Output (Saturation & Assignment)
593 // -------------------------------------------------------------
594 // 最终的安全钳位，防止溢出 ARR
595 mod->pwm_out[NPC_OUTER_IDX(i)] = pwm_sat(cmp_outer, mod->pwm_full_scale, 0);
596 mod->pwm_out[NPC_INNER_IDX(i)] = pwm_sat(cmp_inner, mod->pwm_full_scale, 0);
597 }
