137interrupt void MainISR(void)
138{
139 //
140 // call GMP ISR Controller operation callback function
141 //
142 gmp_base_ctl_step();
143
144 //
145 // Call GMP Timer
146 //
147 gmp_step_system_tick();
148
149 //
150 // Blink LED
151 //
152 if (gmp_base_get_system_tick() % 1000 < 500)
153 GPIO_WritePin(IRIS_LED1, 0);
154 else
155 GPIO_WritePin(IRIS_LED1, 1);
156
157 //
158 // Clear the interrupt flag
159 //
160 ADC_clearInterruptStatus(IRIS_ADCA_BASE, ADC_INT_NUMBER1);
161
162 //
163 // Check if overflow has occurred
164 //
165 if (true == ADC_getInterruptOverflowStatus(IRIS_ADCA_BASE, ADC_INT_NUMBER1))
166 {
167 ADC_clearInterruptOverflowStatus(IRIS_ADCA_BASE, ADC_INT_NUMBER1);

169 }
170
171 //
172 // Acknowledge the interrupt
173 //
174 Interrupt_clearACKGroup(INT_IRIS_ADCA_1_INTERRUPT_ACK_GROUP);

374interrupt void INT_IRIS_UART_USB_RX_ISR(void)
375{
376 uint32_t rxStatus;
377
378 // ---------------------------------------------------------
379 // 策略优化：使用 while 循环一次性抽干 FIFO
380 // ---------------------------------------------------------
381 // 原因：设置为 1/16 触发虽然保证了响应度，但为了减少中断频率，
382 // 我们应该在一次中断里尽可能把 FIFO 里堆积的数据全读完，
383 // 而不是读几个字节就退出去等下一次中断。
384
385// while((fifoLevel = SCI_getRxFIFOStatus(IRIS_UART_USB_BASE)) > 0)
386// {
391//
392// // 读取数据
393// SCI_readCharArray(IRIS_UART_USB_BASE, rxBuf, fifoLevel);
394//
395// // 推送数据到 AT 核心
396// at_device_rx_isr(&at_dev, (char*)rxBuf, fifoLevel);
397//
398// // 如果 FIFO 此时已经空了，while 循环会自动结束
399// }
400 at_device_flush_rx_buffer();
401
402 // ---------------------------------------------------------
403 // 错误处理优化：不要盲目复位 FIFO
404 // ---------------------------------------------------------
405 rxStatus = SCI_getRxStatus(IRIS_UART_USB_BASE);
406
407 if(rxStatus & SCI_RXSTATUS_OVERRUN)
408 {
409 // 仅处理溢出错误：清除溢出标志位，而不是复位整个 FIFO
410 // C2000 DriverLib 通常通过写入 RXFFOVRCLR 位来清除
411 // 如果没有直接API，可以使用 HWREG 操作，或者保持 resetRxFIFO 但仅针对 Overrun
412
413 // 修正建议：只在确实溢出卡死时才 Reset，普通 Error 不要 Reset
414 SCI_clearOverflowStatus(IRIS_UART_USB_BASE); // 假设有此函数，或者手动操作寄存器
415
416 // 如果必须使用 resetRxFIFO，请确保仅在严重故障下使用
417 // SCI_resetRxFIFO(IRIS_UART_USB_BASE);
418 }
419
420 if(rxStatus & SCI_RXSTATUS_ERROR)
421 {
422 // 对于 Frame Error / Parity Error (比如噪声 0xFF)
423 // 读取数据寄存器通常会自动清除这些错误标志
424 // 这里只需要做一个软件复位给 SCI 状态机（不清除 FIFO），或者单纯清除标志
425 // 绝对不要调用 SCI_resetRxFIFO() !!!
426 }
427
428 // ---------------------------------------------------------
429 // 清除中断标志 (关键)
430 // ---------------------------------------------------------
431 // 必须清除 RXFF (FIFO中断) 和 RXERR (错误中断) 标志，否则中断会卡死
432 SCI_clearInterruptStatus(IRIS_UART_USB_BASE, SCI_INT_RXFF | SCI_INT_RXERR);
433
434 // PIE ACK
435 Interrupt_clearACKGroup(INT_IRIS_UART_USB_RX_INTERRUPT_ACK_GROUP);
436}
