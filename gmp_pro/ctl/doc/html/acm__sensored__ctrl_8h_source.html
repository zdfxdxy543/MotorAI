285GMP_STATIC_INLINE void ctl_step_acm_sensored_ctrl(acm_sensored_bare_controller_t* ctrl)
286{
287 ctl_vector2_t phasor;
288 ctrl_gt etheta;
289
290 ctrl->isr_tick++;
291
292 // Ramp generator
293 ctl_step_slope_f(&ctrl->rg);
294
295 // Clarke Transformation
296#if MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 3
297 ctl_ct_clarke(&ctrl->mtr_interface.iabc->value, &ctrl->iab0);
298#elif MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 2
299 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->mtr_interface.iabc->value, (ctl_vector2_t*)&ctrl->iab0);
300 ctrl->iab0.dat[phase_0] = 0;
301#endif
302
303 // Park Transformation
304 etheta = ctl_get_mtr_elec_theta(&ctrl->mtr_interface);
305 ctl_set_phasor_via_angle(etheta, &phasor);
306 ctl_ct_park(&ctrl->iab0, &phasor, &ctrl->idq0);
307
308 // Speed calculation
309 ctl_step_spd_calc(&ctrl->spd_enc);
310
311 if (ctrl->flag_enable_controller)
312 {
313 // Flux estimation
314 if (ctrl->flag_enable_flux_est)
315 {
316 ctl_step_im_spd_calc(&ctrl->flux_calc, ctrl->idq0.dat[phase_d], ctrl->idq0.dat[phase_q],
317 ctl_get_mtr_velocity(&ctrl->mtr_interface));
318 }
319
320 // Set speed target
321 if (ctrl->flag_enable_velocity_ctrl)
322 {
323 ctrl->rg.target_frequency = ctl_mul(ctrl->flux_calc.omega_s, ctrl->speed_pu_rg_sf);
324 }
325 else
326 {
327 ctrl->rg.target_frequency = ctl_mul(ctrl->speed_set, ctrl->speed_pu_rg_sf);
328 }
329
330 // Velocity Controller
331 if (ctrl->flag_enable_velocity_ctrl)
332 {
333 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
334#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
335 ctrl->idq_set.dat[phase_q] =
336 ctl_step_tracking_pid(&ctrl->spd_ctrl, ctrl->speed_set, ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
337 ctrl->idq_ff.dat[phase_q];
338#else
339 ctrl->idq_set.dat[phase_q] = ctl_step_tracking_continuous_pid(&ctrl->spd_ctrl, ctrl->speed_set,
340 ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
341 ctrl->idq_ff.dat[phase_q];
342#endif
343 }
344 else
345 {
346 ctl_vector2_copy(&ctrl->idq_set, &ctrl->idq_ff);
347 }
348
349 // Current Controller
350 if (ctrl->flag_enable_current_ctrl)
351 {
352#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
353 ctrl->vdq_set.dat[phase_d] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_d],
354 ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
355 ctrl->vdq_ff.dat[phase_d];
356 ctrl->vdq_set.dat[phase_q] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_q],
357 ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
358 ctrl->vdq_ff.dat[phase_q];
359#else
360 ctrl->vdq_set.dat[phase_d] =
361 ctl_step_pid_ser(&ctrl->current_ctrl[phase_d], ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
362 ctrl->vdq_ff.dat[phase_d];
363 ctrl->vdq_set.dat[phase_q] =
364 ctl_step_pid_ser(&ctrl->current_ctrl[phase_q], ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
365 ctrl->vdq_ff.dat[phase_q];
366#endif
367 ctrl->vdq_set.dat[phase_0] = 0;
368 }
369 else
370 {
371 ctl_vector2_copy((ctl_vector2_t*)&ctrl->vdq_set, &ctrl->vdq_ff);
372 ctrl->vdq_set.dat[phase_0] = 0;
373 }
374
375 // Inverse Park Transformation
376 if (ctrl->flag_enable_modulation)
377 {
378 ctl_ct_ipark(&ctrl->vdq_set, &phasor, &ctrl->vab0_set);
379 }
380
381 // SVPWM
382 if (ctrl->flag_enable_output)
383 {
384 ctl_ct_svpwm_calc(&ctrl->vab0_set, &ctrl->pwm_out->value);
385 }
386 else
387 {
388 ctl_vector3_set(&ctrl->pwm_out->value, float2ctrl(0.5));
389 }
390 }
391 else
392 {
393 ctl_vector3_clear(&ctrl->pwm_out->value);
394 ctl_clear_acm_sensored_ctrl(ctrl);
395 }
396}
