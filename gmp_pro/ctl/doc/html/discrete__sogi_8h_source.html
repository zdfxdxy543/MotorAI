120GMP_STATIC_INLINE void ctl_step_discrete_sogi(discrete_sogi_t* sogi, ctrl_gt u)
121{
122 // Store current input
123 sogi->u[0] = u;
124
125 // Calculate the direct (in-phase) output y_d(k)
126 sogi->osg_u[0] = ctl_mul(sogi->b0, (sogi->u[0] - sogi->u[2])) + ctl_mul(sogi->a1, sogi->osg_u[1]) +
127 ctl_mul(sogi->a2, sogi->osg_u[2]);
128
129 // Calculate the quadrature output y_q(k)
130 sogi->osg_qu[0] = ctl_mul(sogi->qb0, sogi->u[0]) + ctl_mul(sogi->qb1, sogi->u[1]) + ctl_mul(sogi->qb2, sogi->u[2]) +
131 ctl_mul(sogi->a1, sogi->osg_qu[1]) + ctl_mul(sogi->a2, sogi->osg_qu[2]);
132
133 // Update state buffers for the next iteration
134 sogi->u[2] = sogi->u[1];
135 sogi->u[1] = sogi->u[0];
136
137 sogi->osg_u[2] = sogi->osg_u[1];
138 sogi->osg_u[1] = sogi->osg_u[0];
139
140 sogi->osg_qu[2] = sogi->osg_qu[1];
141 sogi->osg_qu[1] = sogi->osg_qu[0];
142}

243GMP_STATIC_INLINE void ctl_step_discrete_sogi_dc(discrete_sogi_dc_t* sogi_dc, ctrl_gt u)
244{
245 // 1. DC Estimation Loop (Integrator)
246 // Error = (Input_clean) - Alpha
247 // Input_clean was (Input_raw - v_dc_est)
248 // So Error = Input_raw - v_dc_est - Alpha
249 // Note: We use previous values to avoid algebraic loop in simplest implementation.
250
251 ctrl_gt alpha_prev = ctl_get_discrete_sogi_ds(&sogi_dc->core);
252
253 // Calculate the "leakage" that the SOGI core didn't catch (which is DC)
254 // Error signal driving the DC integrator
255 ctrl_gt dc_loop_err = (sogi_dc->input_prev - sogi_dc->v_dc_est) - alpha_prev;
256
257 // Update DC estimate: v_dc[k] = v_dc[k-1] + Gain * Error
258 sogi_dc->v_dc_est += ctl_mul(dc_loop_err, sogi_dc->dc_integ_gain);
259
260 // 2. Clean Input (Remove DC)
261 ctrl_gt u_clean = u - sogi_dc->v_dc_est;
262
263 // 3. Step SOGI Core
264 ctl_step_discrete_sogi(&sogi_dc->core, u_clean);
265
266 // Store current input for next step
267 sogi_dc->input_prev = u;
268}
