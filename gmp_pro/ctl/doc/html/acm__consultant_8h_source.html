87fast_gt ctl_init_im_dsn_consultant_by_experiment(im_dsn_consultant_t* acm, motor_test_data_t* noload_data,
88 motor_test_data_t* lockedrotor_data, uint16_t pole_pairs,
89 parameter_gt test_freq_Hz, parameter_gt R_dc_line_to_line)
90{
91 // --- Input Validation ---
92 if (!acm || !noload_data || !lockedrotor_data || test_freq_Hz <= 0)
93 {
94 return false;
95 }
96 if (noload_data->I_line <= 0 || lockedrotor_data->I_line <= 0)
97 {
98 return false; // Avoid division by zero
99 }
100
101 // --- Basic Parameters ---
102 acm->pole_pair = pole_pairs;
103 parameter_gt omega_e = 2.0f * M_PI * test_freq_Hz;
104
105 // --- Step 1: Stator Resistance (from DC Test) ---
106 // Assuming a Y-connected motor, phase resistance is half the line-to-line resistance.
107 acm->Rs = R_dc_line_to_line / 2.0f;
108
109 // --- Step 2: Locked-Rotor Test Analysis (to find Rr and leakage inductances) ---
110 parameter_gt Vlr_ph = lockedrotor_data->V_line / (parameter_gt)sqrtf(3.0);
111 parameter_gt Ilr_ph = lockedrotor_data->I_line;
112
113 // Total resistance and impedance from the locked-rotor test
114 parameter_gt R_lr = lockedrotor_data->P_total / (3.0f * Ilr_ph * Ilr_ph);
115 parameter_gt Z_lr = Vlr_ph / Ilr_ph;
116
117 // Calculate rotor resistance (Rr')
118 acm->Rr = R_lr - acm->Rs;
119
120 // Calculate total leakage reactance (Xlk = Xls + Xr')
121 parameter_gt X_lk = (parameter_gt)sqrtf(Z_lr * Z_lr - R_lr * R_lr);
122
123 // --- Step 3: No-Load Test Analysis (to find magnetizing inductance) ---
124 parameter_gt Vnl_ph = noload_data->V_line / (parameter_gt)sqrtf(3.0);
125 parameter_gt Inl_ph = noload_data->I_line;
126
127 // Calculate apparent and reactive power from no-load test
128 parameter_gt S_nl = 3.0f * Vnl_ph * Inl_ph;
129 parameter_gt Q_nl = (parameter_gt)sqrtf(S_nl * S_nl - noload_data->P_total * noload_data->P_total);
130
131 // The no-load reactance is approximately Xls + Xm
132 parameter_gt X_nl = Q_nl / (3.0f * Inl_ph * Inl_ph);
133
134 // --- Step 4: Separate Inductances and Finalize ---
135 // An assumption is required to split the total leakage reactance X_lk into
136 // stator (Xls) and rotor (Xr') components. A common IEEE standard assumption
137 // for NEMA Class A/B motors is to split them equally.
138 parameter_gt X_ls = X_lk / 2.0f;
139 parameter_gt X_r_prime = X_lk / 2.0f;
140
141 // Now find the magnetizing reactance
142 parameter_gt X_m = X_nl - X_ls;
143
144 // Calculate final inductance values in Henrys
145 parameter_gt L_ls = X_ls / omega_e;
146 parameter_gt L_r_prime = X_r_prime / omega_e;
147 acm->Lm = X_m / omega_e;
148
149 // Populate Ls and Lr as self-inductances (leakage + mutual)
150 acm->Ls = L_ls + acm->Lm;
151 acm->Lr = L_r_prime + acm->Lm;
152
153 // --- Mechanical Parameters ---
154 // Note: These electrical tests cannot determine mechanical parameters.
155 // They should be initialized to zero or determined by other means.
156 acm->inertia = 0.0f;
157 acm->damp = 0.0f;
158
159 return true;
160}

240GMP_STATIC_INLINE void ctl_acm_tune_current_pi_params(const im_dsn_consultant_t* im_dsn, parameter_gt bandwidth_hz,
241 parameter_gt* kp, parameter_gt* ki)
242{
243 parameter_gt alpha_c = M_2_PI * bandwidth_hz; // Convert Hz to rad/s
244 parameter_gt sigma = ctl_acm_get_leakage_coefficient(im_dsn);
245 parameter_gt transient_inductance = sigma * im_dsn->Ls;
246
247 *kp = alpha_c * transient_inductance;
248 *ki = alpha_c * im_dsn->Rs;
249}

265GMP_STATIC_INLINE void ctl_acm_tune_speed_pi_params(const im_dsn_consultant_t* im_dsn, parameter_gt bandwidth_hz,
266 parameter_gt rotor_flux_mag, parameter_gt* kp, parameter_gt* ki)
267{
268 if (im_dsn->Lr < 1e-9)
269 {
270 *kp = 0.0;
271 *ki = 0.0;
272 return;
273 }
274
275 parameter_gt alpha_s = M_2_PI * bandwidth_hz; // Convert Hz to rad/s
276
277 // Calculate torque constant Kt
278 parameter_gt kt = 1.5 * im_dsn->pole_pair * (im_dsn->Lm / im_dsn->Lr) * rotor_flux_mag;
279
280 if (kt < 1e-9)
281 {
282 *kp = 0.0;
283 *ki = 0.0;
284 return;
285 }
286
287 *kp = (alpha_s * im_dsn->inertia) / kt;
288 *ki = (alpha_s * im_dsn->damp) / kt; // Note: This depends on an accurate damping/friction parameter 'B'.
289}

307GMP_STATIC_INLINE parameter_gt ctl_consult_acm_estimate_slip(im_dsn_consultant_t* acm, parameter_gt V_phase_rms,
308 parameter_gt freq_Hz, parameter_gt T_load_Nm)
309{
310 if (!acm || freq_Hz <= 0 || V_phase_rms <= 0 || T_load_Nm < 0)
311 {
312 return -1.0; // Invalid input
313 }
314
315 // Step 1: Calculate synchronous speed and reactances at the operating frequency
316 parameter_gt omega_e = 2.0f * M_PI * freq_Hz; // Electrical angular velocity [rad/s]
317
318 // Note: Using Ls and Lr directly as stator and rotor reactance components.
319 // This is a common simplification where the magnetizing branch is moved to the terminals.
320 parameter_gt Xs = omega_e * acm->Ls; // Stator reactance
321 parameter_gt Xr = omega_e * acm->Lr; // Rotor reactance
322
323 // Step 2: Calculate the critical slip (slip at maximum torque)
324 parameter_gt X_total_sq = (Xs + Xr) * (Xs + Xr);
325 parameter_gt Rs_sq = acm->Rs * acm->Rs;
326 parameter_gt sk_denominator = (parameter_gt)sqrtf(Rs_sq + X_total_sq);
327 if (sk_denominator == 0)
328 return -1.0;
329 parameter_gt sk = acm->Rr / sk_denominator; // Critical slip (s_k)
330
331 // Step 3: Calculate the maximum (breakdown) torque
332 parameter_gt Tmax_denominator = 2.0f * omega_e * (acm->Rs + sk_denominator);
333 if (Tmax_denominator == 0)
334 return -1.0;
335 parameter_gt T_max = (3.0f * V_phase_rms * V_phase_rms) / Tmax_denominator;
336
337 // Step 4: Check if the load torque is achievable
338 if (T_load_Nm > T_max)
339 {
340 return -2.0; // Load torque exceeds breakdown torque, motor would stall.
341 }
342
343 if (T_load_Nm == 0)
344 {
345 return 0.0; // No load, no slip.
346 }
347
348 // Step 5: Solve the Kloss equation for slip 's'
349 // T_load / T_max = 2 / (s/sk + sk/s) => s^2 - (2 * T_max / T_load * sk) * s + sk^2 = 0
350 // This is a quadratic equation: a*s^2 + b*s + c = 0
351 parameter_gt a = 1.0f;
352 parameter_gt b = -(2.0f * T_max / T_load_Nm) * sk;
353 parameter_gt c = sk * sk;
354
355 // Use the quadratic formula to solve for s: s = [-b ± sqrt(b^2 - 4ac)] / 2a
356 parameter_gt discriminant = b * b - 4.0f * a * c;
357 if (discriminant < 0)
358 {
359 return -3.0; // Should not happen if T_load <= T_max, but good practice to check
360 }
361
362 // We need the smaller root for stable operation (motor region)
363 parameter_gt s = (-b - (parameter_gt)sqrtf(discriminant)) / (2.0f * a);
364
365 return s;
366}
