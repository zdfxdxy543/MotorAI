113GMP_STATIC_INLINE parameter_gt ctl_pmsm_get_Ke_Vrms_per_krpm(const ctl_pmsm_dsn_consultant_t* pmsm_dsn)
114{
115 // Ke(Vp/rads) * (rads/s / RPM) * (RPM / kRPM) * Vrms/Vp
116 // Ke(Vp/rads) * (2*pi/60) * 1000 * (1/sqrt(2)) * sqrt(3) for L-L
117 return ctl_pmsm_get_Ke_Vpeak_per_rads(pmsm_dsn) * (CTL_PARAM_CONST_PI * 100.0f / 3.0f) *
118 CTL_PARAM_CONST_SQRT3_OVER_SQRT2;
119}

135GMP_STATIC_INLINE void ctl_pmsm_tune_current_pi_params(const ctl_pmsm_dsn_consultant_t* pmsm_dsn,
136 parameter_gt bandwidth_hz, parameter_gt* kp_d,
137 parameter_gt* ki_d, parameter_gt* kp_q, parameter_gt* ki_q)

159GMP_STATIC_INLINE void ctl_pmsm_tune_speed_pi_params(const ctl_pmsm_dsn_consultant_t* pmsm_dsn,
160 parameter_gt bandwidth_hz, parameter_gt* kp, parameter_gt* ki)
161{
162 parameter_gt kt = ctl_pmsm_get_Kt(pmsm_dsn);
163 if (kt < float2ctrl(0.000001))
164 {
165 *kp = 0.0;
166 *ki = 0.0;
167 return;
168 }
169
170 parameter_gt alpha_s = CTL_PARAM_CONST_2PI * bandwidth_hz; // Convert Hz to rad/s
171
172 *kp = (alpha_s * pmsm_dsn->inertia) / kt;
173 // Note: Ki calculation depends on an accurate damping/friction parameter 'B'.
174 // If B is unknown, Ki might be set to a small value or tuned experimentally.
175 *ki = (alpha_s * pmsm_dsn->damp) / kt;
176}

203GMP_STATIC_INLINE void ctl_pmsm_convert_current_pi_to_pu(const ctl_per_unit_consultant_t* pu, parameter_gt kp_real,
204 parameter_gt ki_real, parameter_gt* kp_pu, parameter_gt* ki_pu)
205{
206 if (pu->base_impedence < float2ctrl(0.000001) || pu->base_omega < float2ctrl(0.000001))
207 {
208 *kp_pu = 0.0;
209 *ki_pu = 0.0;
210 return;

213 *ki_pu = ki_real / (pu->base_impedence * pu->base_omega);
214}
