91GMP_STATIC_INLINE ctrl_gt ctl_step_tracking_pid(ctl_tracking_discrete_pid_t* tp, ctrl_gt target, ctrl_gt fbk)
92{
93 // Check if it's time to execute the controller based on the division factor
94 if (ctl_step_divider(&tp->div))
95 {
96 // If so, first update the trajectory by one step towards the target
97 ctl_step_slope_limiter(&tp->traj, target);
98 // Then, execute the PID controller with the error between the current
99 // trajectory point and the feedback value.
100 return ctl_step_discrete_pid(&tp->pid, tp->traj.out - fbk);
101 }
102
103 // If it's not time to execute, just return the last held PID output value.
104 return ctl_get_discrete_pid_output(&tp->pid);
105}
