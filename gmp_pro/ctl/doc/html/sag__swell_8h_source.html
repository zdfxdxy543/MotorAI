103GMP_STATIC_INLINE ctl_voltage_event_status_t ctl_step_voltage_event_detector(ctl_voltage_event_detector_t* ved,
104 ctrl_gt input)
105{
106 // Step the SOGI to get the fundamental components
107 ctl_step_discrete_sogi(&ved->sogi, input);
108 ctrl_gt d_out = ctl_get_discrete_sogi_ds(&ved->sogi);
109 ctrl_gt q_out = ctl_get_discrete_sogi_qs(&ved->sogi);
110
111 // Calculate the amplitude
112 ved->amplitude = ctl_sqrt(ctl_mul(d_out, d_out) + ctl_mul(q_out, q_out));
113
114 // Do not check for new faults if one is already latched
115 if (ved->fault_status != VOLTAGE_EVENT_OK)
116 {
117 return ved->fault_status;
118 }
119
120 // Check for swell condition
121 if (ved->amplitude > ved->swell_threshold)
122 {
123 ved->swell_counter++;
124 if (ved->swell_counter >= ved->swell_delay_samples)
125 {
126 ved->fault_status = VOLTAGE_EVENT_SWELL;
127 }
128 }
129 else
130 {
131 ved->swell_counter = 0;
132 }
133
134 // Check for sag condition
135 if (ved->amplitude < ved->sag_threshold)
136 {
137 ved->sag_counter++;
138 if (ved->sag_counter >= ved->sag_delay_samples)
139 {
140 ved->fault_status = VOLTAGE_EVENT_SAG;
141 }
142 }
143 else
144 {
145 ved->sag_counter = 0;
146 }
147
148 return ved->fault_status;
149}
