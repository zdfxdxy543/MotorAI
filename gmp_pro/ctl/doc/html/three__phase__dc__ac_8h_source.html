242GMP_STATIC_INLINE void ctl_step_inv_ctrl(inv_ctrl_t* ctrl)
243{
244 // loop variables
245 size_gt i;
246
247 // assert critical pointer
248 gmp_base_assert(ctrl);
249 gmp_base_assert(ctrl->pwm_out);
250 gmp_base_assert(ctrl->adc_iabc);
251 gmp_base_assert(ctrl->adc_vabc);
252 gmp_base_assert(ctrl->adc_iuvw);
253 gmp_base_assert(ctrl->adc_vuvw);
254
255 ctrl->isr_tick += 1;
256
257 // --- 1. Input Filtering and Coordinate Transformation ---
258 ctl_step_lowpass_filter(&ctrl->lpf_udc, ctrl->adc_udc->value);
259 ctl_step_lowpass_filter(&ctrl->lpf_idc, ctrl->adc_idc->value);
260
261#if CURRENT_SAMPLE_PHASE_MODE == 3
262 for (i = 0; i < 3; ++i)
263 ctrl->iabc.dat[i] = ctl_step_lowpass_filter(&ctrl->lpf_iabc[i], ctrl->adc_iabc->value.dat[i]);
264 ctl_ct_clarke(&ctrl->iabc, &ctrl->iab0);
265#elif CURRENT_SAMPLE_PHASE_MODE == 2
266 for (i = 0; i < 2; ++i)
267 ctrl->iabc.dat[i] = ctl_step_lowpass_filter(&ctrl->lpf_iabc[i], ctrl->adc_iabc->value.dat[i]);
268 ctrl->iabc.dat[phase_C] = 0;
269 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->iabc, (ctl_vector2_t*)&ctrl->iab0);
270 ctrl->iab0.dat[phase_0] = 0;
271#endif
272
273#if VOLTAGE_SAMPLE_PHASE_MODE == 3
274 for (size_gt i = 0; i < 3; ++i)
275 ctrl->vabc.dat[i] = ctl_step_lowpass_filter(&ctrl->lpf_vabc[i], ctrl->adc_vabc->value.dat[i]);
276 ctl_ct_clarke(&ctrl->vabc, &ctrl->vab0);
277#elif VOLTAGE_SAMPLE_PHASE_MODE == 2
278 for (size_gt i = 0; i < 2; ++i)
279 ctrl->vabc.dat[i] = ctl_step_lowpass_filter(&ctrl->lpf_vabc[i], ctrl->adc_vabc->value.dat[i]);
280 ctrl->vabc.dat[phase_C] = 0;
281 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->vabc, (ctl_vector2_t*)&ctrl->vab0);
282 ctrl->vab0.dat[phase_0] = 0;
283#elif VOLTAGE_SAMPLE_PHASE_MODE == 1
284 for (i = 0; i < 2; ++i)
285 ctrl->vabc.dat[i] = ctl_step_lowpass_filter(&ctrl->lpf_vabc[i], ctrl->adc_vabc->value.dat[i]);
286 ctrl->vabc.dat[phase_C] = 0;
287 ctl_ct_clarke_from_line((ctl_vector2_t*)&ctrl->vabc, (ctl_vector2_t*)&ctrl->vab0);
288 ctrl->vab0.dat[phase_0] = 0;
289#endif
290
291 // --- 2. Grid Synchronization (PLL) ---
292 if (ctrl->flag_enable_pll)
293 {
294 ctl_step_pll_3ph(&ctrl->pll, ctrl->vab0.dat[phase_alpha], ctrl->vab0.dat[phase_beta]);
295 }
296
297 // --- 3. Main Control Logic ---
298 if (ctrl->flag_enable_system)
299 {
300 // --- 3a. Angle and Phasor Generation ---
301 if (ctrl->flag_enable_freerun)
302 {
303 // CHECK HERE!
304 ctl_set_ramp_generator_slope(&ctrl->rg, ctl_mul(ctrl->rg_slope_default, ctrl->rg_freq_pu));
305 ctrl->angle = ctl_step_ramp_generator(&ctrl->rg);
306 ctl_set_phasor_via_angle(ctrl->angle, &ctrl->phasor);
307 }
308 else
309 {
310 ctrl->angle = ctrl->pll.theta;
311 //ctrl->phasor = ctrl->pll.phasor;
312 ctl_vector2_copy(&ctrl->phasor, &ctrl->pll.phasor);
313 }
314
315 // --- 3b. Park Transformation ---
316 ctl_ct_park2((ctl_vector2_t*)&ctrl->iab0, &ctrl->phasor, &ctrl->idq);
317 ctl_ct_park2((ctl_vector2_t*)&ctrl->vab0, &ctrl->phasor, &ctrl->vdq);
318
319 // --- 3c. Droop Control ---
320 if (ctrl->flag_enable_droop)
321 {
322 ctrl->idq_droop_set.dat[phase_d] = ctl_step_saturation(
323 &ctrl->idq_droop_sat[phase_d], ctl_mul(ctrl->kq_droop, ctrl->v_droop_ref - ctrl->vdq.dat[phase_d]));
324 ctrl->idq_droop_set.dat[phase_q] = ctl_step_saturation(
325 &ctrl->idq_droop_sat[phase_q], ctl_mul(ctrl->kp_droop, float2ctrl(1.0f) - ctrl->pll.freq_pu));
326 }
327 else
328 {
329 ctl_vector2_clear(&ctrl->idq_droop_set);
330 }
331
332 // --- 3d. Outer Voltage Loop ---
333 if (ctrl->flag_enable_voltage_ctrl)
334 {
335 ctrl->idq_set.dat[phase_d] =
336 ctl_step_pid_ser(&ctrl->voltage_ctrl[phase_d], ctrl->vdq_set.dat[phase_d] - ctrl->vdq.dat[phase_d]);
337 ctrl->idq_set.dat[phase_q] =
338 ctl_step_pid_ser(&ctrl->voltage_ctrl[phase_q], ctrl->vdq_set.dat[phase_q] - ctrl->vdq.dat[phase_q]);
339 }
340
341 // --- 3e. Inner Positive-Sequence Current Loop ---
342 if (ctrl->flag_enable_current_ctrl)
343 {
344 ctrl_gt id_ref = ctrl->idq_ff.dat[phase_d] + ctrl->idq_set.dat[phase_d] + ctrl->idq_droop_set.dat[phase_d];
345 ctrl_gt iq_ref = ctrl->idq_ff.dat[phase_q] + ctrl->idq_set.dat[phase_q] + ctrl->idq_droop_set.dat[phase_q];
346
347 ctrl->vdq_pos_out.dat[phase_d] =
348 ctl_step_pid_ser(&ctrl->current_ctrl[phase_d], id_ref - ctrl->idq.dat[phase_d]);
349 ctrl->vdq_pos_out.dat[phase_q] =
350 ctl_step_pid_ser(&ctrl->current_ctrl[phase_q], iq_ref - ctrl->idq.dat[phase_q]);
351
352 if (ctrl->flag_enable_decouple_lf)
353 {
354 ctrl->vdq_pos_out.dat[phase_d] -= ctl_mul(ctrl->omega_L, ctrl->idq.dat[phase_q]);
355 // CORRECTED: q-axis voltage is decoupled using d-axis current.
356 ctrl->vdq_pos_out.dat[phase_q] += ctl_mul(ctrl->omega_L, ctrl->idq.dat[phase_d]);
357 }
358 }
359 ctl_ct_ipark2(&ctrl->vdq_pos_out, &ctrl->phasor, &ctrl->vab_pos);
360
361 // --- 3f. Negative-Sequence Voltage/Current Loop ---
362 if (ctrl->flag_enable_negative_voltage_ctrl)
363 {
364 ctl_ct_park2_neg((ctl_vector2_t*)&ctrl->vab0, &ctrl->phasor, &ctrl->vdq_neg);
365 ctrl->idq_net_set.dat[phase_d] =
366 ctl_step_pid_ser(&ctrl->neg_voltage_ctrl[phase_d], -ctrl->vdq_neg.dat[phase_d]);
367 ctrl->idq_net_set.dat[phase_q] =
368 ctl_step_pid_ser(&ctrl->neg_voltage_ctrl[phase_q], -ctrl->vdq_neg.dat[phase_q]);
369 }
370 else
371 {
372 ctl_vector2_clear(&ctrl->idq_net_set);
373 }
374
375 if (ctrl->flag_enable_negative_current_ctrl)
376 {
377 ctl_ct_park2_neg((ctl_vector2_t*)&ctrl->iab0, &ctrl->phasor, &ctrl->idq_neg);
378 ctrl->vdq_neg_out.dat[phase_d] = ctl_step_pid_ser(
379 &ctrl->neg_current_ctrl[phase_d], ctrl->idq_net_set.dat[phase_d] - ctrl->idq_neg.dat[phase_d]);
380 ctrl->vdq_neg_out.dat[phase_q] = ctl_step_pid_ser(
381 &ctrl->neg_current_ctrl[phase_q], ctrl->idq_net_set.dat[phase_q] - ctrl->idq_neg.dat[phase_q]);
382 ctl_ct_ipark2_neg(&ctrl->vdq_neg_out, &ctrl->phasor, &ctrl->vab_neg);
383 }
384 else
385 {
386 ctl_vector2_clear(&ctrl->vab_neg);
387 }
388
389 // --- 3g. Harmonic Compensation ---
390 if (ctrl->flag_enable_harm_ctrl)
391 {
392 for (i = 0; i < 2; ++i)
393 {
394 ctrl->vab_harm.dat[i] = ctl_step_qr_controller(&ctrl->harm_qr_5[i], -ctrl->iab0.dat[i]);
395 ctrl->vab_harm.dat[i] += ctl_step_qr_controller(&ctrl->harm_qr_7[i], -ctrl->iab0.dat[i]);
396 }
397 }
398 else
399 {
400 ctl_vector2_clear(&ctrl->vab_harm);
401 }
402
403 // --- 3h. Zero-Sequence Control ---
404 if (ctrl->flag_enable_zero_current_ctrl)
405 {
406 ctrl->v0_set = ctl_step_pid_ser(&ctrl->zero_pid, -ctrl->iab0.dat[phase_0]);
407 ctrl->v0_set += ctl_step_qr_controller(&ctrl->zero_qr_3, -ctrl->iab0.dat[phase_0]);
408 ctrl->v0_set += ctl_step_qr_controller(&ctrl->zero_qr_9, -ctrl->iab0.dat[phase_0]);
409 }
410 else
411 {
412 ctrl->v0_set = 0;
413 }
414
415 // --- 3i. Final Modulation Signal Synthesis ---
416 ctrl->vab_out.dat[0] = ctrl->vab_pos.dat[0] + ctrl->vab_neg.dat[0] + ctrl->vab_harm.dat[0];
417 ctrl->vab_out.dat[1] = ctrl->vab_pos.dat[1] + ctrl->vab_neg.dat[1] + ctrl->vab_harm.dat[1];
418 ctl_ct_iclarke2(&ctrl->vab_out, &ctrl->abc_out);
419 ctrl->abc_out.dat[phase_A] += ctrl->v0_set;
420 ctrl->abc_out.dat[phase_B] += ctrl->v0_set;
421 ctrl->abc_out.dat[phase_C] += ctrl->v0_set;
422
423 // --- 4. SVPWM or SPWM Output ---
424 ctrl->pwm_out->value.dat[phase_A] = ctl_div2(ctl_add(ctrl->abc_out.dat[phase_A], float2ctrl(1)));
425 ctrl->pwm_out->value.dat[phase_B] = ctl_div2(ctl_add(ctrl->abc_out.dat[phase_B], float2ctrl(1)));
426 ctrl->pwm_out->value.dat[phase_C] = ctl_div2(ctl_add(ctrl->abc_out.dat[phase_C], float2ctrl(1)));
427 }
428 else
429 {
430 // Disable output if system is not enabled
431 ctl_vector3_set(&ctrl->pwm_out->value, float2ctrl(0.5));
432 }
433}
