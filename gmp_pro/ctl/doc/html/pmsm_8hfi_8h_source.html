134GMP_STATIC_INLINE ctrl_gt ctl_step_pmsm_hfi(pmsm_hfi_t* hfi, ctrl_gt iq)
135{
136 // 1. Generate the high-frequency carrier signal and the injection voltage.
137 hfi->ud_inj = ctl_mul(ctl_get_sine_generator_cos(&hfi->hfi_sincos_gen), hfi->hfi_inj_amp);
138
139 // 2. Demodulate the q-axis current using the carrier signal.
140 hfi->iq_demodulate = ctl_mul(iq, ctl_get_sine_generator_sin(&hfi->hfi_sincos_gen));
141
142 // 3. Advance the high-frequency signal generator for the next cycle.
143 ctl_step_sine_generator(&hfi->hfi_sincos_gen);
144
145 // 4. Low-pass filter the demodulated signal to extract the position error.
146 hfi->theta_error = ctl_step_biquad_filter(&hfi->iq_lp_filter, hfi->iq_demodulate);
147
148 // 5. Process the error signal through the PLL to track speed and position.
149 // 5a. PI controller generates the estimated speed in rad/tick.
150 hfi->wr_est = ctl_step_pid_par(&hfi->pid_pll, hfi->theta_error);
151
152 // 5b. Integrate the mechanical speed to get the mechanical angle.
153 hfi->theta_r_est += ctl_mul(hfi->wr_est, CTL_CTRL_CONST_1_OVER_2PI);
154 hfi->theta_r_est = ctrl_mod_1(hfi->theta_r_est); // Wrap angle to [0, 1.0)
155
156 // 6. Update the output interfaces.
157 // 6a. Convert estimated speed to p.u. for the velocity interface.
158 hfi->spdif.speed = ctl_mul(hfi->wr_est, hfi->spd_sf);
159
160 // 6b. Provide the estimated mechanical and electrical angles.
161 hfi->encif.position = hfi->theta_r_est;
162 hfi->encif.elec_position = ctrl_mod_1(ctl_mul(hfi->theta_r_est, hfi->pole_pairs));
163
164 // 7. Return the injection voltage to be used by the main controller.
165 return hfi->ud_inj;
166}
