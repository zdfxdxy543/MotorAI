105GMP_STATIC_INLINE ctl_trip_status_t ctl_step_trip_protector(ctl_trip_protector_t* prot)
106{
107 // If a fault is already latched, do nothing further.
108 if (prot->fault_status != TRIP_STATUS_OK)
109 {
110 return prot->fault_status;
111 }
112
113 // Use the absolute value for overcurrent checks.
114 ctrl_gt current_abs = ctl_abs(*prot->source);
115
116 // 1. Check for Instantaneous Trip
117 if (current_abs >= prot->level_inst)
118 {
119 prot->fault_status = TRIP_STATUS_INST_TRIP;
120 return prot->fault_status;
121 }
122
123 // 2. Check for Short-Time Delay Trip
124 if (current_abs >= prot->level_std)
125 {
126 prot->std_counter++;
127 if (prot->std_counter >= prot->delay_std_samples)
128 {
129 prot->fault_status = TRIP_STATUS_STD_TRIP;
130 }
131 }
132 else
133 {
134 prot->std_counter = 0; // Reset counter if current drops below level
135 }
136
137 // 3. Check for Long-Time Delay Trip
138 if (current_abs >= prot->level_ltd)
139 {
140 prot->ltd_counter++;
141 if (prot->ltd_counter >= prot->delay_ltd_samples)
142 {
143 prot->fault_status = TRIP_STATUS_LTD_TRIP;
144 }
145 }
146 else
147 {
148 prot->ltd_counter = 0; // Reset counter if current drops below level
149 }
150
151 return prot->fault_status;
152}
