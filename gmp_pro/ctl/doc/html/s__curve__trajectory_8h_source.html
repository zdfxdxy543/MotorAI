125GMP_STATIC_INLINE void _ctl_s_curve_replan(ctl_s_curve_planner_t* planner)
126{
127 // Determine direction
128 ctrl_gt delta_v = planner->target_vel - planner->current_vel;
129 planner->accel_dir = (delta_v > 0.0f) ? 1.0f : -1.0f;
130
131 // Time to ramp acceleration to max_accel
132 ctrl_gt t_jerk = planner->max_accel / planner->max_jerk;
133 // Velocity change during acceleration ramp-up
134 ctrl_gt v_jerk = planner->max_jerk * t_jerk * t_jerk;
135
136 if (fabsf(delta_v) > v_jerk)
137 {
138 // 7-segment profile (trapezoidal acceleration)
139 ctrl_gt v_const_accel = fabsf(delta_v) - v_jerk;
140 planner->const_accel_steps = (uint32_t)(v_const_accel / planner->max_accel);
141 }
142 else
143 {
144 // 5-segment profile (triangular acceleration)
145 planner->const_accel_steps = 0;
146 t_jerk = sqrtf(fabsf(delta_v) / planner->max_jerk);
147 }
148
149 planner->accel_target = planner->max_accel * planner->accel_dir;
150 planner->step_counter = 0;
151 planner->state = SCURVE_STATE_ACCEL_RAMP;
152 planner->last_target_vel = planner->target_vel;
153}

163GMP_STATIC_INLINE ctrl_gt ctl_step_s_curve(ctl_s_curve_planner_t* planner)
164{
165 // Check if the target has changed, requiring a new plan
166 if (planner->target_vel != planner->last_target_vel)
167 {
168 _ctl_s_curve_replan(planner);
169 }
170
171 ctrl_gt jerk = 0.0f;
172
173 switch (planner->state)
174 {
175 case SCURVE_STATE_IDLE:
176 // Do nothing, hold current velocity
177 break;
178
179 case SCURVE_STATE_ACCEL_RAMP:
180 jerk = planner->max_jerk * planner->accel_dir;
181 if (fabsf(planner->current_accel) >= fabsf(planner->accel_target))
182 {
183 planner->current_accel = planner->accel_target;
184 planner->state = SCURVE_STATE_CONST_ACCEL;
185 }
186 break;
187
188 case SCURVE_STATE_CONST_ACCEL:
189 jerk = 0.0f;
190 planner->step_counter++;
191 if (planner->step_counter >= planner->const_accel_steps)
192 {
193 planner->accel_target = 0.0f; // Target zero acceleration
194 planner->state = SCURVE_STATE_DECEL_RAMP;
195 }
196 break;
197
198 case SCURVE_STATE_DECEL_RAMP:
199 jerk = -planner->max_jerk * planner->accel_dir;
200 // Check if we are approaching the target velocity
201 if (fabsf(planner->target_vel - planner->current_vel) < fabsf(planner->current_accel))
202 {
203 planner->state = SCURVE_STATE_STOP;
204 }
205 break;
206
207 case SCURVE_STATE_STOP:
208 // Ramp acceleration to zero
209 if (fabsf(planner->current_accel) < planner->max_jerk)
210 {
211 planner->current_accel = 0.0f;
212 planner->current_vel = planner->target_vel; // Snap to final value
213 planner->state = SCURVE_STATE_IDLE;
214 }
215 else
216 {
217 jerk = (planner->current_accel > 0.0f) ? -planner->max_jerk : planner->max_jerk;
218 }
219 break;
220 }
221
222 // Integrate jerk to get acceleration, then acceleration to get velocity
223 if (planner->state != SCURVE_STATE_IDLE)
224 {
225 planner->current_accel += jerk;
226 planner->current_vel += planner->current_accel;
227 }
228
229 return planner->current_vel;
230}
