90GMP_STATIC_INLINE fast_gt dsa_step_trigger(basic_trigger_t* trigger, ctrl_gt monitor)
91{
92 // If already triggered, continue recording.
93 if (trigger->flag_triggered)
94 {
95 trigger->target_index += 1;
96
97 // Check if the recording sequence is complete.
98 if (trigger->target_index > trigger->cell_size)
99 {
100 // Reset the trigger
101 trigger->flag_triggered = 0;
102 trigger->target_index = 0;
103 trigger->last_data = monitor;
104 }
105 }
106 // If not triggered, check for the trigger condition.
107 else
108 {
109 // Trigger condition: negative-to-positive zero crossing.
110 if (trigger->last_data < 0 && monitor >= 0)
111 {
112 trigger->flag_triggered = 1;
113 }
114
115 trigger->last_data = monitor;
116 }
117
118 return trigger->flag_triggered;
119}
