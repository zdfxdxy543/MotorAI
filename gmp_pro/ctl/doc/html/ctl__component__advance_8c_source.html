20void ctl_init_fuzzy_pid(ctl_fuzzy_pid_t* fp, parameter_gt base_kp, parameter_gt base_ti, parameter_gt base_td,
21 ctrl_gt sat_max, ctrl_gt sat_min, parameter_gt e_q_factor, parameter_gt ec_q_factor,
22 ctl_lut2d_t d_kp_lut, ctl_lut2d_t d_ki_lut, ctl_lut2d_t d_kd_lut, parameter_gt fs)
23{
24 // 1. Store the base parameters and settings
25 fp->base_kp = base_kp;
26 fp->base_ti = base_ti;
27 fp->base_td = base_td;
28 fp->e_q_factor = e_q_factor;
29 fp->ec_q_factor = ec_q_factor;
30 fp->fs = fs;
31
32 // 2. Assign the look-up table structures.
33 // Note: This performs a shallow copy. The user must ensure the LUT data
34 // (pointed to by axis and surface pointers) remains valid for the lifetime
35 // of the fuzzy PID controller.
36 fp->d_kp_lut = d_kp_lut;
37 fp->d_ki_lut = d_ki_lut;
38 fp->d_kd_lut = d_kd_lut;
39
40 // 3. Initialize the underlying standard PID controller with the base parameters.
41 // The step function will dynamically update the kp, ki, kd coefficients,
42 // but we initialize them here to their starting values.
43 ctl_init_pid_Tmode(&fp->pid, base_kp, base_ti, base_td, fs);
44
45 // 4. Set the output and integrator saturation limits for the PID controller.
46 fp->pid.out_max = sat_max;
47 fp->pid.out_min = sat_min;
48 // It's good practice to also limit the integrator to prevent wind-up.
49 // Here, we tie the integrator limits to the output limits.
50 fp->pid.integral_max = sat_max;
51 fp->pid.integral_min = sat_min;
52
53 // 5. Clear all state variables to ensure a clean start.
54 ctl_clear_fuzzy_pid(fp);
55}

147int ctl_init_imc(ctl_imc_controller_t* imc, const ctl_imc_init_t* init)
148{
149 ctrl_gt Ts = 1.0f / (ctrl_gt)init->f_ctrl;
150
151 // --- Discretize Plant Model (ZOH) ---
152 imc->a_p_d = expf(-(Ts / (ctrl_gt)init->tau_p));
153 imc->b_p_d = (ctrl_gt)init->K_p * (1.0f - imc->a_p_d);
154
155 // --- Calculate Dead Time ---
156 imc->dead_time_samples = (uint16_t)roundf((ctrl_gt)init->theta_p / Ts);
157 if (imc->dead_time_samples >= IMC_MAX_DEAD_TIME_SAMPLES)
158 {
159 return -1; // Error: Dead time exceeds buffer size
160 }
161
162 // --- Discretize IMC Controller Q(s) (Tustin's method) ---
163 // Q(s) = (1/Kp) * (tau_p*s + 1) / (lambda*s + 1)
164 ctrl_gt lambda = (ctrl_gt)init->lambda;
165 ctrl_gt tau_p = (ctrl_gt)init->tau_p;
166 ctrl_gt K_p = (ctrl_gt)init->K_p;
167
168 imc->a_q_d = (2.0f * lambda - Ts) / (2.0f * lambda + Ts);
169 imc->b0_q_d = (1.0f / K_p) * (2.0f * tau_p + Ts) / (2.0f * lambda + Ts);
170 imc->b1_q_d = (1.0f / K_p) * (Ts - 2.0f * tau_p) / (2.0f * lambda + Ts);
171
172 // Reset states
173 ctl_clear_imc(imc);
174 return 0;
175}

182fast_gt ctl_init_lms_filter(ctl_lms_filter_t* lms, uint32_t order, parameter_gt mu)
183{
184 uint32_t i;
185
186 lms->order = order;
187 lms->mu = float2ctrl(mu);
188 lms->output = 0.0f;
189 lms->error = 0.0f;
190 lms->buffer_index = 0;
191
192 // --- Key Point Analysis 1: Dynamic Memory Allocation ---
193 // The filter's memory requirement depends on its order. Dynamic allocation
194 // provides flexibility. Ensure the MCU has enough heap space.
195 lms->weights = (ctrl_gt*)malloc(order * sizeof(ctrl_gt));
196 lms->buffer = (ctrl_gt*)malloc(order * sizeof(ctrl_gt));
197
198 if (lms->weights == NULL || lms->buffer == NULL)
199 {
200 // Free any partially allocated memory before returning
201 if (lms->weights)
202 free(lms->weights);
203 if (lms->buffer)
204 free(lms->buffer);
205 return 0; // Memory allocation failed
206 }
207
208 // Initialize weights and buffer to zero
209 for (i = 0; i < order; ++i)
210 {
211 lms->weights[i] = 0.0f;
212 lms->buffer[i] = 0.0f;
213 }
214
215 return 1; // Success
216}

236void ctl_init_mrac(ctl_mrac_controller_t* mrac, const ctl_mrac_init_t* init)
237{
238 ctrl_gt Ts = 1.0f / (ctrl_gt)init->f_ctrl;
239
240 // Discretize the reference model using Zero-Order Hold (ZOH)
241 mrac->a_m_d = expf(-(ctrl_gt)init->a_m * Ts);
242 mrac->b_m_d = (ctrl_gt)init->b_m / (ctrl_gt)init->a_m * (1.0f - mrac->a_m_d);
243
244 // Discretize the adaptation rates
245 mrac->gamma_r_d = (ctrl_gt)init->gamma_r * Ts;
246 mrac->gamma_y_d = (ctrl_gt)init->gamma_y * Ts;
247
248 // Reset states
249 ctl_clear_mrac(mrac);
250}

296fast_gt ctl_init_sinc_interpolator(ctl_sinc_interpolator_t* sinc, uint32_t num_taps, uint32_t table_size)
297{
298 uint32_t i, j;
299
300 sinc->num_taps = num_taps;
301 sinc->table_size = table_size;
302 sinc->buffer_index = 0;
303 sinc->output = 0.0f;
304
305 // Allocate memory for the data buffer
306 sinc->buffer = (ctrl_gt*)malloc(num_taps * sizeof(ctrl_gt));
307 if (sinc->buffer == NULL)
308 return 0;
309
310 // Allocate memory for the look-up table (array of pointers)
311 sinc->sinc_table = (ctrl_gt**)malloc(table_size * sizeof(ctrl_gt*));
312 if (sinc->sinc_table == NULL)
313 {
314 free(sinc->buffer);
315 return 0;
316 }
317
318 // Allocate memory for each row of the table
319 for (i = 0; i < table_size; i++)
320 {
321 sinc->sinc_table[i] = (ctrl_gt*)malloc(num_taps * sizeof(ctrl_gt));
322 if (sinc->sinc_table[i] == NULL)
323 {
324 // Clean up on failure
325 for (j = 0; j < i; j++)
326 free(sinc->sinc_table[j]);
327 free(sinc->sinc_table);
328 free(sinc->buffer);
329 return 0;
330 }
331 }
332
333 // --- Key Point Analysis 1: Pre-calculate the Windowed-Sinc Coefficient Table ---
334 // This is the core of the module. By pre-calculating all potentially needed FIR
335 // filter coefficients at once, it avoids expensive sin/cos calculations in the real-time loop.
336 for (i = 0; i < table_size; i++)
337 {
338 // 'fractional_offset' represents the sub-sample offset (0.0 to 1.0) for the filter currently being calculated.
339 float fractional_offset = (float)i / table_size;
340
341 for (j = 0; j < num_taps; j++)
342 {
343 // 't' is the time-axis variable for the Sinc function, shifted and centered.
344 float t = (float)j - (float)(num_taps - 1) / 2.0f - fractional_offset;
345
346 // Calculate the Sinc function value: sin(pi*t) / (pi*t)
347 float sinc_val;
348 if (t == 0.0f)
349 {
350 sinc_val = 1.0f;
351 }
352 else
353 {
354 sinc_val = sinf(CTL_PARAM_CONST_PI * t) / (CTL_PARAM_CONST_PI * t);
355 }
356
357 // Calculate the Blackman window value to smooth the truncation effects of the Sinc function
358 // and reduce Gibbs phenomenon in the frequency domain.
359 float window_val = 0.42f - 0.5f * cosf(2.0f * CTL_PARAM_CONST_PI * j / (num_taps - 1)) +
360 0.08f * cosf(4.0f * CTL_PARAM_CONST_PI * j / (num_taps - 1));
361
362 // The final filter coefficient is the product of the Sinc value and the window value.
363 sinc->sinc_table[i][j] = (ctrl_gt)(sinc_val * window_val);
364 }
365 }
366
367 // Initialize the data buffer
368 ctl_clear_sinc_interpolator(sinc);
369 return 1;
370}
