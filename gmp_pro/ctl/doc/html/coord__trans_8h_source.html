291{

337GMP_STATIC_INLINE void ctl_ct_svpwm_calc(const ctl_vector3_t* ab0, GMP_CTL_OUTPUT_TAG ctl_vector3_t* Tabc)
338{
339 ctrl_gt Ua, Ub, Uc, Umax, Umin, Ucom;
340 ctrl_gt Ualpha_tmp = -ctl_div2(ab0->dat[phase_alpha]);
341 ctrl_gt Ubeta_tmp = ctl_mul(ab0->dat[phase_beta], CTL_CTRL_CONST_SQRT_3_OVER_2);
342
343 Ua = ab0->dat[phase_alpha];
344 Ub = Ualpha_tmp + Ubeta_tmp;
345 Uc = Ualpha_tmp - Ubeta_tmp;
346
347 if (Ua > Ub)
348 {
349 Umax = Ua;
350 Umin = Ub;
351 }
352 else
353 {
354 Umax = Ub;
355 Umin = Ua;
356 }
357 if (Uc > Umax)
358 Umax = Uc;
359 else if (Uc < Umin)
360 Umin = Uc;
361
362 Ucom = ctl_div2(Umax + Umin);
363 Tabc->dat[phase_A] = Ua - Ucom + CTL_CTRL_CONST_1_OVER_2;
364 Tabc->dat[phase_B] = Ub - Ucom + CTL_CTRL_CONST_1_OVER_2;
365 Tabc->dat[phase_C] = Uc - Ucom + CTL_CTRL_CONST_1_OVER_2;

412GMP_STATIC_INLINE void ctl_ct_svpwm_calc_theorem(const ctl_vector3_t* ab0, GMP_CTL_OUTPUT_TAG ctl_vector3_t* Tabc)
413{
414 ctrl_gt X, Y, Z, T1, T2, Ta, Tb, Tc;
415 uint16_t N;
416 ctrl_gt Uabc[3] = {0};
417
418 Uabc[0] = ab0->dat[phase_beta];
419 Uabc[1] = ctl_mul(CTL_CTRL_CONST_SQRT_3_OVER_2, ab0->dat[phase_alpha]) - ctl_div2(ab0->dat[phase_beta]);
420 Uabc[2] = -ctl_mul(CTL_CTRL_CONST_SQRT_3_OVER_2, ab0->dat[phase_alpha]) - ctl_div2(ab0->dat[phase_beta]);
421
422 N = ((Uabc[0] > 0)) + ((Uabc[1] > 0) << 1) + ((Uabc[2] > 0) << 2);
423 X = ctl_mul(CTL_CTRL_CONST_SQRT_3, ab0->dat[phase_beta]);
424 Y = ctl_mul(CTL_CTRL_CONST_3_OVER_2, ab0->dat[phase_alpha]) +
425 ctl_mul(CTL_CTRL_CONST_SQRT_3_OVER_2, ab0->dat[phase_beta]);
426 Z = -ctl_mul(CTL_CTRL_CONST_3_OVER_2, ab0->dat[phase_alpha]) +
427 ctl_mul(CTL_CTRL_CONST_SQRT_3_OVER_2, ab0->dat[phase_beta]);
428
429 switch (N)
430 {
431 case 1:
432 T1 = Z;
433 T2 = Y;
434 break;
435 case 2:
436 T1 = Y;
437 T2 = -X;
438 break;
439 case 3:
440 T1 = -Z;
441 T2 = X;
442 break;
443 case 4:
444 T1 = -X;
445 T2 = Z;
446 break;
447 case 5:
448 T1 = X;
449 T2 = -Y;
450 break;
451 case 6:
452 T1 = -Y;
453 T2 = -Z;
454 break;
455 default:
456 T1 = 0;
457 T2 = 0;
458 break;
459 }
460
461 if ((T1 + T2) > CTL_CTRL_CONST_1)
462 {
463 T1 = ctl_div(T1, (T1 + T2));
464 T2 = CTL_CTRL_CONST_1 - T1;
465 }
466
467 Ta = ctl_div4(CTL_CTRL_CONST_1 - T1 - T2);
468 Tb = Ta + ctl_div2(T1);
469 Tc = Tb + ctl_div2(T2);
470 Ta = -ctl_mul2(Ta) + CTL_CTRL_CONST_1_OVER_2;
471 Tb = -ctl_mul2(Tb) + CTL_CTRL_CONST_1_OVER_2;
472 Tc = -ctl_mul2(Tc) + CTL_CTRL_CONST_1_OVER_2;
473
474 switch (N)
475 {
476 case 1:
477 Tabc->dat[phase_A] = Tb;
478 Tabc->dat[phase_B] = Ta;
479 Tabc->dat[phase_C] = Tc;
480 break;
481 case 2:
482 Tabc->dat[phase_A] = Ta;
483 Tabc->dat[phase_B] = Tc;
484 Tabc->dat[phase_C] = Tb;
485 break;
486 case 3:
487 Tabc->dat[phase_A] = Ta;
488 Tabc->dat[phase_B] = Tb;
489 Tabc->dat[phase_C] = Tc;
490 break;
491 case 4:
492 Tabc->dat[phase_A] = Tc;
493 Tabc->dat[phase_B] = Tb;
494 Tabc->dat[phase_C] = Ta;
495 break;
496 case 5:
497 Tabc->dat[phase_A] = Tc;
498 Tabc->dat[phase_B] = Ta;
499 Tabc->dat[phase_C] = Tb;
500 break;
501 case 6:
502 Tabc->dat[phase_A] = Tb;
503 Tabc->dat[phase_B] = Tc;
504 Tabc->dat[phase_C] = Ta;
505 break;
506 default:
507 ctl_vector3_clear(Tabc);
508 break;
509 }
