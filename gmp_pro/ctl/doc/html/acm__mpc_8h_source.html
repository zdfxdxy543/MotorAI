129GMP_STATIC_INLINE void ctl_step_acm_mpc(ctl_acm_mpc_controller_t* mpc, ctl_vector2_t* idq_ref,
130 ctl_vector2_t* is_ab_fbk, ctrl_gt omega_r, ctrl_gt _udc)
131{
132 fast_gt i;
133
134 ctl_vector2_t is_ab_ref, is_pred, psi_r_pred;
135 ctl_vector2_t J_psi_r_est = {{mpc->psi_r_est.dat[1], -mpc->psi_r_est.dat[0]}}; // Rotated by -90 deg
136 ctrl_gt min_cost = 1e12f;
137 uint8_t best_vector_idx = 0;
138 ctrl_gt voltage_scale = (2.0f / 3.0f) * _udc;
139 ctrl_gt omega_e_rotor = omega_r * mpc->pole_pairs;
140
141 // 1. Transform current reference from d-q frame to stationary alpha-beta frame
142 // The d-q frame is aligned with the estimated rotor flux from the previous step.
143 ctl_vector2_t flux_phasor;
144 ctl_vector2_normalize(&flux_phasor, &mpc->psi_r_est);
145 ctl_ct_ipark2(idq_ref, &flux_phasor, &is_ab_ref);
146
147 // 2. Iterate through all 8 possible voltage vectors
148 for (i = 0; i < 8; ++i)
149 {
150 ctl_vector2_t u_ab;
151 ctl_vector2_scale(&u_ab, &MPC_VOLTAGE_VECTORS_NORMALIZED_ACM[i], voltage_scale);
152
153 ctl_vector2_t temp1, temp2;
154
155 // 3. Predict next state (current and flux) using the discrete model
156 //tex:
157 // $$ i_s(k+1) = c_{iss}*i_s(k) + c_i s_u*u_s(k) + c_i s_pr \times \psi_r(k) + c_i s_p r_w \times omega_r \times J \times psi_r(k) $$
158 ctl_vector2_scale(&temp1, is_ab_fbk, mpc->c_iss);
159 ctl_vector2_scale(&temp2, &u_ab, mpc->c_is_u);
160 ctl_vector2_add(&is_pred, &temp1, &temp2);
161 ctl_vector2_scale(&temp1, &mpc->psi_r_est, mpc->c_is_pr);
162 ctl_vector2_add(&is_pred, &is_pred, &temp1);
163 ctl_vector2_scale(&temp2, &J_psi_r_est, ctl_mul(mpc->c_is_pr_w, omega_e_rotor));
164 ctl_vector2_add(&is_pred, &is_pred, &temp2);
165
166 //tex:
167 // $$ \psi_r(k+1) = c_pr_is*i_s(k) + c_pr_pr\times \psi_r(k) - Ts\times \omega_r*J\times \psi_r(k) $$
168 ctl_vector2_scale(&temp1, is_ab_fbk, mpc->c_pr_is);
169 ctl_vector2_scale(&temp2, &mpc->psi_r_est, mpc->c_pr_pr);
170 ctl_vector2_add(&psi_r_pred, &temp1, &temp2);
171 ctl_vector2_scale(&temp1, &J_psi_r_est, -ctl_mul(mpc->Ts, omega_e_rotor));
172 ctl_vector2_add(&psi_r_pred, &psi_r_pred, &temp1);
173
174 // 4. Calculate cost function
175 ctl_vector2_t current_error;
176 ctl_vector2_sub(&current_error, &is_ab_ref, &is_pred);
177 ctrl_gt flux_error = mpc->flux_ref_sq - ctl_vector2_mag_sq(&psi_r_pred);
178 ctrl_gt cost = ctl_vector2_mag_sq(&current_error) + mpc->lambda_flux * (flux_error * flux_error);
179
180 // 5. Find the vector that minimizes the cost
181 if (cost < min_cost)
182 {
183 min_cost = cost;
184 best_vector_idx = (uint8_t)i;
185 }
186 }
187
188 // 6. Set the optimal vector index as the output
189 mpc->optimal_vector_index = best_vector_idx;
190
191 // 7. Update the internal flux observer state for the next cycle
192 // using the prediction generated by the chosen optimal vector.
193 ctl_vector2_t u_ab_opt;
194 ctl_vector2_scale(&u_ab_opt, &MPC_VOLTAGE_VECTORS_NORMALIZED_ACM[best_vector_idx], voltage_scale);
195 ctl_vector2_t temp1, temp2;
196 ctl_vector2_scale(&temp1, is_ab_fbk, mpc->c_pr_is);
197 ctl_vector2_scale(&temp2, &mpc->psi_r_est, mpc->c_pr_pr);
198 ctl_vector2_add(&mpc->psi_r_est, &temp1, &temp2);
199 ctl_vector2_scale(&temp1, &J_psi_r_est, -ctl_mul(mpc->Ts, omega_e_rotor));
200 ctl_vector2_add(&mpc->psi_r_est, &mpc->psi_r_est, &temp1);
201}
