107GMP_STATIC_INLINE
108ctrl_gt ctl_step_mppt_PnO_algo(mppt_PnO_algo_t* mppt, ctrl_gt current_voltage, ctrl_gt current_current)
109{
110 // Calculate and filter the current power
111 mppt->current_power = ctl_step_lowpass_filter(&mppt->power_filter, ctl_mul(current_voltage, current_current));
112
113 // This block only executes when the divider triggers, i.e., at the MPPT frequency
114 if (mppt->flag_enable_mppt && ctl_step_divider(&mppt->divider))
115 {
116 // Calculate power and voltage deltas
117 mppt->delta_power = mppt->current_power - mppt->last_power;
118 mppt->delta_voltage = current_voltage - mppt->last_voltage;
119
120 // --- Core P&O Logic ---
121 if (mppt->delta_power > CTL_MPPT_PNO_EPSILON) // Power is increasing, continue in the same direction
122 {
123 if (mppt->delta_voltage > 0)
124 {
125 mppt->v_ref += mppt->voltage_increment; // Increase V_ref
126 mppt->mppt_record = (mppt->mppt_record << 1) + 1;
127 }
128 else
129 {
130 mppt->v_ref -= mppt->voltage_increment; // Decrease V_ref
131 mppt->mppt_record = mppt->mppt_record << 1;
132 }
133 }
134 else // Power is decreasing, reverse the direction
135 {
136 if (mppt->delta_voltage > 0)
137 {
138 mppt->v_ref -= mppt->voltage_increment; // Decrease V_ref
139 mppt->mppt_record = mppt->mppt_record << 1;
140 }
141 else
142 {
143 mppt->v_ref += mppt->voltage_increment; // Increase V_ref
144 mppt->mppt_record = (mppt->mppt_record << 1) + 1;
145 }
146 }
147
148 // --- Adaptive Step Size Logic ---
149 if (mppt->flag_enable_adaptive_step_size)
150 {
151 // If oscillating around MPP, decrease the step size for fine-tuning
152 if (((mppt->mppt_record & 0x07) == 2) || ((mppt->mppt_record & 0x07) == 5)) // Pattern 010 or 101
153 {
154 mppt->voltage_increment = ctl_mul(mppt->searching_step_min, mppt->inc_attenuation) +
155 ctl_mul(mppt->voltage_increment, CTL_CTRL_CONST_1 - mppt->inc_attenuation);
156 }
157 // Advanced oscillation detection and correction
158 else if (((mppt->mppt_record & 0x0F) == 12)) // Pattern 1100
159 {
160 mppt->v_ref += mppt->voltage_increment; // Backtrack
161 mppt->voltage_increment = ctl_mul(mppt->searching_step_min, mppt->inc_attenuation) +
162 ctl_mul(mppt->voltage_increment, CTL_CTRL_CONST_1 - mppt->inc_attenuation);
163 mppt->v_ref -= mppt->voltage_increment; // Restore with smaller step
164 }
165 else if (((mppt->mppt_record & 0x0F) == 3)) // Pattern 0011
166 {
167 mppt->v_ref -= mppt->voltage_increment; // Backtrack
168 mppt->voltage_increment = ctl_mul(mppt->searching_step_min, mppt->inc_attenuation) +
169 ctl_mul(mppt->voltage_increment, CTL_CTRL_CONST_1 - mppt->inc_attenuation);
170 mppt->v_ref += mppt->voltage_increment; // Restore with smaller step
171 }
172 // If moving consistently in one direction, increase step size for fast tracking
173 else if (((mppt->mppt_record & 0x0F) == 0x0F) || ((mppt->mppt_record & 0x0F) == 0x00))
174 {
175 mppt->voltage_increment = mppt->searching_step_max;
176 }
177 }
178
179 // Apply voltage limits
180 mppt->v_ref = ctl_sat(mppt->v_ref, mppt->max_voltage_limit, mppt->min_voltage_limit);
181
182 // Update state for the next iteration
183 mppt->last_power = mppt->current_power;
184 mppt->last_voltage = current_voltage;
185 }
186
187 return mppt->v_ref;
188}
