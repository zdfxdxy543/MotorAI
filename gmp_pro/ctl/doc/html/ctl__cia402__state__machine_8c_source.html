22cia402_cmd_t get_cia402_control_cmd(uint16_t control_word)
23{
24 // 1. Fault Reset (Bit 7)
25 // 这是一个特殊的动作位，通常检测上升沿。
26 // 但如果作为静态命令解析，只要置位即视为复位请求。
27 // target state is CIA402_SM_FAULT_REACTIVE and then FAULT
28 if ((control_word & 0x0080) != 0)
29 {
30 return CIA402_CMD_FAULT_RESET;
31 }
32
33 // 2. Disable Voltage (Bit 1 = 0)
34 // 掩码: xxxx xxxx xxxx xx0x
35 // 表格逻辑: 只要 Bit 1 为 0，就是 Disable Voltage (Transition 7,9,10,12)
36 // 这是最高优先级的停机命令。
37 // target state is Switch On Disabled
38 if ((control_word & 0x0002) == 0)
39 {
40 return CIA402_CMD_DISABLE_VOLTAGE;
41 }
42
43 // 3. Quick Stop (Bit 2 = 0)
44 // 掩码: xxxx xxxx xxxx x01x
45 // 前提: Bit 1 必须为 1 (上面已经判断过了)
46 // 表格逻辑: Bit 2 为 0，Bit 1 为 1 => Quick Stop (Transition 7,10,11)
47 // target state is CIA402_SM_QUICK_STOP_ACTIVE and then Switch On Disabled
48 if ((control_word & 0x0004) == 0)
49 {
50 return CIA402_CMD_QUICK_STOP;
51 }
52
53 // 4. Shutdown (Bit 0 = 0)
54 // 掩码: xxxx xxxx xxxx x110
55 // 前提: Bit 1=1, Bit 2=1
56 // 表格逻辑: Bit 0 为 0 => Shutdown (Transition 2,6,8)
57 // target state is Ready to Switch On
58 if ((control_word & 0x0001) == 0)
59 {
60 return CIA402_CMD_SHUTDOWN;
61 }
62
63 // 5. Enable Operation (Bit 3 = 1)
64 // 掩码: 0000 0000 0000 1111 (0xF) -> 值必须为 0xF (1111)
65 if ((control_word & 0x000F) == 0x000F)
66 {
67 return CIA402_CMD_ENABLE_OPERATION;
68 }
69
70 // 6. Switch On (Bit 3 = 0)
71 // 掩码: 0000 0000 0000 1111 (0xF) -> 值必须为 0x7 (0111)
72 if ((control_word & 0x000F) == 0x0007)
73 {
74 return CIA402_CMD_SWITCHON;
75 }
76
77 // 7. Disable Operation (通常也是 0x7, 但在 Operation Enabled 状态下处理)
78 // 如果都不匹配，返回 NULL 或者 KEEP
79 return CIA402_CMD_NULL;
80}

87cia402_state_t get_cia402_state(uint16_t status_word)
88{
89 // ---------------------------------------------------------
90 // 掩码定义 (基于表格中的 x 位)
91 // ---------------------------------------------------------
92
93 // MASK_FULL: 关注 Bit 0, 1, 2, 3, 5, 6
94 // 二进制: 0000 0000 0110 1111 -> 0x006F
95 // 用于判断: Ready, Switched On, Op Enabled, Quick Stop
96 const uint16_t MASK_FULL = 0x006F;
97
98 // MASK_PARTIAL: 关注 Bit 0, 1, 2, 3, 6 (忽略 Bit 5)
99 // 二进制: 0000 0000 0100 1111 -> 0x004F
100 // 用于判断: Fault, Fault Reactive, Switch On Disabled, Not Ready
101 const uint16_t MASK_PARTIAL = 0x004F;
102
103 // ---------------------------------------------------------
104 // 状态判定 (优先级顺序通常不影响结果，因为特征值是互斥的)
105 // ---------------------------------------------------------
106
107 // 1. (7) Fault Reactive: x0xx 1111 (Bit 6=0, Bit 5=x, Bits 0-3=1)
108 if ((status_word & MASK_PARTIAL) == 0x000F)
109 {
110 return CIA402_SM_FAULT_REACTION;
111 }
112
113 // 2. (8) Fault: x0xx 1000 (Bit 6=0, Bit 5=x, Bit 3=1)
114 if ((status_word & MASK_PARTIAL) == 0x0008)
115 {
116 return CIA402_SM_FAULT;
117 }
118
119 // 3. (5) Operation Enabled: x01x 0111 (Bit 6=0, Bit 5=1, Bit 3=0, Bit 2=1, Bit 1=1, Bit 0=1)
120 if ((status_word & MASK_FULL) == 0x0027)
121 {
122 return CIA402_SM_OPERATION_ENABLED;
123 }
124
125 // 4. (4) Switched On: x01x 0011 (Bit 6=0, Bit 5=1, Bit 3=0, Bit 2=0, Bit 1=1, Bit 0=1)
126 if ((status_word & MASK_FULL) == 0x0023)
127 {
128 return CIA402_SM_SWITCHED_ON;
129 }
130
131 // 5. (3) Ready to Switch On: x01x 0001 (Bit 6=0, Bit 5=1, Bit 3=0, Bit 2=0, Bit 1=0, Bit 0=1)
132 if ((status_word & MASK_FULL) == 0x0021)
133 {
134 return CIA402_SM_READY_TO_SWITCH_ON;
135 }
136
137 // 6. (6) Quick Stop Active: x00x 0111 (Bit 6=0, Bit 5=0, Bit 3=0, Bit 2=1, Bit 1=1, Bit 0=1)
138 // 注意: 这里 Bit 5 (Quick Stop) 为 0，表示正在急停中
139 if ((status_word & MASK_FULL) == 0x0007)
140 {
141 return CIA402_SM_QUICK_STOP_ACTIVE;
142 }
143
144 // 7. (2) Switch On Disabled: x1xx 0000 (Bit 6=1, Bit 5=x, Bits 0-3=0)
145 if ((status_word & MASK_PARTIAL) == 0x0040)
146 {
147 return CIA402_SM_SWITCH_ON_DISABLED;
148 }
149
150 // 8. (1) Not Ready to Switch On: x0xx 0000 (Bit 6=0, Bit 5=x, Bits 0-3=0)
151 if ((status_word & MASK_PARTIAL) == 0x0000)
152 {
153 return CIA402_SM_NOT_READY_TO_SWITCH_ON;
154 }
155
156 // 如果都不匹配 (可能处于中间过渡态，通常归类为 Not Ready 或 Unknown)
157 return CIA402_SM_UNKNOWN; // 或者 return CIA402_SM_NOT_READY_TO_SWITCH_ON;
158}

160// init cia402 state machine structure, state machine will switch to Not ready to switch on.
161void init_cia402_state_machine(cia402_sm_t* sm)
162{
163 gmp_base_assert(sm);
164
165 // init state
166 sm->current_state = CIA402_SM_NOT_READY_TO_SWITCH_ON;
167 sm->current_cmd = CIA402_CMD_DISABLE_VOLTAGE;
168 sm->current_state_counter = 0;
169 sm->current_tick = 0;
170 sm->control_word.all = 0;
171 sm->state_word.all = 0;
172
173 sm->flag_fault_reset_request = 0;
174
175#if defined CIA402_CONFIG_DISABLE_CONTROL_WORD_DEFAULT
176 sm->flag_enable_control_word = 0; // 默认禁用控制字
177#else // CIA402_CONFIG_DISABLE_CONTROL_WORD_DEFAULT
178 sm->flag_enable_control_word = 1; // 默认使能控制字
179#endif // CIA402_CONFIG_DISABLE_CONTROL_WORD_DEFAULT
180 sm->last_cb_result = CIA402_EC_KEEP;
181 sm->last_fault_reset_bit = 0;
182
183 // default delay config
184 sm->minimum_transit_delay[0] = CIA402_CONFIG_MIN_DELAY_READY;
185 sm->minimum_transit_delay[1] = CIA402_CONFIG_MIN_DELAY_SHUTDOWN;
186 sm->minimum_transit_delay[2] = CIA402_CONFIG_MIN_DELAY_SWITCHON;
187 sm->minimum_transit_delay[3] = CIA402_CONFIG_MIN_DELAY_OPERATION_EN;
188
189 // init default callback function
190 sm->switch_on_disabled = default_cb_fn_switch_on_disabled;
191 sm->ready_to_switch_on = default_cb_fn_ready_to_switch_on;
192 sm->switched_on = default_cb_fn_switched_on;
193 sm->operation_enabled = default_cb_fn_operation_enabled;
194 sm->quick_stop_active = default_cb_fn_quick_stop_active;
195 sm->fault_reaction = default_cb_fn_fault_reaction;
196 sm->fault = default_cb_fn_fault;
197}

199void cia402_update_status_word(cia402_sm_t* sm)
200{
201 gmp_base_assert(sm);
202
203 // 清除与状态相关的核心位: Bit 0,1,2,3,5,6
204 // 保留 Bit 4 (Voltage), Bit 7 (Warning) 等，因为这些可能由外部逻辑设置
205 // 这里为了简化，我们重写核心状态位
206
207 cia402_state_word_t s = sm->state_word;
208
209 // 默认值：QuickStop=1 (正常), Fault=0, SwitchOnDisabled=0
210 s.all &= ~0x006F; // 清除 Bit 0,1,2,3,5,6
211 s.all |= CIA402_STATEWORD_QUICKSTOP; // Bit 5 默认为 1 (Not Active)
212
213 switch (sm->current_state)
214 {
215 case CIA402_SM_NOT_READY_TO_SWITCH_ON:
216 // x0xx 0000
217 // 通常初始化时 QuickStop 位也为 0
218 s.bits.quick_stop = 0;
219 break;
220
221 case CIA402_SM_SWITCH_ON_DISABLED:
222 // x1xx 0000
223 s.bits.switch_on_disabled = 1;
224 break;
225
226 case CIA402_SM_READY_TO_SWITCH_ON:
227 // x01x 0001
228 s.bits.ready_to_switch_on = 1;
229 break;
230
231 case CIA402_SM_SWITCHED_ON:
232 // x01x 0011
233 s.bits.ready_to_switch_on = 1;
234 s.bits.switched_on = 1;
235 break;
236
237 case CIA402_SM_OPERATION_ENABLED:
238 // x01x 0111
239 s.bits.ready_to_switch_on = 1;
240 s.bits.switched_on = 1;
241 s.bits.operation_enabled = 1;
242 break;
243
244 case CIA402_SM_QUICK_STOP_ACTIVE:
245 // x00x 0111
246 s.bits.quick_stop = 0;
247 s.bits.ready_to_switch_on = 1;
248 s.bits.switched_on = 1;
249 s.bits.operation_enabled = 1;
250 break;
251
252 case CIA402_SM_FAULT_REACTION:
253 // x0xx 1111 (Fault 但不改变当前系统状态)
254 // Fault Reaction 期间，Bit 0-2 通常保持为 1 (看起来像 Op Enabled)，同时 Bit 3 (Fault) 置位
255 s.bits.fault = 1;
256 s.bits.ready_to_switch_on = 1;
257 s.bits.switched_on = 1;
258 s.bits.operation_enabled = 1;
259 break;
260
261 case CIA402_SM_FAULT:
262 // x0xx 1000 (Fault only)
263 s.bits.fault = 1;
264 break;
265
266 default:
267 break;
268 }
269
270 // 更新回结构体
271 sm->state_word.all = s.all;
272}

274// Fault condition cannot escape by this function
275void cia402_transit(cia402_sm_t* sm, cia402_state_t next_state)
276{
277 // Common transit
278 if (sm->current_state != next_state && sm->current_state != CIA402_SM_FAULT &&
279 sm->current_state != CIA402_SM_FAULT_REACTION)
280 {
281 // change to next state
282 sm->current_state = next_state;
283
284 // start a new delay stage
285 sm->flag_delay_stage = 0;
286 sm->entry_state_tick = gmp_base_get_system_tick();
287 sm->current_state_counter = 0;
288
289 // update state word here
290 cia402_update_status_word(sm);
291 }
292
293 // fault reaction transit
294 else if (sm->current_state == CIA402_SM_FAULT_REACTION && next_state != CIA402_SM_FAULT_REACTION)
295 {
296 // change to next state
297 sm->current_state = CIA402_SM_FAULT;
298
299 // start a new delay stage
300 sm->flag_delay_stage = 0;
301 sm->entry_state_tick = gmp_base_get_system_tick();
302 sm->current_state_counter = 0;
303
304 // update state word here
305 cia402_update_status_word(sm);
306 }

309static void _switch_on_disable_routine(cia402_sm_t* sm)
310{
311 gmp_base_assert(sm->switch_on_disabled);
312
313 sm->last_cb_result = sm->switch_on_disabled(sm);
314
315 // error condition
316 if (sm->last_cb_result <= CIA402_EC_ERROR)
317 {
318 cia402_fault_request(sm);
319 return;
320 }
321
322 // request next state
323 if (sm->last_cb_result >= CIA402_EC_NEXT_STATE)
324 {
325#if defined CIA402_CONFIG_ENABLE_SEQUENCE_SWITCH
326 if (sm->current_cmd == CIA402_CMD_SHUTDOWN || sm->current_cmd == CIA402_CMD_SWITCHON ||
327 sm->current_cmd == CIA402_CMD_ENABLE_OPERATION)
328#else
329 if (sm->current_cmd == CIA402_CMD_SHUTDOWN)
330#endif // CIA402_CONFIG_ENABLE_SEQUENCE_SWITCH
331 {
332 // The first time enter ready state, log current tick.
333 if (sm->flag_delay_stage == 0)
334 {
335 sm->state_ready_tick = sm->current_tick;
336 sm->flag_delay_stage = 1;
337 }
338
339 // time_diff = current_tick - state_ready_tick;
340 time_gt time_diff = gmp_base_time_sub(sm->current_tick, sm->state_ready_tick);
341
342 // judge if delay condition is meet.
343 if (time_diff >= sm->minimum_transit_delay[1])
344 cia402_transit(sm, CIA402_SM_READY_TO_SWITCH_ON);
345 }
346 return;
347 }
348
349 if (sm->last_cb_result == CIA402_EC_KEEP)
350 {
351 sm->flag_delay_stage = 0;
352 }

582// dispatch routine in mainloop
583// This function would be called in mainloop
584void cia402_dispatch(cia402_sm_t* sm)
585{
586 gmp_base_assert(sm);
587
588 sm->current_tick = gmp_base_get_system_tick();
589
590 // snap shot of control word
591 cia402_ctrl_word_t control_word = sm->control_word;
592
593 // 1. get control command, get control word
594 if (sm->flag_enable_control_word)
595 {
596 // get request state by control word
597 sm->current_cmd = get_cia402_control_cmd(control_word.all);
598 }
599
600 // 2. judge if fault reset is request
601 // 只有在falut状态下reset才是恢复到CIA402_SM_SWITCH_ON_DISABLED，其他的都通过request_state恢复。
602
603 // Detection of Fault Reset Edge (0 -> 1)
604 if (sm->flag_enable_control_word)
605 {
606 fast_gt current_reset_bit = control_word.bits.fault_reset;
607 if (current_reset_bit && !sm->last_fault_reset_bit)
608 {
609 sm->flag_fault_reset_request = 1;
610 }
611 sm->last_fault_reset_bit = current_reset_bit;
612 }
613
614 // 3. if in other states Call CiA402 callback function
615 switch (sm->current_state)
616 {
617 case CIA402_SM_NOT_READY_TO_SWITCH_ON:
618 // transit to DISABLED, no callback function, all codes should implement in init stage.
619 cia402_transit(sm, CIA402_SM_SWITCH_ON_DISABLED);
620 break;
621
622 case CIA402_SM_SWITCH_ON_DISABLED:
623 _switch_on_disable_routine(sm);
624 break;
625
626 case CIA402_SM_READY_TO_SWITCH_ON:
627 _ready_to_switch_on_routine(sm);
628 break;
629
630 case CIA402_SM_SWITCHED_ON:
631 _switched_on_routine(sm);
632 break;
633
634 case CIA402_SM_OPERATION_ENABLED:
635 _operation_enabled_routine(sm);
636 break;
637
638 case CIA402_SM_QUICK_STOP_ACTIVE:
639 _quick_stop_active_routine(sm);
640 break;
641
642 case CIA402_SM_FAULT_REACTION:
643 _fault_reaction_routine(sm);
644 break;
645
646 case CIA402_SM_FAULT:
647 _fault_routine(sm);
648 break;
649
650 // unknown state
651 default:
652 cia402_fault_request(sm);
653 }
654
655 // 4. update status word
656 cia402_update_status_word(sm);
657
658 // update counter
659 sm->current_state_counter += 1;
660}

713// =========================================================================
714// 1. Switch On Disabled (初始化与待机)
715// =========================================================================
716cia402_sm_error_code_t default_cb_fn_switch_on_disabled(cia402_sm_t* sm)
717{
718 // [Entry Action] 进入状态的第一拍执行
719 if (sm->current_state_counter <= 1)
720 {
721 // output_disable & power_off
722 ctl_disable_pwm();
723 ctl_disable_main_contactor();
724 ctl_disable_grid_relay();
725 ctl_disable_precharge_relay();
726 ctl_restore_brake(); // 抱闸
727 }
728
729 // [Do Action] ctl_if_adc_calibrate
730 // 执行 ADC 校准 (非阻塞)
731 if (ctl_exec_adc_calibration() == 0)
732 {
733 // 检查是否超时 (利用 entry_state_tick)
734 if ((sm->current_tick - sm->entry_state_tick) > TIMEOUT_ADC_CALIB_MS)
735 {
736 return CIA402_EC_ERROR; // 校准超时，报错
737 }
738 return CIA402_EC_KEEP; // 等待校准完成
739 }
740
741 // 检查编码器/传感器状态
742 if (ctl_check_encoder() == 0)
743 {
744 return CIA402_EC_KEEP; // 传感器未就绪
745 }
746
747 // 硬件准备就绪
748 return CIA402_EC_NEXT_STATE;
749}
