123GMP_STATIC_INLINE ctrl_gt ctl_step_mrac(ctl_mrac_controller_t* mrac, ctrl_gt r, ctrl_gt y_p)
124{
125 // 1. Update the reference model to get the desired output for this step
126 // y_m(k) = a_m_d * y_m(k-1) + b_m_d * r(k-1)
127 mrac->y_m = mrac->a_m_d * mrac->y_m + mrac->b_m_d * r;
128
129 // 2. Calculate the tracking error
130 ctrl_gt error = y_p - mrac->y_m;
131
132 // 3. Update the adaptive gains using the discretized MIT rule
133 // k(k) = k(k-1) - gamma * error * signal
134 mrac->k_r -= mrac->gamma_r_d * error * r;
135 mrac->k_y += mrac->gamma_y_d * error * y_p;
136
137 // 4. Calculate the control law
138 mrac->u_out = mrac->k_r * r - mrac->k_y * y_p;
139
140 return mrac->u_out;
141}
