7static void calculate_all_base_values(ctl_per_unit_consultant_t* pu)
8{
9 if (pu->base_voltage <= 0 || pu->base_power <= 0 || pu->base_freq <= 0 || pu->pole_pairs == 0)
10 {
11 return;
12 }
13 // 1. elec base value
14 pu->base_omega = 2.0f * CTL_PARAM_CONST_PI * pu->base_freq;
15 pu->base_current = pu->base_power / pu->base_voltage;
16 pu->base_impedence = pu->base_voltage / pu->base_current;
17 pu->base_inductance = pu->base_impedence / pu->base_omega;
18 pu->base_capacitance = 1.0f / (pu->base_impedence * pu->base_omega);
19 pu->base_flux = pu->base_voltage / pu->base_omega;
20 pu->base_inst_voltage = pu->base_voltage * sqrtf(2.0);
