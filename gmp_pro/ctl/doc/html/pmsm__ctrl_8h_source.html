220GMP_STATIC_INLINE void ctl_step_pmsm_ctrl(pmsm_controller_t* ctrl)
221{
222 ctl_vector2_t phasor;
223 ctrl_gt etheta;
224 ctrl_gt vq_limit = float2ctrl(1.0);
225
226 ctrl->isr_tick++;
227
228 if (ctrl->flag_enable_controller)
229 {
230 // Clarke Transformation
231#if MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 3
232 ctl_ct_clarke(&ctrl->mtr_interface.iabc->value, &ctrl->iab0);
233#elif MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 2
234 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->mtr_interface.iabc->value, (ctl_vector2_t*)&ctrl->iab0);
235#endif
236
237 // Park Transformation
238 etheta = ctl_get_mtr_elec_theta(&ctrl->mtr_interface);
239 ctl_set_phasor_via_angle(etheta, &phasor);
240 ctl_ct_park(&ctrl->iab0, &phasor, &ctrl->idq0);
241
242 // Position controller
243 if (ctrl->flag_enable_position_ctrl)
244 { /* To be implemented */
245 }
246
247 // Velocity Controller
248 if (ctrl->flag_enable_velocity_ctrl)
249 {
250 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
251#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
252 ctrl->idq_set.dat[phase_q] =
253 ctl_step_tracking_pid(&ctrl->spd_ctrl, ctrl->speed_set, ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
254 ctrl->idq_ff.dat[phase_q];
255#else
256 ctrl->idq_set.dat[phase_q] = ctl_step_tracking_continuous_pid(&ctrl->spd_ctrl, ctrl->speed_set,
257 ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
258 ctrl->idq_ff.dat[phase_q];
259#endif
260 }
261 else
262 {
263 ctl_vector2_copy(&ctrl->idq_set, &ctrl->idq_ff);
264 }
265
266 // Current Controller
267 if (ctrl->flag_enable_current_ctrl)
268 {
269#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
270 ctrl->vdq_set.dat[phase_d] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_d],
271 ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
272 ctrl->vdq_ff.dat[phase_d];
273 ctrl->vdq_set.dat[phase_q] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_q],
274 ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
275 ctrl->vdq_ff.dat[phase_q];
276#else
277 ctrl->vdq_set.dat[phase_d] =
278 ctl_step_pid_ser(&ctrl->current_ctrl[phase_d], ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
279 ctrl->vdq_ff.dat[phase_d];
280 vq_limit = ctl_sqrt(float2ctrl(1.0) - ctl_mul(ctrl->vdq_set.dat[phase_d], ctrl->vdq_set.dat[phase_d]));
281 ctl_set_pid_limit(&ctrl->current_ctrl[phase_q], vq_limit, -vq_limit);
282 ctrl->vdq_set.dat[phase_q] =
283 ctl_step_pid_ser(&ctrl->current_ctrl[phase_q], ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
284 ctrl->vdq_ff.dat[phase_q];
285#endif
286 ctrl->vdq_set.dat[phase_0] = 0;
287 }
288 else
289 {
290 ctl_vector2_copy((ctl_vector2_t*)&ctrl->vdq_set, &ctrl->vdq_ff);
291 ctrl->vdq_set.dat[phase_0] = 0;
292 }
293
294 // Inverse Park and SVPWM
295 if (ctrl->flag_enable_modulation)
296 {
297 ctl_ct_ipark(&ctrl->vdq_set, &phasor, &ctrl->vab0_set);
298 }
299
300 if (ctrl->flag_enable_output)
301 {
302 ctl_ct_svpwm_calc(&ctrl->vab0_set, &ctrl->pwm_out->value);
303 }
304 else
305 {
306 ctl_vector3_set(&ctrl->pwm_out->value, float2ctrl(0.5));
307 }
308 }
309 else
310 {
311 ctl_vector3_set(&ctrl->pwm_out->value, float2ctrl(0.5));
312 //ctl_clear_pmsm_ctrl(ctrl);
313 }
314}
