186GMP_STATIC_INLINE ctrl_gt ctl_step_autoturn_pos_encoder(pos_autoturn_encoder_t* enc, uint32_t raw)
187{
188 enc->raw = raw;
189 enc->encif.position = ctl_div(raw, enc->position_base);
190
191 ctrl_gt elec_pos = (enc->encif.position + CTL_CTRL_CONST_1 - enc->offset) * enc->pole_pairs;
192 ctrl_gt elec_pos_pu = ctrl_mod_1(elec_pos);
193 enc->encif.elec_position = elec_pos_pu;
194
195 // Check for revolution crossing
196 if (enc->encif.position - enc->last_pos > CTL_CTRL_CONST_1_OVER_2)
197 {
198 enc->encif.revolutions -= 1; // Negative direction rollover
199 }
200 if (enc->last_pos - enc->encif.position > CTL_CTRL_CONST_1_OVER_2)
201 {
202 enc->encif.revolutions += 1; // Positive direction rollover
203 }
204
205 enc->last_pos = enc->encif.position;
206 return enc->encif.elec_position;
207}

293GMP_STATIC_INLINE void ctl_step_spd_calc(spd_calculator_t* sc)
294{
295 if (ctl_step_divider(&sc->div))
296 {
297 ctrl_gt current_pos = ctl_get_encoder_position(sc->pos_encif);
298 ctrl_gt delta = current_pos - sc->old_position;
299
300 // Correct for position rollover
301 if (delta < -CTL_CTRL_CONST_1_OVER_2)
302 {
303 delta += CTL_CTRL_CONST_1;
304 }
305 else if (delta > CTL_CTRL_CONST_1_OVER_2)
306 {
307 delta -= CTL_CTRL_CONST_1;
308 }
309
310 ctrl_gt new_spd = ctl_mul(delta, sc->scale_factor);
311 sc->encif.speed = ctl_step_lowpass_filter(&sc->spd_filter, new_spd);
312
313 sc->old_position = current_pos;
314 }
315}
