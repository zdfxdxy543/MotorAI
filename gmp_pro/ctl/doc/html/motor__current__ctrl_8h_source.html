234GMP_STATIC_INLINE void ctl_step_current_controller(mtr_current_ctrl_t* mc)
235{
236 mc->isr_tick += 1;
237
238 //
239 // 1. enable theta input or phasor input
240 //
241 if (mc->flag_enable_theta_calc)
242 {
243 ctl_set_phasor_via_angle(mc->pos_if->elec_position, &mc->phasor);
244 }
245 else
246 {
247 gmp_base_assert(mc->phasor_input);
248 ctl_vector2_copy(&mc->phasor, mc->phasor_input);
249 }
250
251 //
252 // 2. input filter
253 //
254#if MC_CURRENT_SAMPLE_PHASE_MODE == 3
255 mc->iuvw.dat[phase_U] = ctl_step_filter_iir1(&mc->filter_iuvw[phase_U], mc->adc_iuvw->value.dat[phase_A]);
256 mc->iuvw.dat[phase_V] = ctl_step_filter_iir1(&mc->filter_iuvw[phase_V], mc->adc_iuvw->value.dat[phase_B]);
257 mc->iuvw.dat[phase_W] = ctl_step_filter_iir1(&mc->filter_iuvw[phase_W], mc->adc_iuvw->value.dat[phase_C]);
258
259 // 1. Clarke Transform: 3-phase currents to alpha-beta stationary frame.
260 ctl_ct_clarke(&mc->iuvw, &mc->iab0);
261
262#elif MC_CURRENT_SAMPLE_PHASE_MODE == 2
263 mc->iuvw.dat[phase_U] = ctl_step_filter_iir1(&mc->filter_iuvw[phase_U], mc->adc_iuvw->value.dat[phase_A]);
264 mc->iuvw.dat[phase_V] = ctl_step_filter_iir1(&mc->filter_iuvw[phase_V], mc->adc_iuvw->value.dat[phase_B]);
265
266 // 1. Clarke Transform: 3-phase currents to alpha-beta stationary frame.
267 ctl_ct_clarke_2ph((ctl_vector2_t*)&mc->iuvw, (ctl_vector2_t*)&mc->iab0);
268 mc->iab0.dat[phase_0] = 0;
269
270#endif // GFL_CURRENT_SAMPLE_PHASE_MODE
271
272 // input DC Bus Voltage filter
273 mc->udc = ctl_step_filter_iir1(&mc->filter_udc, mc->adc_udc->value);
274
275 //
276 // 3. Park Transform: alpha-beta -> d-q
277 //
278 ctl_ct_park(&mc->iab0, &mc->phasor, &mc->idq0);
279
280 //
281 // 4. Controller kernel, current loop
282 //
283 if (mc->flag_enable_current_ctrl)
284 {
285 //
286 // 4.1 Calculate error and step the PI controllers
287 //
288 ctrl_gt err_d = mc->idq_ref.dat[phase_d] - mc->idq0.dat[phase_d];
289 ctrl_gt err_q = mc->idq_ref.dat[phase_q] - mc->idq0.dat[phase_q];
290
291 // d axis controller limited by Vs,max, q axis controller limited by vd
292 //ctl_set_pid_limit(&mc->idq_ctrl[phase_d], mc->max_vs_mag, -mc->max_vs_mag);
293 //ctl_set_pid_limit(&mc->idq_ctrl[phase_q], mc->max_vq_mag, -mc->max_vq_mag);
294
295 mc->vdq_ctrl_out.dat[phase_d] = ctl_step_pid_ser(&mc->idq_ctrl[phase_d], err_d);
296 mc->vdq_ctrl_out.dat[phase_q] = ctl_step_pid_ser(&mc->idq_ctrl[phase_q], err_q);
297
298 //
299 // 4.2 Calculate feed forward decoupling
300 //
301 if (mc->flag_enable_decouple)
302 {
303 // decoupling
304 mc->vdq_decouple.dat[phase_d] =
305 -ctl_mul(mc->spd_if->speed, ctl_mul(mc->coef_ff_decouple[phase_d], mc->idq0.dat[phase_q]));
306 mc->vdq_decouple.dat[phase_q] =
307 ctl_mul(mc->spd_if->speed, ctl_mul(mc->coef_ff_decouple[phase_q], mc->idq0.dat[phase_d]));
308
309 mc->vdq_ctrl_out.dat[phase_d] += mc->vdq_decouple.dat[phase_d];
310 mc->vdq_ctrl_out.dat[phase_q] += mc->vdq_decouple.dat[phase_q];
311 }
312
313 //
314 // 4.3 lead compensator
315 //
316 if (mc->flag_enable_lead_compensator)
317 {
318 mc->vdq_ref.dat[phase_d] = ctl_step_lead(&mc->lead_compensator[phase_d], mc->vdq0.dat[phase_d]);
319 mc->vdq_ref.dat[phase_q] = ctl_step_lead(&mc->lead_compensator[phase_q], mc->vdq0.dat[phase_q]);
320 }
321 else
322 {
323 ctl_vector2_copy(&mc->vdq_ref, &mc->vdq_ctrl_out);
324 }
325 }
326
327 //
328 // 5. vdq feed forward
329 //
330 if (mc->flag_enable_vdq_feedforward)
331 {
332 ctl_vector2_add((ctl_vector2_t*)&mc->vdq_out, &mc->vdq_ref, &mc->vdq_ff);
333 }
334 else
335 {
336 ctl_vector2_copy((ctl_vector2_t*)&mc->vdq_out, &mc->vdq_ref);
337 }
338
339 //
340 // 6. Bus Voltage Compensation, and vdq feed forward
341 //
342 if (mc->flag_enable_bus_compensation)
343 {
344 ctrl_gt v_scale;
345 if (mc->udc > float2ctrl(0.5f)) // prevent div 0
346 v_scale = mc->max_dcbus_voltage / mc->udc; // udc is per unit value
347 else
348 v_scale = mc->max_dcbus_voltage;
349
350 mc->vdq_out_bus_compensator.dat[phase_d] = ctl_mul(mc->vdq_out.dat[phase_d], v_scale);
351 mc->vdq_out_bus_compensator.dat[phase_q] = ctl_mul(mc->vdq_out.dat[phase_q], v_scale);
352 }
353 else
354 {
355 ctl_vector2_copy(&mc->vdq_out_bus_compensator, (ctl_vector2_t*)&mc->vdq_out);
356 }
357
358 //
359 // 6. output Circular Saturation
360 //
361
362 // 6.1 Saturation output
363 mc->vdq_out_sat.dat[phase_d] = ctl_sat(mc->vdq_out_bus_compensator.dat[phase_d], mc->max_vs_mag, -mc->max_vs_mag);
364
365 // q axis controller is limited by d axis output
366#if MC_CURRENT_OUTPUT_LIMIT_BY_SQRT == 1
367 ctrl_gt val_sq =
368 ctl_mul(mc->max_vs_mag, mc->max_vs_mag) - ctl_mul(mc->vdq_out_sat.dat[phase_d], mc->vdq_out_sat.dat[phase_d]);
369
370 if (val_sq < 0)
371 mc->max_vq_mag = 0;
372 else
373 mc->max_vq_mag = ctl_sqrt(val_sq);
374#else // default case
375 mc->max_vq_mag = mc->max_vs_mag - mc->vdq_out_sat.dat[phase_d];
376#endif // MC_CURRENT_OUTPUT_LIMIT_BY_SQRT
377
378 mc->vdq_out_sat.dat[phase_q] =
379 ctl_sat(mc->vdq_out.dat[phase_q] + mc->vdq_ff.dat[phase_q], mc->max_vq_mag, -mc->max_vq_mag);
380
381 // 6.2 PID Anti-Windup Back-calculation
382 if (mc->flag_enable_current_ctrl)
383 {
384 // 逻辑：PID real output = vdq_out_sat - all feed forward items
385
386 // --- D Axis Correction ---
387 ctrl_gt v_pid_d_real = mc->vdq_out_sat.dat[phase_d];
388 if (mc->flag_enable_decouple)
389 v_pid_d_real -= mc->vdq_decouple.dat[phase_d];
390 if (mc->flag_enable_vdq_feedforward)
391 v_pid_d_real -= mc->vdq_ff.dat[phase_d];
392
393 ctl_pid_clamping_correction_using_real_output(&mc->idq_ctrl[phase_d], v_pid_d_real);
394
395 // --- Q Axis Correction ---
396 ctrl_gt v_pid_q_real = mc->vdq_out_sat.dat[phase_q];
397 if (mc->flag_enable_decouple)
398 v_pid_q_real -= mc->vdq_decouple.dat[phase_q];
399 if (mc->flag_enable_vdq_feedforward)
400 v_pid_q_real -= mc->vdq_ff.dat[phase_q];
401
402 ctl_pid_clamping_correction_using_real_output(&mc->idq_ctrl[phase_q], v_pid_q_real);
403 }
404
405 mc->vdq_out.dat[phase_0] = 0;
406
407 // 7. iPark: d-q -> alpha-beta
408 ctl_ct_ipark(&mc->vdq_out, &mc->phasor, &mc->vab0);
409}
