170GMP_STATIC_INLINE ctrl_gt ctl_step_buck_ctrl(buck_ctrl_t* buck)
171{
172 // Filter the raw ADC inputs
173 ctl_step_lowpass_filter(&buck->lpf_il, buck->adc_il->value);
174 ctl_step_lowpass_filter(&buck->lpf_uo, buck->adc_uo->value);
175 ctl_step_lowpass_filter(&buck->lpf_ui, buck->adc_ui->value);
176
177 if (buck->flag_enable_system)
178 {
179 // Outer voltage loop
180 if (buck->flag_enable_voltage_ctrl)
181 {
182 buck->current_set =
183 ctl_step_pid_ser(&buck->voltage_pid, buck->voltage_set - buck->lpf_uo.out) + buck->current_ff;
184 }
185
186 // Inner current loop
187 if (buck->flag_enable_current_ctrl)
188 {
189 buck->voltage_out =
190 ctl_step_pid_ser(&buck->current_pid, buck->current_set - buck->lpf_il.out) + buck->voltage_ff;
191 }
192
193#ifdef CTL_BUCK_CTRL_OUTPUT_WITHOUT_UIN
194 // Duty cycle calculation without input voltage feedforward (for debugging)
195 buck->pwm_out_pu = buck->voltage_out;
196#else
197 // Standard duty cycle calculation: D = U_out_cmd / U_in
198 buck->ui_sat = ctl_step_saturation(&buck->modulation_saturation, buck->lpf_ui.out);
199 buck->pwm_out_pu = ctl_div(buck->voltage_out, buck->ui_sat);
200
201#endif // CTL_BUCK_CTRL_OUTPUT_WITHOUT_UIN
202 }
203 else
204 {
205 // For safety, ensure the Buck switch is off by default when disabled.
206 buck->pwm_out_pu = float2ctrl(0);
207 }
208
209 return buck->pwm_out_pu;
210}
