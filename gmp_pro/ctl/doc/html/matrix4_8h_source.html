170GMP_STATIC_INLINE ctl_matrix4_t ctl_matrix4_mul(ctl_matrix4_t a, ctl_matrix4_t b)
171{
172 ctl_matrix4_t result;
173 int i, j, k;
174
175 for (i = 0; i < 4; ++i)
176 {
177 for (j = 0; j < 4; ++j)
178 {
179 ctrl_gt sum = 0;
180 for (k = 0; k < 4; ++k)
181 {
182 sum += ctl_matrix4_get(&a, i, k) * ctl_matrix4_get(&b, k, j);
183 }
184 ctl_matrix4_set(&result, i, j, sum);
185 }
186 }
187 return result;
