125GMP_STATIC_INLINE void ctl_start_new_iteration(ctl_ilc_controller_t* ilc)
126{
127 uint32_t i;
128
129 // The control signal from the completed iteration becomes the base for the next one.
130 // The error from the completed iteration is now the error from the "previous" run.
131 for (i = 0; i < ilc->trajectory_length; ++i)
132 {
133 ilc->u_k_minus_1[i] = ilc->u_k[i];
134 // The error buffer e_k_minus_1 is now ready to be used for the new iteration.
135 }
136
137 // Reset the time step to the beginning of the trajectory
138 ilc->time_step = 0;
139}

148GMP_STATIC_INLINE ctrl_gt ctl_step_ilc(ctl_ilc_controller_t* ilc, ctrl_gt r, ctrl_gt y_p)
149{
150 if (ilc->time_step >= ilc->trajectory_length)
151 {
152 // Trajectory is finished for this iteration, hold the last output
153 return ilc->u_out;
154 }
155
156 // 1. Calculate the ILC update law
157 if (ilc->is_learning)
158 {
159 // u_k(t) = u_{k-1}(t) + L * e_{k-1}(t)
160 ilc->u_out = ilc->u_k_minus_1[ilc->time_step] + ctl_mul(ilc->learning_gain, ilc->e_k_minus_1[ilc->time_step]);
161 }
162 else
163 {
164 // If learning is disabled, just use the control signal from the last converged iteration
165 ilc->u_out = ilc->u_k_minus_1[ilc->time_step];
166 }
167
168 // 2. Calculate and store the error for the *next* iteration
169 ctrl_gt current_error = r - y_p;
170 ilc->e_k_minus_1[ilc->time_step] = current_error; // This will be e_{k-1} in the next run
171
172 // 3. Store the control signal for the *next* iteration
173 ilc->u_k[ilc->time_step] = ilc->u_out;
174
175 // 4. Advance the time step
176 ilc->time_step++;
177
178 return ilc->u_out;
179}
