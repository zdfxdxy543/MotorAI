123GMP_STATIC_INLINE void ctl_step_acm_smo(ctl_acm_smo_t* smo, ctrl_gt u_alpha, ctrl_gt u_beta, ctrl_gt i_alpha,
124 ctrl_gt i_beta)
125{
126 // 1. Apply the sliding control law based on current error
127 smo->z.dat[0] = ctl_mul(ctl_sat(smo->i_s_est.dat[0] - i_alpha, 1.0f, -1.0f), smo->k_slide);
128 smo->z.dat[1] = ctl_mul(ctl_sat(smo->i_s_est.dat[1] - i_beta, 1.0f, -1.0f), smo->k_slide);
129
130 // 2. Update the observer model using the discrete-time equations
131 // Rotor flux estimation
132 ctrl_gt d_psi_r_alpha = smo->c5 * smo->i_s_est.dat[0] - (1.0f / smo->c5 * smo->c3) * smo->psi_r_est.dat[0] -
133 smo->omega_e_est * smo->psi_r_est.dat[1];
134 ctrl_gt d_psi_r_beta = smo->c5 * smo->i_s_est.dat[1] - (1.0f / smo->c5 * smo->c3) * smo->psi_r_est.dat[1] +
135 smo->omega_e_est * smo->psi_r_est.dat[0];
136 smo->psi_r_est.dat[0] += d_psi_r_alpha * smo->Ts;
137 smo->psi_r_est.dat[1] += d_psi_r_beta * smo->Ts;
138
139 // Stator current estimation
140 ctrl_gt d_i_s_alpha = smo->c1 * u_alpha - smo->c2 * smo->i_s_est.dat[0] + smo->c3 * smo->psi_r_est.dat[0] +
141 smo->c4 * smo->omega_e_est * smo->psi_r_est.dat[1] - smo->z.dat[0];
142 ctrl_gt d_i_s_beta = smo->c1 * u_beta - smo->c2 * smo->i_s_est.dat[1] + smo->c3 * smo->psi_r_est.dat[1] -
143 smo->c4 * smo->omega_e_est * smo->psi_r_est.dat[0] - smo->z.dat[1];
144 smo->i_s_est.dat[0] += d_i_s_alpha * smo->Ts;
145 smo->i_s_est.dat[1] += d_i_s_beta * smo->Ts;
146
147 // 3. Process the estimated rotor flux through the PLL
148 // 3a. Calculate the PLL error signal. This is the cross product of the estimated
149 // flux vector and a unit vector aligned with the estimated angle.
150 ctrl_gt pll_error = -smo->psi_r_est.dat[0] * sinf(smo->theta_est * CTL_CTRL_CONST_2_PI) +
151 smo->psi_r_est.dat[1] * cosf(smo->theta_est * CTL_CTRL_CONST_2_PI);
152
153 // 3b. PI controller generates the estimated electrical speed (rad/s).
154 smo->omega_e_est = ctl_step_pid_ser(&smo->pid_pll, pll_error);
155
156 // 3c. Integrate the speed to get the next estimated angle.
157 smo->theta_est += smo->omega_e_est * smo->Ts / CTL_CTRL_CONST_2_PI;
158 smo->theta_est = ctrl_mod_1(smo->theta_est); // Wrap angle to [0, 1.0)
159
160 // 4. Update the output interfaces
161 // 4a. Calculate mechanical speed, filter it, and convert to p.u.
162 ctrl_gt omega_m_est =
163 (smo->omega_e_est - (smo->c5 * smo->i_s_est.dat[1] / smo->psi_r_est.dat[0])) / smo->pole_pairs;
164 ctrl_gt speed_filtered = ctl_step_lowpass_filter(&smo->filter_spd, omega_m_est);
165 smo->spdif.speed = speed_filtered * smo->speed_pu_sf;
166
167 // 4b. Provide the estimated electrical and mechanical angles.
168 smo->encif.elec_position = smo->theta_est;
169 smo->encif.position = ctrl_mod_1(smo->theta_est / smo->pole_pairs);
170}
