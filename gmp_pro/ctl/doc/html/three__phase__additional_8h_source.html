142GMP_STATIC_INLINE void ctl_step_dq_hcm(inv_dq_hcm_t* hcm)
143{
144 // Safety Assertions
145 gmp_base_assert(hcm->dq_meas);
146 gmp_base_assert(hcm->dq_set);
147
148 ctrl_gt err_d = hcm->dq_set->dat[phase_d] - hcm->dq_meas->dat[phase_d];
149 ctrl_gt err_q = hcm->dq_set->dat[phase_q] - hcm->dq_meas->dat[phase_q];
150
151 ctrl_gt out_d = 0.0f;
152 ctrl_gt out_q = 0.0f;
153
154 // --- 6th Harmonic (Physical 5th & 7th) ---
155 if (hcm->flag_enable_6th)
156 {
157 out_d += ctl_step_qr_controller(&hcm->qr_d_6th, err_d);
158 out_q += ctl_step_qr_controller(&hcm->qr_q_6th, err_q);
159 }
160
161 // --- 12th Harmonic (Physical 11th & 13th) ---
162 if (hcm->flag_enable_12th)
163 {
164 out_d += ctl_step_qr_controller(&hcm->qr_d_12th, err_d);
165 out_q += ctl_step_qr_controller(&hcm->qr_q_12th, err_q);
166 }
167
168 // --- Saturation (Critical Safety) ---
169 // Limits the total harmonic injection authority
170 hcm->dq_out.dat[phase_d] = ctl_step_saturation(&hcm->sat_d, out_d);
171 hcm->dq_out.dat[phase_q] = ctl_step_saturation(&hcm->sat_q, out_q);
172}

328GMP_STATIC_INLINE void ctl_step_neg_inv_ctrl(inv_neg_ctrl_t* neg)
329{
330 // Safety checks
331 if (!neg->flag_enable_negative_current_ctrl)
332 {
333 ctl_vector2_clear(&neg->vab_out);
334 // Optional: Reset internal states if desired when disabled
335 return;
336 }
337
338 // Check pointers are valid
339 gmp_base_assert(neg->iab);
340 gmp_base_assert(neg->phasor);
341
342 // --- 1. Coordinate Transformation (AlphaBeta -> Neg DQ) ---
343 ctl_ct_park2_neg(neg->iab, neg->phasor, &neg->idqn_raw);
344
345 if (neg->vab != NULL)
346 {
347 ctl_ct_park2_neg(neg->vab, neg->phasor, &neg->vdqn_raw);
348 }
349
350 // --- 2. Filtering (Critical: 2nd Order Biquad) ---
351 // Remove 2*omega ripple to extract DC negative sequence component
352 neg->idqn.dat[phase_d] = ctl_step_biquad_filter(&neg->filter_idqn[phase_d], neg->idqn_raw.dat[phase_d]);
353 neg->idqn.dat[phase_q] = ctl_step_biquad_filter(&neg->filter_idqn[phase_q], neg->idqn_raw.dat[phase_q]);
354
355 if (neg->flag_enable_negative_voltage_ctrl && neg->vab != NULL)
356 {
357 neg->vdqn.dat[phase_d] = ctl_step_biquad_filter(&neg->filter_vdqn[phase_d], neg->vdqn_raw.dat[phase_d]);
358 neg->vdqn.dat[phase_q] = ctl_step_biquad_filter(&neg->filter_vdqn[phase_q], neg->vdqn_raw.dat[phase_q]);
359 }
360
361 // --- 3. Outer Loop: Voltage Control (Optional) ---
362 if (neg->flag_enable_negative_voltage_ctrl)
363 {
364 // Calculate Error: V_set - V_meas
365 ctrl_gt err_vd = neg->vdqn_set.dat[phase_d] - neg->vdqn.dat[phase_d];
366 ctrl_gt err_vq = neg->vdqn_set.dat[phase_q] - neg->vdqn.dat[phase_q];
367
368 neg->idqn_ref_int.dat[phase_d] = ctl_step_pid_ser(&neg->pid_vdqn[phase_d], err_vd);
369 neg->idqn_ref_int.dat[phase_q] = ctl_step_pid_ser(&neg->pid_vdqn[phase_q], err_vq);
370 }
371 else
372 {
373 // Direct Current Suppression (Ref usually 0)
374 neg->idqn_ref_int.dat[phase_d] = neg->idqn_set.dat[phase_d];
375 neg->idqn_ref_int.dat[phase_q] = neg->idqn_set.dat[phase_q];
376 }
377
378 // --- 4. Inner Loop: Current Control ---
379 // Error: I_ref - I_meas
380 ctrl_gt err_id = neg->idqn_ref_int.dat[phase_d] - neg->idqn.dat[phase_d];
381 ctrl_gt err_iq = neg->idqn_ref_int.dat[phase_q] - neg->idqn.dat[phase_q];
382
383 neg->vdqn_out.dat[phase_d] = ctl_step_pid_ser(&neg->pid_idqn[phase_d], err_id);
384 neg->vdqn_out.dat[phase_q] = ctl_step_pid_ser(&neg->pid_idqn[phase_q], err_iq);
385
386 // --- 5. Coordinate Transformation (Neg DQ -> AlphaBeta) ---
387 // Uses the provided optimized function with Positive Phasor
388 ctl_ct_ipark2_neg(&neg->vdqn_out, neg->phasor, &neg->vab_out);
389}
