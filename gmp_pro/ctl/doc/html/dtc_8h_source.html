132GMP_STATIC_INLINE void ctl_step_dtc(ctl_dtc_controller_t* dtc, ctrl_gt i_alpha, ctrl_gt i_beta, ctrl_gt _udc)
133{
134 // 1. Determine the applied alpha-beta voltages from the PREVIOUS cycle's vector.
135 ctrl_gt u_alpha = V_ALPHA_BETA_TABLE[dtc->voltage_vector_index][0] * _udc;
136 ctrl_gt u_beta = V_ALPHA_BETA_TABLE[dtc->voltage_vector_index][1] * _udc;
137
138 // 2. Estimate Stator Flux (Voltage Model Integration).
139 dtc->stator_flux.dat[0] += dtc->ts * (u_alpha - dtc->rs * i_alpha);
140 dtc->stator_flux.dat[1] += dtc->ts * (u_beta - dtc->rs * i_beta);
141
142 // 3. Calculate Flux Magnitude and Electromagnetic Torque.
143 dtc->flux_mag_est = ctl_sqrt(ctl_mul(dtc->stator_flux.dat[0], dtc->stator_flux.dat[0]) +
144 ctl_mul(dtc->stator_flux.dat[1], dtc->stator_flux.dat[1]));
145 dtc->torque_est = 1.5f * dtc->pole_pairs * (dtc->stator_flux.dat[0] * i_beta - dtc->stator_flux.dat[1] * i_alpha);
146
147 // 4. Determine the Flux Vector Sector (1-6).
148 ctrl_gt angle = ctl_atan2(dtc->stator_flux.dat[1], dtc->stator_flux.dat[0]); // Angle in radians
149 if (angle < 0)
150 angle += 2.0f * CTL_CTRL_CONST_PI;
151 dtc->flux_sector = (uint8_t)(angle / (CTL_CTRL_CONST_PI / 3.0f)) + 1;
152 if (dtc->flux_sector > 6)
153 dtc->flux_sector = 6; // Clamp to sector 6
154
155 // 5. Run Hysteresis Controllers.
156 fast_gt flux_status = ctl_step_hysteresis_controller(&dtc->flux_hcc, dtc->flux_mag_est);
157 fast_gt torque_status = ctl_step_hysteresis_controller(&dtc->torque_hcc, dtc->torque_est);
158
159 // 6. Look up the Optimal Voltage Vector from the Switching Table.
160 fast_gt table_col = (flux_status << 1) | torque_status;
161 dtc->voltage_vector_index = DTC_SWITCH_TABLE[dtc->flux_sector - 1][table_col];
162}
