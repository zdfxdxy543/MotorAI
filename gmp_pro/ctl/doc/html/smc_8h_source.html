101GMP_STATIC_INLINE ctrl_gt ctl_step_smc(ctl_smc_t* smc, ctrl_gt input, ctrl_gt input_diff)
102{
103 ctrl_gt etax1;
104 ctrl_gt etax2;
105 ctrl_gt rhos;
106
107 // Calculate the sliding surface value: s = lambda*x1 + x2
108 smc->slide = ctl_mul(smc->lambda, input) + input_diff;
109
110 if (smc->slide > 0)
111 {
112 // Equivalent control part based on state signs
113 etax1 = (input > 0) ? ctl_mul(smc->eta11, input) : ctl_mul(smc->eta12, input);
114 etax2 = (input_diff > 0) ? ctl_mul(smc->eta21, input_diff) : ctl_mul(smc->eta22, input_diff);
115 // Switching control part
116 rhos = smc->rho;
117 }
118 else
119 {
120 // Equivalent control part based on state signs (gains are swapped)
121 etax1 = (input > 0) ? ctl_mul(smc->eta12, input) : ctl_mul(smc->eta11, input);
122 etax2 = (input_diff > 0) ? ctl_mul(smc->eta22, input_diff) : ctl_mul(smc->eta21, input_diff);
123 // Switching control part
124 rhos = -smc->rho;
125 }
126
127 // Total control output
128 smc->output = etax1 + etax2 + rhos;
129
130 return smc->output;
131}
