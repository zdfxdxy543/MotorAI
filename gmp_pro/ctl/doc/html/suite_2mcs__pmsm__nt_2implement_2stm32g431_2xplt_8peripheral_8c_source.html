306void HAL_UARTEx_RxEventCallback(UART_HandleTypeDef* huart, uint16_t Size)
307{
308
309 if (huart == &huart2)
310 {
311 // Stop UART DMA Receive
312 HAL_UART_DMAStop(huart);
313
314 // Copy Data from DMA buffer
315 at_device_rx_isr(&at_dev, (char*)uart_rx_dma_buffer, Size);
316
317 // Clear buffer
318 memset(uart_rx_dma_buffer, 0, UART_RX_DMA_BUFFER_SIZE);
319
320 // Enable UART Receive DMA
321 HAL_UARTEx_ReceiveToIdle_DMA(&huart2, uart_rx_dma_buffer, UART_RX_DMA_BUFFER_SIZE);
322
323 // Close half-full interrupt
324 __HAL_DMA_DISABLE_IT(&hdma_usart2_rx, DMA_IT_HT);
325 }
