91GMP_STATIC_INLINE void ctl_step_pmsm_fo(pmsm_fo_t* fo, const vector2_gt* i_ab_star, const vector2_gt* sin_cos_theta)
92{
93 ctrl_gt i_alpha = i_ab_star->dat[0];
94 ctrl_gt i_beta = i_ab_star->dat[1];
95 ctrl_gt sin_theta = sin_cos_theta->dat[0];
96 ctrl_gt cos_theta = sin_cos_theta->dat[1];
97
98 // Calculate stator flux components (psi_alpha, psi_beta)
99 //tex:
100 // $$\psi_\alpha^* = \omega_b L_s^* I_\alpha^* + \psi_{PM}^* \cos(\theta) $$
101 fo->flux.dat[0] = ctl_mul(fo->wb_ls, i_alpha);
102 fo->flux.dat[0] += ctl_mul(fo->psi_pm, cos_theta);
103
104 //tex:
105 // $$ \psi_\beta^* = \omega_b L_s^*I_\beta^* + \psi_{PM}^* \sin(\theta) $$
106 fo->flux.dat[1] = ctl_mul(fo->wb_ls, i_beta);
107 fo->flux.dat[1] += ctl_mul(fo->psi_pm, sin_theta);
108
109 // Calculate flux magnitude
110 //tex:
111 // $$ \psi^* = \sqrt{\psi_\alpha^{*2} + \psi_\beta^{*2}}$$
112 fo->flux_mag = ctl_vector2_mag(&fo->flux);
113
114 // Calculate electromagnetic torque
115 //tex:
116 // $$ T^* = \frac{\psi_\alpha^* i_\beta^* - \psi_\beta^* i_\alpha^*}{\psi_{PM}} $$
117 ctrl_gt torque_term = ctl_sub(ctl_mul(fo->flux.dat[0], i_beta), ctl_mul(fo->flux.dat[1], i_alpha));
118 fo->torque = ctl_mul(fo->inv_psi_pm, torque_term);
119}
