67void ctl_init_nladrc(ctl_nladrc_controller_t* adrc, parameter_gt fs, parameter_gt r, parameter_gt omega_o,
68 parameter_gt omega_c, parameter_gt b0, parameter_gt alpha1, parameter_gt alpha2)
69{
70 ctrl_gt h = 1.0f / fs;
71
72 // --- Key Point Analysis 1: Initialize Tracking Differentiator (TD) ---
73 // The role of the TD is to arrange a smooth transition for a given input v(t) while providing its differential signal.
74 // v1(k) -> tracks v(t)
75 // v2(k) -> tracks the derivative of v(t)
76 // 'r' is the speed factor that determines the tracking speed. A larger 'r' means faster tracking but may introduce overshoot.
77 adrc->td.x1 = 0.0f;
78 adrc->td.x2 = 0.0f;
79 adrc->td.h = h;
80 adrc->td.r = r;
81
82 // --- Key Point Analysis 2: Initialize Extended State Observer (ESO) ---
83 // The ESO is the core of ADRC. It treats the system as an integrator chain model.
84 // z1 -> tracks the system output y (e.g., motor speed)
85 // z2 -> tracks the derivative of y (e.g., motor acceleration)
86 // z3 -> tracks the "total disturbance" f, which includes all internal and external uncertainties like load, friction, parameter variations, etc.
87 // The observer bandwidth 'omega_o' determines the ESO's response speed and noise immunity. Higher bandwidth means faster tracking but more sensitivity to noise.
88 // The observer gains (beta) are typically configured based on the bandwidth omega_o.
89 adrc->eso.z1 = 0.0f;
90 adrc->eso.z2 = 0.0f;
91 adrc->eso.z3 = 0.0f;
92 adrc->eso.h = h;
93 adrc->eso.b0 = b0; // b0 is the estimate of the plant's control gain, a critical tuning parameter.
94 adrc->eso.beta01 = 3.0f * omega_o;
95 adrc->eso.beta02 = 3.0f * omega_o * omega_o;
96 adrc->eso.beta03 = omega_o * omega_o * omega_o;
97
98 // --- Key Point Analysis 3: Initialize Nonlinear State Error Feedback (NLSEF) ---
99 // NLSEF generates the control signal based on the error between the TD's output (target trajectory) and the ESO's estimates (actual states).
100 // The controller bandwidth 'omega_c' determines the closed-loop response speed.
101 // Kp and Kd are typically configured based on the bandwidth omega_c.
102 // alpha1, alpha2, and delta are parameters for the nonlinear fal() function, giving the controller the characteristic of "high gain for small errors, low gain for large errors".
103 adrc->nlsef.kp = omega_c * omega_c;
104 adrc->nlsef.kd = 2.0f * omega_c;
105 adrc->nlsef.alpha1 = alpha1;
106 adrc->nlsef.alpha2 = alpha2;
107 adrc->nlsef.delta = h * omega_c; // delta can also be configured automatically based on the controller bandwidth.
108
109 adrc->output = 0.0f;
110 adrc->out_max = 1.0f; // Default limits
111 adrc->out_min = -1.0f;
112}
