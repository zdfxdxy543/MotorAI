222GMP_STATIC_INLINE ctl_matrix3_t ctl_matrix3_inv(ctl_matrix3_t mat)
223{
224 ctl_matrix3_t result;
225 ctrl_gt det = ctl_matrix3_det(mat);
226
227 if (fabs(det) < 1e-9) // Avoid division by zero
228 {
229 ctl_matrix3_clear(&result);
230 return result;
231 }
232
233 ctrl_gt inv_det = 1.0f / det;
234
235 result.dat[0] = (mat.dat[4] * mat.dat[8] - mat.dat[5] * mat.dat[7]) * inv_det;
236 result.dat[1] = (mat.dat[2] * mat.dat[7] - mat.dat[1] * mat.dat[8]) * inv_det;
237 result.dat[2] = (mat.dat[1] * mat.dat[5] - mat.dat[2] * mat.dat[4]) * inv_det;
238 result.dat[3] = (mat.dat[5] * mat.dat[6] - mat.dat[3] * mat.dat[8]) * inv_det;
239 result.dat[4] = (mat.dat[0] * mat.dat[8] - mat.dat[2] * mat.dat[6]) * inv_det;
240 result.dat[5] = (mat.dat[2] * mat.dat[3] - mat.dat[0] * mat.dat[5]) * inv_det;
241 result.dat[6] = (mat.dat[3] * mat.dat[7] - mat.dat[4] * mat.dat[6]) * inv_det;
242 result.dat[7] = (mat.dat[1] * mat.dat[6] - mat.dat[0] * mat.dat[7]) * inv_det;
243 result.dat[8] = (mat.dat[0] * mat.dat[4] - mat.dat[1] * mat.dat[3]) * inv_det;
