86GMP_STATIC_INLINE
87void ctl_step_single_phase_H_modulation(single_phase_H_modulation_t* bridge, ctrl_gt u_target, ctrl_gt inverter_current)
88{
89 // 1. Detect current direction with hysteresis to prevent chattering near zero crossing.
90 if (inverter_current > bridge->current_deadband)
91 {
92 bridge->current_dir = 1; // Positive current
93 }
94 else if (inverter_current < -bridge->current_deadband)
95 {
96 bridge->current_dir = -1; // Negative current
97 }
98 // If within the deadband, keep the previous direction.
99
100 // 2. Calculate ideal unipolar PWM duty cycles for each leg.
101 // Duty_L = (1 - u_target) / 2
102 // Duty_N = (1 + u_target) / 2
103 ctrl_gt modulate_target_L = ctl_sat(ctl_div2(-u_target + float2ctrl(1)), 0, float2ctrl(1));
104 ctrl_gt modulate_target_N = ctl_sat(ctl_div2(u_target + float2ctrl(1)), 0, float2ctrl(1));
105
106 // 3. Apply dead-time compensation based on current direction.
107 if (bridge->current_dir == 1) // Positive current: V_actual < V_ideal, need to increase duty cycle.
108 {
109 bridge->phase_L = pwm_mul(modulate_target_L, bridge->pwm_full_scale) + bridge->pwm_deadband;
110 bridge->phase_N = pwm_mul(modulate_target_N, bridge->pwm_full_scale) - bridge->pwm_deadband;
111 }
112 else if (bridge->current_dir == -1) // Negative current: V_actual > V_ideal, need to decrease duty cycle.
113 {
114 bridge->phase_L = pwm_mul(modulate_target_L, bridge->pwm_full_scale) - bridge->pwm_deadband;
115 bridge->phase_N = pwm_mul(modulate_target_N, bridge->pwm_full_scale) + bridge->pwm_deadband;
116 }
117 else // No current or within deadband: no compensation.
118 {
119 bridge->phase_L = pwm_mul(modulate_target_L, bridge->pwm_full_scale);
120 bridge->phase_N = pwm_mul(modulate_target_N, bridge->pwm_full_scale);
121 }
122
123 // 4. Saturate the final compare values to ensure they are within the valid PWM range.
124 bridge->phase_L = pwm_sat(bridge->phase_L, bridge->pwm_full_scale, 0);
125 bridge->phase_N = pwm_sat(bridge->phase_N, bridge->pwm_full_scale, 0);
126}
