239GMP_STATIC_INLINE void ctl_step_pmsm_smo_ctrl(pmsm_smo_controller_t* ctrl)
240{
241 ctl_vector2_t phasor;
242 ctrl_gt etheta;
243
244 if (!ctrl->flag_enable_controller)
245 {
246 // When disabled, ensure PWM is off and states are cleared.
247 if (ctrl->pwm_out)
248 {
249 ctrl->pwm_out->value.dat[phase_A] = 0;
250 ctrl->pwm_out->value.dat[phase_B] = 0;
251 ctrl->pwm_out->value.dat[phase_C] = 0;
252 }
253 ctl_clear_pmsm_smo_ctrl(ctrl);
254 return;
255 }
256
257 // --- State Estimation & Startup ---
258 // Note: Ramp generator and SMO are run unconditionally.
259 // The active angle source is managed by ctl_attach_mtr_position.
260 ctl_step_slope_f(&ctrl->ramp_gen); // Open-loop ramp for startup
261 ctl_step_pmsm_smo(&ctrl->smo, ctrl->vab0_set.dat[phase_alpha], ctrl->vab0_set.dat[phase_beta],
262 ctrl->iab0.dat[phase_alpha], ctrl->iab0.dat[phase_beta]); // SMO for sensorless estimation
263
264 // --- FOC Transformations ---
265 // Clark Transformation
266#if MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 3
267 ctl_ct_clarke(&ctrl->mtr_interface.iabc->value, &ctrl->iab0);
268#elif MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 2
269 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->mtr_interface.iabc->value, (ctl_vector2_t*)&ctrl->iab0);
270#else
271#error("Wrong parameter for macro MTR_CTRL_CURRENT_MEASUREMENT_PHASES")
272#endif
273
274#if MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 3
275 ctl_ct_clarke(&ctrl->mtr_interface.uabc->value, &ctrl->uab0);
276#elif MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 2
277 ctl_ct_clarke_2ph(&ctrl->mtr_interface.uabc->value, &ctrl->uab0);
278#elif MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 0
279 // If no voltage measurement, use the setpoint voltage for the observer
280 ctrl->uab0.dat[phase_alpha] = ctrl->vab0_set.dat[phase_alpha];
281 ctrl->uab0.dat[phase_beta] = ctrl->vab0_set.dat[phase_beta];
282 ctrl->uab0.dat[phase_0] = ctrl->vab0_set.dat[phase_0];
283#else
284#error("Wrong parameter for macro MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES")
285#endif
286
287 // Park Transformation
288 etheta = ctl_get_mtr_elec_theta(&ctrl->mtr_interface);
289 ctl_set_phasor_via_angle(etheta, &phasor);
290 ctl_ct_park(&ctrl->iab0, &phasor, &ctrl->idq0);
291 ctl_ct_park(&ctrl->uab0, &phasor, &ctrl->udq0);
292
293 // --- Control Loops ---
294 // Velocity Controller
295 if (ctrl->flag_enable_velocity_ctrl)
296 {
297 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
298#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
299 ctrl->idq_set.dat[phase_q] =
300 ctl_step_tracking_pid(&ctrl->spd_ctrl, ctrl->speed_set, ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
301 ctrl->idq_ff.dat[phase_q];
302#else // using continuous controller
303 ctrl->idq_set.dat[phase_q] = ctl_step_tracking_continuous_pid(&ctrl->spd_ctrl, ctrl->speed_set,
304 ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
305 ctrl->idq_ff.dat[phase_q];
306#endif
307 }
308 else
309 {
310 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
311 ctrl->idq_set.dat[phase_q] = ctrl->idq_ff.dat[phase_q];
312 }
313
314 // Current Controller
315 if (ctrl->flag_enable_current_ctrl)
316 {
317#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
318 ctrl->vdq_set.dat[phase_d] =
319 ctl_step_discrete_pid(&ctrl->current_ctrl[phase_d], ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
320 ctrl->vdq_ff.dat[phase_d];
321 ctrl->vdq_set.dat[phase_q] =
322 ctl_step_discrete_pid(&ctrl->current_ctrl[phase_q], ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
323 ctrl->vdq_ff.dat[phase_q];
324#else // using continuous controller
325 ctrl->vdq_set.dat[phase_d] =
326 ctl_step_pid_ser(&ctrl->current_ctrl[phase_d], ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
327 ctrl->vdq_ff.dat[phase_d];
328 ctrl->vdq_set.dat[phase_q] =
329 ctl_step_pid_ser(&ctrl->current_ctrl[phase_q], ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
330 ctrl->vdq_ff.dat[phase_q];
331#endif
332 ctrl->vdq_set.dat[phase_0] = 0;
333 }
334 else
335 {
336 ctrl->vdq_set.dat[phase_d] = ctrl->vdq_ff.dat[phase_d];
337 ctrl->vdq_set.dat[phase_q] = ctrl->vdq_ff.dat[phase_q];
338 ctrl->vdq_set.dat[phase_0] = 0;
339 }
340
341 // --- Output Stage ---
342 // Inverse Park Transformation
343 if (ctrl->flag_enable_modulation)
344 {
345 ctl_ct_ipark(&ctrl->vdq_set, &phasor, &ctrl->vab0_set);
346 }
347
348 // SVPWM
349 if (ctrl->flag_enable_output && ctrl->pwm_out)
350 {
351 ctl_ct_svpwm_calc(&ctrl->vab0_set, &ctrl->pwm_out->value);
352 }
353 else if (ctrl->pwm_out)
354 {
355 ctrl->pwm_out->value.dat[phase_A] = 0;
356 ctrl->pwm_out->value.dat[phase_B] = 0;
357 ctrl->pwm_out->value.dat[phase_C] = 0;
358 }
359}

452GMP_STATIC_INLINE void ctl_switch_pmsm_smo_ctrl_using_smo(pmsm_smo_controller_t* ctrl)
453{
454 if ((ctrl->flag_enable_smo) && (!ctrl->flag_switch_cplt))
455 {
456 ctl_vector2_t phasor;
457 vector3_gt udq0_at_switch;
458 vector3_gt idq0_at_switch;
459
460 // 1. Switch angle source from ramp generator to SMO observer
461 ctl_attach_mtr_position(&ctrl->mtr_interface, &ctrl->smo.encif);
462
463 // 2. Set the controller to closed-loop velocity mode
464 ctl_pmsm_smo_ctrl_velocity_mode(ctrl);
465 ctl_set_pmsm_smo_ctrl_speed(ctrl, ctl_mul(ctrl->ramp_gen.target_frequency, ctrl->ramp_freq_spd_set_sf));
466
467 // 3. Calculate current system state for bumpless transfer
468 ctl_set_phasor_via_angle(ctrl->smo.encif.elec_position, &phasor);
469 ctl_ct_park(&ctrl->vab0_set, &phasor, &udq0_at_switch);
470 ctl_ct_park(&ctrl->iab0, &phasor, &idq0_at_switch);
471
472 // 4. Clear any feedforward from the startup phase
473 ctrl->idq_ff.dat[phase_q] = 0;
474 ctrl->idq_ff.dat[phase_d] = 0;
475
476 // 5. Initialize PID controller internal states for bumpless transfer
477#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
478 // Pre-load the output of the PID controllers from the last cycle
479 ctrl->spd_ctrl.pid.output_1 = idq0_at_switch.dat[phase_q];
480 ctrl->current_ctrl[phase_d].output_1 = udq0_at_switch.dat[phase_d];
481 ctrl->current_ctrl[phase_q].output_1 = udq0_at_switch.dat[phase_q];
482#else // using continuous controller \
483 // Pre-load the integrator and output of the PID controllers
484 ctl_set_pid_integrator(&ctrl->spd_ctrl.pid, idq0_at_switch.dat[phase_q]);
485 ctl_set_pid_integrator(&ctrl->current_ctrl[phase_d], udq0_at_switch.dat[phase_d]);
486 ctl_set_pid_integrator(&ctrl->current_ctrl[phase_q], udq0_at_switch.dat[phase_q]);
487#endif // PMSM_CTRL_USING_DISCRETE_CTRL
488
489 ctrl->flag_switch_cplt = 1;
490 }
491}
