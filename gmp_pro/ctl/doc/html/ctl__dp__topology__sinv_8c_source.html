23void ctl_upgrade_sinv_param(sinv_ctrl_t* sinv, sinv_init_t* init)
24{
25 // --- Initialize grid synchronization modules ---

35 ctl_init_lp_filter(&sinv->lpf_ugrid, init->f_ctrl, init->adc_filter_fc);
36
37 // --- Initialize main controllers ---
38 ctl_init_pid_Tmode(&sinv->voltage_pid, init->v_ctrl_kp, init->v_ctrl_Ti, init->v_ctrl_Td, init->f_ctrl);
39 ctl_init_qpr_controller(&sinv->sinv_qpr_base, init->i_ctrl_kp, init->i_ctrl_kr, init->base_freq,
40 init->i_ctrl_cut_freq, init->f_ctrl);

42 // --- Initialize harmonic compensation (Quasi-Resonant controllers) ---
43 ctl_init_qr_controller(&sinv->sinv_qr_3, init->harm_ctrl_kr_3, init->base_freq * 3.0f, init->harm_ctrl_cut_freq_3,

47 ctl_init_qr_controller(&sinv->sinv_qr_7, init->harm_ctrl_kr_7, init->base_freq * 7.0f, init->harm_ctrl_cut_freq_7,
48 init->f_ctrl);
49 ctl_init_qr_controller(&sinv->sinv_qr_9, init->harm_ctrl_kr_9, init->base_freq * 9.0f, init->harm_ctrl_cut_freq_9,
50 init->f_ctrl);
51
52 // --- Initialize AC signal measurement modules ---
53 ctl_init_sine_analyzer(&sinv->ac_current_measure, 0.01f, init->base_freq * 0.8f, init->base_freq * 1.2f,
54 init->base_freq, init->f_ctrl);
55 ctl_init_sine_analyzer(&sinv->ac_voltage_measure, 0.01f, init->base_freq * 0.8f, init->base_freq * 1.2f,
56 init->base_freq, init->f_ctrl);
57}
