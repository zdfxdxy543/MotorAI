108GMP_STATIC_INLINE ctrl_gt ctl_step_im_spd_calc(ctl_im_spd_calc_t* calc, ctrl_gt isd, ctrl_gt isq, ctrl_gt omega_r)
109{
110 // 1. Estimate the rotor magnetizing current (proportional to rotor flux)
111 // This is a discrete-time low-pass filter: imds(k) = (1-a)*imds(k-1) + a*isd(k)
112 calc->imds += ctl_mul(calc->kr, (isd - calc->imds));
113
114 // 2. Calculate the slip speed.
115 // To prevent division by zero at startup, a fallback is used.
116 if (ctl_abs(calc->imds) < float2ctrl(0.005) || ctl_abs(calc->imds) > -float2ctrl(0.005))
117 {
118 // If magnetizing current is near zero, set slip to a high value to avoid instability.
119 calc->slip = float2ctrl(1.0);
120 }
121 else
122 {
123 calc->slip = ctl_div(ctl_mul(calc->kt, isq), calc->imds);
124 }
125
126 // 3. Calculate the stator electrical frequency.
127 // omega_s = omega_r_elec + slip_speed
128 // A small constant boost is added to improve low-speed performance.
129 calc->omega_s = ctl_mul(omega_r, calc->ksync) + calc->slip + float2ctrl(0.03);
130
131 // 4. Integrate the stator frequency to get the electrical angle.
132 calc->enc.elec_position += ctl_mul(calc->ktheta, calc->omega_s);
133
134 // 5. Wrap the angle to the range [0, 1.0)
135 calc->enc.elec_position = ctrl_mod_1(float2ctrl(1.0) + calc->enc.elec_position);
136
137 // The mechanical position is assumed to be the same as electrical for this estimator's purpose.
138 calc->enc.position = calc->enc.elec_position;
139
140 return calc->enc.elec_position;
141}
