61void ctl_init_adc_calibrator(adc_bias_calibrator_t* obj, parameter_gt fc, parameter_gt Q, parameter_gt fs)
62{
63 uint32_t total_period = (uint32_t)(10 * fc);
64
65 // setup the filter
66 ctl_init_biquad_lpf(&obj->filter, fs, fc, Q);
67
68 // at least 1000 period
69 if (total_period > 1000)
70 obj->filter_period = total_period;
71 else
72 obj->filter_period = 1000;
73
74 // init parameters
75 obj->filter_tick = 0;
76 obj->raw = 0;
77
