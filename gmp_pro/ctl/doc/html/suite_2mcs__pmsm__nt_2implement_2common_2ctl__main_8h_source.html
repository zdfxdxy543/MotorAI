83GMP_STATIC_INLINE void ctl_dispatch(void)
84{
85 // ADC calibrator routine
86 if (flag_enable_adc_calibrator)
87 {
88 if (index_adc_calibrator == 7)
89 ctl_step_adc_calibrator(&adc_calibrator, idc.control_port.value);
90 else if (index_adc_calibrator == 6)
91 ctl_step_adc_calibrator(&adc_calibrator, udc.control_port.value);
92 else if (index_adc_calibrator <= 5 && index_adc_calibrator >= 3)
93 ctl_step_adc_calibrator(&adc_calibrator, uuvw.control_port.value.dat[index_adc_calibrator - 3]);
94 else if (index_adc_calibrator <= 2)
95 ctl_step_adc_calibrator(&adc_calibrator, iuvw.control_port.value.dat[index_adc_calibrator]);
96 }
97
98 // normal controller routine
99 else
100 {
101 // ramp generator
102 ctl_step_slope_f_pu(&rg);
103
104 // Calculate Motor Speed
105 ctl_step_spd_calc(&spd_enc);
106
107#if BUILD_LEVEL > 3
108 // motion controller
109 ctl_step_vel_pos_ctrl(&motion_ctrl);
110
111 // current command dispatch
112 ctl_set_mtr_current_ctrl_ref(&mtr_ctrl, 0, ctl_get_vel_pos_cmd(&motion_ctrl));
113
114#endif
115
116 // motor current controller
117 ctl_step_current_controller(&mtr_ctrl);
118
119#ifdef ENABLE_SMO
120 ctrl_gt udc_for_smo = ctl_mul(CTL_CTRL_CONST_1_OVER_SQRT3, mtr_ctrl.udc);
121
122#if (PWM_MODULATOR_USING_NEGATIVE_LOGIC == 1)
123 ctrl_gt v_alpha = ctl_mul(udc_for_smo, - mtr_ctrl.vab0.dat[phase_alpha]);
124 ctrl_gt v_beta = ctl_mul(udc_for_smo, - mtr_ctrl.vab0.dat[phase_beta]);
125#else
126 ctrl_gt v_alpha = ctl_mul(udc_for_smo, mtr_ctrl.vab0.dat[phase_alpha]);
127 ctrl_gt v_beta = ctl_mul(udc_for_smo, mtr_ctrl.vab0.dat[phase_beta]);
128#endif
129 ctl_step_pmsm_smo(
130 // SMO object
131 &smo,
132 // uab
133 v_alpha, v_beta,
134 // iab
135 mtr_ctrl.iab0.dat[phase_alpha], mtr_ctrl.iab0.dat[phase_beta]);
136#endif // ENABLE_SMO
137
138#ifdef ENABLE_MOTOR_FAULT_PROTECTION
139 // Motor protection callback, fast task
140 if (ctl_step_mtr_protect_fast(&protection))
141 {
142 cia402_fault_request(&cia402_sm);
143 }
144#endif // ENABLE_MOTOR_FAULT_PROTECTION
145
146 // mix all output
147 spwm.vab0_out.dat[phase_alpha] = mtr_ctrl.vab0.dat[phase_alpha];
148 spwm.vab0_out.dat[phase_beta] = mtr_ctrl.vab0.dat[phase_beta];
149 spwm.vab0_out.dat[phase_0] = mtr_ctrl.vab0.dat[phase_0];
150
151 // modulation
152#if defined USING_NPC_MODULATOR
153 ctl_step_npc_modulator(&spwm);
154#else
155 ctl_step_svpwm_modulator(&spwm);
156#endif // USING_NPC_MODULATOR
157 }
158}
