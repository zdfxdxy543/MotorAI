205GMP_STATIC_INLINE ctl_matrix2_t ctl_matrix2_inv(ctl_matrix2_t mat)
206{
207 ctl_matrix2_t result;
208 ctrl_gt det = ctl_matrix2_det(mat);
209
210 if (fabs(det) < 1e-9) // Avoid division by zero
211 {
212 ctl_matrix2_clear(&result);
213 return result;
214 }
215
216 ctrl_gt inv_det = 1.0f / det;
217 result.dat[0] = mat.dat[3] * inv_det;
218 result.dat[1] = -mat.dat[1] * inv_det;
219 result.dat[2] = -mat.dat[2] * inv_det;
220 result.dat[3] = mat.dat[0] * inv_det;
221 return result;
222}

275GMP_STATIC_INLINE void ctl_matrix2_eigenvalues_complex(ctl_matrix2_t mat, ctl_complex_t* eigenvalue1,
276 ctl_complex_t* eigenvalue2)
277{
278 ctrl_gt trace = ctl_matrix2_trace(mat);
279 ctrl_gt det = ctl_matrix2_det(mat);
280 ctrl_gt discriminant = trace * trace - 4 * det;
281
282 if (discriminant >= 0)
283 {
284 // Real eigenvalues
285 ctrl_gt sqrt_discriminant = ctl_sqrt(discriminant);
286 eigenvalue1->real = (trace + sqrt_discriminant) / 2.0f;
287 eigenvalue1->imag = 0.0f;
288 eigenvalue2->real = (trace - sqrt_discriminant) / 2.0f;
289 eigenvalue2->imag = 0.0f;
290 }
291 else
292 {
293 // Complex conjugate eigenvalues
294 ctrl_gt sqrt_abs_discriminant = ctl_sqrt(-discriminant);
295 eigenvalue1->real = trace / 2.0f;
296 eigenvalue1->imag = sqrt_abs_discriminant / 2.0f;
297 eigenvalue2->real = trace / 2.0f;
298 eigenvalue2->imag = -sqrt_abs_discriminant / 2.0f;
299 }
300}

310GMP_STATIC_INLINE void ctl_matrix2_eigenvectors_real(ctl_matrix2_t mat, ctl_vector2_t eigenvalues,
311 ctl_vector2_t* eigenvector1, ctl_vector2_t* eigenvector2)
312{
313 ctrl_gt a = mat.dat[0];
314 ctrl_gt b = mat.dat[1];
315 ctrl_gt c = mat.dat[2];
316 ctrl_gt d = mat.dat[3];
317
318 // Calculate eigenvector for the first eigenvalue
319 if (fabs(c) > 1e-9)
320 {
321 eigenvector1->dat[0] = eigenvalues.dat[0] - d;
322 eigenvector1->dat[1] = c;
323 }
324 else if (fabs(b) > 1e-9)
325 {
326 eigenvector1->dat[0] = b;
327 eigenvector1->dat[1] = eigenvalues.dat[0] - a;
328 }
329 else
330 { // Diagonal matrix
331 eigenvector1->dat[0] = 1;
332 eigenvector1->dat[1] = 0;
333 }
334
335 // Calculate eigenvector for the second eigenvalue
336 if (fabs(c) > 1e-9)
337 {
338 eigenvector2->dat[0] = eigenvalues.dat[1] - d;
339 eigenvector2->dat[1] = c;
340 }
341 else if (fabs(b) > 1e-9)
342 {
343 eigenvector2->dat[0] = b;
344 eigenvector2->dat[1] = eigenvalues.dat[1] - a;
345 }
346 else
347 { // Diagonal matrix
348 eigenvector2->dat[0] = 0;
349 eigenvector2->dat[1] = 1;
350 }
351
352 // Normalize the eigenvectors
353 ctl_vector2_normalize(eigenvector1, eigenvector1);
354 ctl_vector2_normalize(eigenvector2, eigenvector2);
355}
