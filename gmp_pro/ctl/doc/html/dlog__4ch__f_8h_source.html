119static inline void DLOG_4CH_F_FUNC(DLOG_4CH_F* v)
120{
121 switch (v->status)
122 {
123 case 1: // Wait for trigger
124 if (((*v->input_ptr1) > v->trig_value) && (v->prev_value < v->trig_value))
125 {
126 // Rising edge detected, start logging data.
127 v->status = 2;
128 }
129 break;
130
131 case 2: // Logging active
132 v->skip_count++;
133 if (v->skip_count == v->pre_scalar)
134 {
135 v->skip_count = 0;
136 v->output_ptr1[v->count] = *v->input_ptr1;
137 v->output_ptr2[v->count] = *v->input_ptr2;
138 v->output_ptr3[v->count] = *v->input_ptr3;
139 v->output_ptr4[v->count] = *v->input_ptr4;
140 v->count++;
141
142 if (v->count == v->size)
143 {
144 // Buffer is full, reset and wait for the next trigger.
145 v->count = 0;
146 v->status = 1;
147 }
148 }
149 break;
150 }
151
152 v->prev_value = *v->input_ptr1;
153}
