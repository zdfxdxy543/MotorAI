202void ctl_mainloop(void)
203{
204 // Get current System Tick
205 time_gt current_tick = gmp_base_get_system_tick();
206
207 //
208 // Judge if PWM is enabled
209 //
210 if(acm_ctrl.flag_enable_output)
211 {
212 ctl_enable_output();
213 }
214 else
215 {
216 ctl_disable_output();
217 }
218
219 //
220 // ADC Auto calibrate
221 //
222 if (flag_enable_adc_calibrator)
223 {
224 if (ctl_is_adc_calibrator_cmpt(&adc_calibrator) && ctl_is_adc_calibrator_result_valid(&adc_calibrator))
225 {
226 // set_adc_bias_via_channel(index_adc_calibrator, ctl_get_adc_calibrator_result(&adc_calibrator));
227
228 if (index_adc_calibrator == 3) // dc bus calibrate
229 {
230 idc.bias = idc.bias + ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), idc.gain);
231
232 flag_enable_adc_calibrator = 0;
233
234 // enable pmsm controller
235 ctl_enable_acm_sensored_ctrl_output(&acm_ctrl);
236 }
237 // index_adc_calibrator == 2 ~ 0, for Iabc
238 else
239 {
240 // iabc get result
241 iabc.bias[index_adc_calibrator] =
242 iabc.bias[index_adc_calibrator] +
243 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), iabc.gain[index_adc_calibrator]);
244
245 // move to next position
246 index_adc_calibrator += 1;
247
248 // clear calibrator
249 ctl_clear_adc_calibrator(&adc_calibrator);
250
251 // enable calibrator to next position
252 ctl_enable_adc_calibrator(&adc_calibrator);
253 }
254
255 if (index_adc_calibrator > MTR_ADC_IDC)
256 flag_enable_adc_calibrator = 0;
257 }
258 }
259
260 //
261 // Speed Loop switch logic
262 // Current Loop -> Speed Loop
263 //
264
265 if(acm_ctrl.speed_set - ctl_get_mtr_velocity(&acm_ctrl.mtr_interface) < ctl_mul(float2ctrl(0.1),acm_ctrl.speed_set) )
266 {
267 ctl_acm_sensored_ctrl_velocity_mode(&acm_ctrl);
268 }
269
270
271 //
272 // Add User Control Process Logic here
273 //
274
275 return;
276}
