129GMP_STATIC_INLINE fast_gt ctl_mtr_protect_debounce(fast_gt condition, uint16_t* counter, uint16_t limit)
130{
131 if (condition)
132 {
133 // Increment (Saturate at limit + 1 to prevent overflow)
134 if (*counter <= limit)
135 (*counter)++;
136 }
137 else
138 {
139 // Decrement
140 if (*counter > 0)
141 (*counter)--;
142 }
143
144 return (*counter > limit);
145}

181GMP_STATIC_INLINE fast_gt ctl_step_mtr_protect_fast(ctl_mtr_protect_t* prot)
182{
183 // 1. Latch Check
184 if ((prot->error_code.all & (~prot->error_mask.all)) != MTR_PROT_NONE)
185 return 1;
186
187 // 2. Load Inputs
188 ctrl_gt u_dc = *(prot->ptr_udc);
189 ctrl_gt id = prot->ptr_idq->dat[0];
190 ctrl_gt iq = prot->ptr_idq->dat[1];
191
192 // 3. Check Over Voltage (Critical)
193 // Condition: u_dc > limit
194 if (ctl_mtr_protect_debounce(ctl_is_greater(u_dc, prot->limit_ov_pu), &prot->cnt_ov, prot->limit_cnt_ov))
195 {
196 prot->error_code.bit.over_voltage = 1;
197
198 if (!prot->error_mask.bit.over_voltage)
199 {
200 return 1; // Trip if not masked
201 }
202 }
203
204 // 4. Check Software Over Current
205 // Condition: (id^2 + iq^2) > limit^2
206 ctrl_gt i_sq = ctl_mul(id, id) + ctl_mul(iq, iq);
207 if (ctl_mtr_protect_debounce(ctl_is_greater(i_sq, prot->limit_oc_sq_pu), &prot->cnt_oc, prot->limit_cnt_oc))
208 {
209 prot->error_code.bit.over_current = 1;
210
211 if (!prot->error_mask.bit.over_current)
212 {
213 return 1;
214 }
215 }
216
217 // 5. Check Control Deviation (Runaway)
218 // Condition: |I_ref - I_meas|^2 > limit
219 ctrl_gt id_ref = prot->ptr_ref->dat[0];
220 ctrl_gt iq_ref = prot->ptr_ref->dat[1];
221
222 ctrl_gt err_d = id_ref - id;
223 ctrl_gt err_q = iq_ref - iq;
224 ctrl_gt err_sq = ctl_mul(err_d, err_d) + ctl_mul(err_q, err_q);
225
226 if (ctl_mtr_protect_debounce(ctl_is_greater(err_sq, prot->limit_dev_sq_pu), &prot->cnt_dev, prot->limit_cnt_dev))
227 {
228 prot->error_code.bit.deviation = 1;
229
230 if (!prot->error_mask.bit.deviation)
231 {
232 return 1;
233 }
234 }
235
236 return 0; // Safe
237}
