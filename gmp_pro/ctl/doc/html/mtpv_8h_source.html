96GMP_STATIC_INLINE void ctl_step_mtpv(ctl_mtpv_controller_t* mtpv, ctrl_gt iq_cmd, ctrl_gt u_max, ctrl_gt omega_e)
97{
98 ctrl_gt we_sq = omega_e * omega_e;
99 ctrl_gt u_max_sq = u_max * u_max;
100
101 // --- Calculate terms for the quadratic equation of Id ---
102 // The voltage equation can be rearranged into a quadratic form: A*id^2 + B*id + C = 0
103 ctrl_gt A = we_sq * mtpv->ld_sq + mtpv->rs_sq;
104 ctrl_gt B = 2.0f * we_sq * mtpv->ld * mtpv->psi_f;
105 ctrl_gt C = we_sq * mtpv->psi_f * mtpv->psi_f + we_sq * mtpv->lq_sq * iq_cmd * iq_cmd +
106 mtpv->rs_sq * iq_cmd * iq_cmd + 2.0f * mtpv->rs * omega_e * mtpv->psi_f * iq_cmd - u_max_sq;
107
108 // --- Solve the quadratic equation for Id ---
109 ctrl_gt discriminant = B * B - 4.0f * A * C;
110
111 if (discriminant >= 0.0f)
112 {
113 // A valid solution exists. We choose the negative Id for field weakening.
114 mtpv->id_ref = (-B + sqrtf(discriminant)) / (2.0f * A);
115 }
116 else
117 {
118 // No real solution, indicates extreme condition.
119 // Clamp Id to a safe negative value based on flux.
120 mtpv->id_ref = -mtpv->psi_f / mtpv->ld;
121 }
122
123 // The q-axis current is the original command from the speed loop.
124 // In a full implementation, Iq might also be adjusted to stay within the current limit circle.
125 mtpv->iq_ref = iq_cmd;
126}
