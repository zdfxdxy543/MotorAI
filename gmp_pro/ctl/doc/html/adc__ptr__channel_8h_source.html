95GMP_STATIC_INLINE ctrl_gt ctl_step_ptr_adc_channel(ptr_adc_channel_t* adc_obj)
96{
97 // Ensure the raw pointer is valid.
98 gmp_base_assert(adc_obj->raw);
99
100 adc_gt raw = *adc_obj->raw;
101 ctrl_gt raw_data;
102
103#if defined CTRL_GT_IS_FIXED
104 // For fixed-point, shift the raw data to the correct IQ format.
105 raw_data = raw << adc_obj->shift_index;
106#elif defined CTRL_GT_IS_FLOAT
107 // For floating-point, scale the raw data to a per-unit value.
108 raw_data = (ctrl_gt)raw * adc_obj->per_unit_base;
109#else
110#error "The system does not specify ctrl_gt is float or fixed. You should define CTRL_GT_IS_FLOAT or CTRL_GT_IS_FIXED."
111#endif // CTRL_GT_IS_XXX
112
113 ctrl_gt raw_without_bias = raw_data - adc_obj->bias;
114 adc_obj->control_port.value = ctl_mul(raw_without_bias, adc_obj->gain);
115
116 return adc_obj->control_port.value;
117}
