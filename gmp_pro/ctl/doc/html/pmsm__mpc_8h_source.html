113GMP_STATIC_INLINE void ctl_step_mpc(ctl_mpc_controller_t* mpc, ctl_vector2_t* idq_ref, ctl_vector2_t* idq_fbk,
114 ctl_vector2_t* phasor, ctrl_gt omega_e, ctrl_gt _udc)
115{
116 ctl_matrix2_t A_full;
117 ctl_vector2_t E;
118 ctl_vector2_t i_pred;
119 ctrl_gt min_cost = 1e12f; // Initialize with a very large number
120 uint8_t best_vector_idx = 0;
121 ctrl_gt voltage_scale = (2.0f / 3.0f) * _udc;
122 fast_gt i;
123
124 // 1. Construct the speed-dependent parts of the model matrices
125 A_full = mpc->A_const;
126 ctl_matrix2_set(&A_full, 0, 1, mpc->Ts * omega_e * mpc->Lq / mpc->Ld);
127 ctl_matrix2_set(&A_full, 1, 0, -mpc->Ts * omega_e * mpc->Ld / mpc->Lq);
128
129 E.dat[0] = 0.0f;
130 E.dat[1] = -mpc->Ts * omega_e * mpc->psi_f / mpc->Lq;
131
132 // 2. Iterate through all 8 possible voltage vectors
133 for (i = 0; i < 8; ++i)
134 {
135 // Get the test voltage vector in the alpha-beta frame
136 ctl_vector2_t u_ab;
137 ctl_vector2_scale(&u_ab, &MPC_VOLTAGE_VECTORS_NORMALIZED[i], voltage_scale);
138
139 // Transform the test voltage to the d-q frame
140 ctl_vector2_t u_dq;
141 ctl_ct_park2(&u_ab, phasor, &u_dq);
142
143 // 3. Predict the next state: i(k+1) = A*i(k) + B*u(k) + E
144 ctl_vector2_t Ax = ctl_matrix2_mul_vector(A_full, *idq_fbk);
145 ctl_vector2_t Bu = ctl_matrix2_mul_vector(mpc->B, u_dq);
146 ctl_vector2_t sumup;
147 ctl_vector2_add(&sumup, &Ax, &Bu);
148 ctl_vector2_add(&i_pred, &sumup, &E);
149
150 // 4. Calculate the cost function:
151 //tex:
152 // $$ J = (id_ref - id_pred)^2 + (iq_ref - iq_pred)^2 $$
153 ctl_vector2_t error_vec;
154 ctl_vector2_sub(&error_vec, idq_ref, &i_pred);
155 ctrl_gt cost = ctl_vector2_mag_sq(&error_vec);
156
157 // 5. Find the vector that minimizes the cost
158 if (cost < min_cost)
159 {
160 min_cost = cost;
161 best_vector_idx = (uint8_t)i;
162 }
163 }
164
165 // 6. Set the optimal vector index as the output
166 mpc->optimal_vector_index = best_vector_idx;
167}
