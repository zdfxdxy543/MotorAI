140GMP_STATIC_INLINE ctrl_gt ctl_step_discrete_pid(discrete_pid_t* ctrl, ctrl_gt input)
141{
142 ctrl->input = input;
143
144 // Calculate output based on the difference equation:
145 // y[k] = y[k-1] + b0*u[k] + b1*u[k-1] + b2*u[k-2]
146#ifdef _USE_DEBUG_DISCRETE_PID
147 // WARNING: Debug logic differs significantly from standard implementation.
148 // It adds an extra proportional term and a decayed output feedback.
149 // This may lead to unexpected behavior. Please verify correctness.
150 ctrl->output = ctl_mul(ctrl->kp, ctrl->input);
151 ctrl->output += ctl_mul(ctrl->b0, ctrl->input);
152 ctrl->output += ctl_mul(ctrl->b1, ctrl->input_1);
153 ctrl->output += ctl_mul(ctrl->b2, ctrl->input_2);
154 ctrl->output += ctl_mul(float2ctrl(0.95), ctrl->output_1);
155#else
156 // Standard mode
157 ctrl->output = ctl_mul(ctrl->b0, ctrl->input);
158 ctrl->output += ctl_mul(ctrl->b1, ctrl->input_1);
159 ctrl->output += ctl_mul(ctrl->b2, ctrl->input_2);
160 ctrl->output += ctrl->output_1;
161#endif // _USE_DEBUG_DISCRETE_PID
162
163 // Handle output saturation
164 ctrl->output = ctl_sat(ctrl->output, ctrl->output_max, ctrl->output_min);
165
166 // Update historical state variables for the next step
167 ctrl->input_2 = ctrl->input_1;
168 ctrl->input_1 = ctrl->input;
169 ctrl->output_1 = ctrl->output;
170
171 return ctrl->output;
172}
