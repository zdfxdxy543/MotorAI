227GMP_STATIC_INLINE void ctl_step_pmsm_hfi_ctrl(pmsm_hfi_controller_t* ctrl)
228{
229 ctl_vector2_t phasor;
230 ctrl_gt etheta;
231 ctrl_gt vq_limit = float2ctrl(1.0);
232
233 ctrl->isr_tick += 1;
234
235 if (ctrl->flag_enable_controller)
236 {
237 // Clark Transformation: (abc -> alpha-beta)
238#if MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 3
239 ctl_ct_clarke(&ctrl->mtr_interface.iabc->value, &ctrl->iab0);
240#elif MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 2
241 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->mtr_interface.iabc->value, (ctl_vector2_t*)&ctrl->iab0);
242#else
243#error("Wrong parameter for macro MTR_CTRL_CURRENT_MEASUREMENT_PHASES")
244#endif
245
246#if MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 3
247 ctl_ct_clarke(&ctrl->mtr_interface.uabc->value, &ctrl->uab0);
248#elif MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 2
249 ctl_ct_clarke_2ph(&ctrl->mtr_interface.uabc->value, &ctrl->uab0);
250#elif MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 0
251 ctl_vector3_clear(&ctrl->uab0);
252#else
253#error("Wrong parameter for macro MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES")
254#endif
255
256 // Park Transformation: (alpha-beta -> d-q)
257 etheta = ctl_get_mtr_elec_theta(&ctrl->mtr_interface);
258 ctl_set_phasor_via_angle(etheta, &phasor);
259 ctl_ct_park(&ctrl->iab0, &phasor, &ctrl->idq0);
260
261#if MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES != 0
262 ctl_ct_park(&ctrl->uab0, &phasor, &ctrl->udq0);
263#else
264 ctl_vector3_clear(&ctrl->udq0); // Should clear udq0, not uab0
265#endif
266
267 // Position Controller
268 if (ctrl->flag_enable_position_ctrl)
269 {
270 // Position control logic is not yet implemented.
271 }
272
273 // Velocity Controller
274 if (ctrl->flag_enable_velocity_ctrl)
275 {
276 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
277#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
278 ctrl->idq_set.dat[phase_q] =
279 ctl_step_tracking_pid(&ctrl->spd_ctrl, ctrl->speed_set, ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
280 ctrl->idq_ff.dat[phase_q];
281#else // using continuous controller
282 ctrl->idq_set.dat[phase_q] = ctl_step_tracking_continuous_pid(&ctrl->spd_ctrl, ctrl->speed_set,
283 ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
284 ctrl->idq_ff.dat[phase_q];
285#endif
286
287#if MTR_CTRL_FEEDFORWARD_STRATEGY == 1
288 // Decoupling logic is commented out.
289 // ctl_mtr_pmsm_decouple(&ctrl->vdq_set, &ctrl->idq_set, ctrl->Ld, ctrl->Lq,
290 // ctl_get_mtr_velocity(ctrl->mtr_interface), ctrl->psi_e);
291#endif
292 }
293 else
294 {
295 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
296 ctrl->idq_set.dat[phase_q] = ctrl->idq_ff.dat[phase_q];
297 }
298
299 // Current Controller
300 if (ctrl->flag_enable_current_ctrl)
301 {
302#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
303 ctrl->vdq_set.dat[phase_d] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_d],
304 ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
305 ctrl->vdq_ff.dat[phase_d];
306
307 // Note: Dynamic saturation for Vq is commented out in discrete implementation.
308 //vq_limit = ctl_sqrt(float2ctrl(1.0) - ctl_mul(ctrl->vdq_set.dat[phase_d], ctrl->vdq_set.dat[phase_d]));
309 //ctl_set_discrete_pid_limit(&ctrl->current_ctrl[phase_q], vq_limit, -vq_limit);
310
311 ctrl->vdq_set.dat[phase_q] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_q],
312 ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
313 ctrl->vdq_ff.dat[phase_q];
314#else // using continuous controller
315 ctrl->vdq_set.dat[phase_d] =
316 ctl_step_pid_ser(&ctrl->current_ctrl[phase_d], ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
317 ctrl->vdq_ff.dat[phase_d];
318
319 vq_limit = ctl_sqrt(float2ctrl(1.0) - ctl_mul(ctrl->vdq_set.dat[phase_d], ctrl->vdq_set.dat[phase_d]));
320 ctl_set_pid_limit(&ctrl->current_ctrl[phase_q], vq_limit, -vq_limit);
321
322 ctrl->vdq_set.dat[phase_q] =
323 ctl_step_pid_ser(&ctrl->current_ctrl[phase_q], ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
324 ctrl->vdq_ff.dat[phase_q];
325#endif
326 ctrl->vdq_set.dat[phase_0] = 0;
327 }
328 else
329 {
330 ctrl->vdq_set.dat[phase_d] = ctrl->vdq_ff.dat[phase_d];
331 ctrl->vdq_set.dat[phase_q] = ctrl->vdq_ff.dat[phase_q];
332 ctrl->vdq_set.dat[phase_0] = 0;
333 }
334
335 // Inverse Park Transformation & Modulation
336 if (ctrl->flag_enable_modulation)
337 {
338 ctl_ct_ipark(&ctrl->vdq_set, &phasor, &ctrl->vab0_set);
339 }
340 // else: vab0_set will be used directly from user input.
341
342 // SVPWM Output Stage
343 if (ctrl->flag_enable_output)
344 {
345 ctl_ct_svpwm_calc(&ctrl->vab0_set, &ctrl->pwm_out->value);
346 }
347 else
348 {
349 // Set PWM to 50% duty cycle for a high-impedance state.
350 ctrl->pwm_out->value.dat[phase_A] = float2ctrl(0.5);
351 ctrl->pwm_out->value.dat[phase_B] = float2ctrl(0.5);
352 ctrl->pwm_out->value.dat[phase_C] = float2ctrl(0.5);
353 }
354 }
355 else // Controller is disabled
356 {
357 // Set PWM to 50% duty cycle for a high-impedance state.
358 ctrl->pwm_out->value.dat[phase_A] = float2ctrl(0.5);
359 ctrl->pwm_out->value.dat[phase_B] = float2ctrl(0.5);
360 ctrl->pwm_out->value.dat[phase_C] = float2ctrl(0.5);
361
362 ctl_clear_pmsm_hfi_ctrl(ctrl);
363 }
364}
