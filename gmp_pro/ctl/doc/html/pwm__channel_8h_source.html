73GMP_STATIC_INLINE pwm_gt ctl_step_pwm_channel(pwm_channel_t* pwm_obj, ctrl_gt raw)
74{
75 if (raw <= 0)
76 pwm_obj->raw.value = 0;
77 else
78 pwm_obj->raw.value = raw;
79
80 pwm_obj->value = pwm_mul(pwm_obj->raw.value, pwm_obj->full_scale) + pwm_obj->phase;
81 pwm_obj->value = pwm_sat(pwm_obj->value, pwm_obj->full_scale, 0);
82 return pwm_obj->value;
83}
