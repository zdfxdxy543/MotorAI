99GMP_STATIC_INLINE ctrl_gt ctl_step_tracking_continuous_pid(ctl_tracking_continuous_pid_t* tp, ctrl_gt target,
100 ctrl_gt fbk)
101{
102 // Check if it's time to execute the controller based on the division factor
103 if (ctl_step_divider(&tp->div))
104 {
105 // If so, first update the trajectory by one step towards the target
106 ctl_step_slope_limiter(&tp->traj, target);
107 // Then, execute the PID controller with the error between the current
108 // trajectory point and the feedback value. The series form is used here.
109 return ctl_step_pid_ser(&tp->pid, tp->traj.out - fbk);
110 }
111
112 // If it's not time to execute, just return the last held PID output value.
113 return tp->pid.out;
114}
