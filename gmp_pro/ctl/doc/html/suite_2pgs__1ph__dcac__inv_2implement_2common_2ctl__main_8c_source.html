217void ctl_mainloop(void)
218{
219 // When the program is reach here, the following things will happen:
220 // 1. software non-block delay 500ms
221 // 2. judge if spll theta error convergence has occurred
222 // 3. then enable system
223
224 if (flag_system_enable)
225 {
226
227 // first time flag
228 // log the first time enable the system
229 if (firsttime_flag == 0)
230 {
231 tick_bias = gmp_base_get_system_tick();
232 firsttime_flag = 1;
233 }
234
235 // a delay of 500ms
236 if ((started_flag == 0) && ((gmp_base_get_system_tick() - tick_bias) > 500) && (startup_flag == 0))
237 {
238 startup_flag = 1;
239 }
240
241 // judge if PLL is close to target
242 if ((started_flag == 0) && (startup_flag == 1) && ctl_ready_mainloop())
243 {
244 ctl_enable_output();
245 started_flag = 1;
246 }
247 }
248 // if system is disabled
249 else // (flag_system_enable == 0)
250 {
251 ctl_disable_output();
252
253 // clear all flags
254 firsttime_flag = 0;
255 startup_flag = 0;
256 started_flag = 0;
257 tick_bias = 0;
258 }
259
260 return;
261}
262
263// This mainloop will run again and again to judge if system meets online condition,
264// when flag_system_enable is set.
265// if return 1 the system is ready to enable.
266// if return 0 the system is not ready to enable
267fast_gt ctl_ready_mainloop(void)
268{
269
270 ctl_clear_sinv(&sinv_ctrl);
271
272 if (ctl_is_sinv_inverter_mode(&sinv_ctrl))
273 return 1;
274 else
275
276 return (ctl_abs(ctl_get_spll_error_fbk(&sinv_ctrl.spll)) < CTRL_SPLL_EPSILON);
277}
