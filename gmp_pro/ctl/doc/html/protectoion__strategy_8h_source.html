108GMP_STATIC_INLINE fast_gt ctl_step_vip_protection(std_vip_protection_t* obj)
109{
110 // If an error has already occurred, latch it and do nothing further.
111 if (obj->flag_error != 0)
112 {
113 return obj->flag_error;
114 }
115
116 // Filter the current and voltage measurements.
117 obj->iout = ctl_step_lowpass_filter(&obj->current_filter, obj->adc_io->value);
118 obj->uout = ctl_step_lowpass_filter(&obj->voltage_filter, obj->adc_uo->value);
119
120 // Calculate and filter the output power.
121 obj->pout = ctl_step_lowpass_filter(&obj->power_filter, ctl_mul(obj->iout, obj->uout));
122
123 // Check if any of the protection limits have been exceeded.
124 if ((obj->iout > obj->current_max) || (obj->uout > obj->voltage_max) || (obj->pout > obj->power_max))
125 {
126 obj->flag_error = 1;
127 }
128
129 return obj->flag_error;
130}

211GMP_STATIC_INLINE fast_gt ctl_step_foldback_protection(std_foldback_protection_t* obj)
212{
213 // Filter the current and voltage measurements.
214 obj->iout = ctl_step_lowpass_filter(&obj->current_filter, obj->adc_io->value);
215 obj->uout = ctl_step_lowpass_filter(&obj->voltage_filter, obj->adc_uo->value);
216
217 // Calculate the new dynamic current limit based on the output voltage.
218 if (obj->uout >= obj->voltage_rated)
219 {
220 // Normal operation: limit is the maximum allowed current.
221 obj->current_limit_output = obj->current_max;
222 }
223 else if (obj->uout > 0)
224 {
225 // Foldback region: linearly decrease the current limit as voltage drops.
226 // Equation of the line: I_limit = slope * V_out + I_short_circuit
227 obj->current_limit_output = ctl_mul(obj->slope, obj->uout);
228 obj->current_limit_output += obj->current_short_circuit;
229 }
230 else
231 {
232 // Short-circuit condition: limit is the minimum short-circuit current.
233 obj->current_limit_output = obj->current_short_circuit;
234 }
235
236 // Ensure the calculated limit does not exceed the absolute maximum.
237 if (obj->current_limit_output > obj->current_max)
238 {
239 obj->current_limit_output = obj->current_max;
240 }
241
242 // Check if the actual measured current is exceeding the newly calculated limit.
243 if (obj->iout > obj->current_limit_output)
244 {
245 obj->flag_overcurrent = 1;
246 }
247
248 // The primary output of this function is the dynamic limit itself.
249 return obj->flag_overcurrent;
250}
