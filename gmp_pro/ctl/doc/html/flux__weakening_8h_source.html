59void ctl_init_fw_ctrl(ctl_fw_ctrl_t* fw, ctrl_gt kp, ctrl_gt ki, ctrl_gt margin, parameter_gt fs)
60{
61 // 1. Initialize PI
62 // Note: The output of this PI is id_ref.
63 // Ideally, when V_err < 0 (Voltage too high), we want id to go Negative.
64 // If we define Error = V_ref - V_s:
65 // V_s > V_ref -> Error < 0.
66 // We need negative output. So Kp should be Positive.
67 ctl_init_pid(&fw->volt_ctrl, kp, ki, 0, fs);
68
69 // 2. Set PI Limits
70 // The FW controller only generates NEGATIVE id current.
71 // So Max is 0, Min is -I_smax (will be updated in step if I_smax changes, but init here).
72 ctl_set_pid_limit(&fw->volt_ctrl, 0, -1000.0f); // Temporary min, sets in step
73 ctl_set_pid_int_limit(&fw->volt_ctrl, 0, -1000.0f);
74
75 fw->v_ref_margin = margin;
76 fw->i_smax = 0; // User needs to set this via setter or input
77 fw->id_ref_fw = 0;
78 fw->iq_max_dynamic = 0;
79 fw->flag_in_fw = 0;
80
81 ctl_vector2_clear(&fw->vdq_curr);
82}

92GMP_STATIC_INLINE void ctl_step_fw_ctrl(ctl_fw_ctrl_t* fw, ctl_vector2_t* vdq_out, ctrl_gt udc, ctrl_gt i_max)
93{
94 // Update inputs
95 fw->i_smax = i_max;
96 ctl_vector2_copy(&fw->vdq_curr, vdq_out);
97
98 // --------------------------------------------------------
99 // 1. Calculate Voltage Magnitude (Vs)
100 // --------------------------------------------------------
101 // Vs = sqrt(vd^2 + vq^2)
102 // Optimization: If your CPU is slow, you can use Alpha/Beta voltages
103 // or compare squared values to avoid sqrt, but for Linear PI control, sqrt is better.
104 ctrl_gt v_s_sq = ctl_mul(fw->vdq_curr.dat[phase_d], fw->vdq_curr.dat[phase_d]) +
105 ctl_mul(fw->vdq_curr.dat[phase_q], fw->vdq_curr.dat[phase_q]);
106
107 ctrl_gt v_s = ctl_sqrt(v_s_sq);
108
109 // --------------------------------------------------------
110 // 2. Calculate Voltage Limit (V_smax)
111 // --------------------------------------------------------
112 // V_smax = (Udc / sqrt(3)) * Margin
113 // Note: If using SVPWM over-modulation, this limit can be higher.
114 // Keeping it linear region: Udc * 0.57735 * Margin.
115 ctrl_gt v_limit = ctl_mul(udc, ctl_mul(float2ctrl(0.5773502f), fw->v_ref_margin));
116
117 // --------------------------------------------------------
118 // 3. Voltage Loop PI
119 // --------------------------------------------------------
120 // Error = V_limit - V_s
121 ctrl_gt v_err = v_limit - v_s;
122
123 // Update PI Limits dynamically based on current I_max
124 ctl_set_pid_limit(&fw->volt_ctrl, 0, -fw->i_smax);
125 ctl_set_pid_int_limit(&fw->volt_ctrl, 0, -fw->i_smax);
126
127 // Run PI
128 // If V_s > V_limit, Error < 0, Output becomes negative (inject negative id).
129 // If V_s < V_limit, Error > 0, Output tries to go positive but clamped to 0.
130 fw->id_ref_fw = ctl_step_pid_par(&fw->volt_ctrl, v_err);
131
132 // Check status
133 fw->flag_in_fw = (fw->id_ref_fw < float2ctrl(-0.01f)) ? 1 : 0;
134
135 // --------------------------------------------------------
136 // 4. Calculate Iq Limit (Priority to Id)
137 // --------------------------------------------------------
138 // Iq_max = sqrt(I_smax^2 - Id_fw^2)
139 ctrl_gt i_rem_sq = ctl_mul(fw->i_smax, fw->i_smax) - ctl_mul(fw->id_ref_fw, fw->id_ref_fw);
140
141 if (i_rem_sq > 0)
142 {
143 fw->iq_max_dynamic = ctl_sqrt(i_rem_sq);
144 }
145 else
146 {
147 fw->iq_max_dynamic = 0;
148 }
149}
