105GMP_STATIC_INLINE void ctl_step_im_fo(im_fo_t* fo, const vector2_gt* v_ab_star, const vector2_gt* i_ab_star)
106{
107 // Calculate stator back-EMF
108 ctrl_gt bemf_alpha = ctl_sub(v_ab_star->dat[0], ctl_mul(fo->rs, i_ab_star->dat[0]));
109 ctrl_gt bemf_beta = ctl_sub(v_ab_star->dat[1], ctl_mul(fo->rs, i_ab_star->dat[1]));
110
111 // Estimate stator flux using a low-pass filter integrator
112 // ”ūs[k] = k_lpf * ”ūs[k-1] + k_ts * BEMF[k]
113 fo->psi_s.dat[0] = ctl_add(ctl_mul(fo->k_lpf, fo->psi_s.dat[0]), ctl_mul(fo->k_ts, bemf_alpha));
114 fo->psi_s.dat[1] = ctl_add(ctl_mul(fo->k_lpf, fo->psi_s.dat[1]), ctl_mul(fo->k_ts, bemf_beta));
115
116 // Calculate rotor flux
117 // ”ūr_”┴ = k_rotor_flux * (”ūs_”┴ - k_sigma_ls * is_”┴)
118 fo->psi_r.dat[0] = ctl_mul(fo->k_rotor_flux, ctl_sub(fo->psi_s.dat[0], ctl_mul(fo->k_sigma_ls, i_ab_star->dat[0])));
119 fo->psi_r.dat[1] = ctl_mul(fo->k_rotor_flux, ctl_sub(fo->psi_s.dat[1], ctl_mul(fo->k_sigma_ls, i_ab_star->dat[1])));
120
121 // Calculate rotor flux magnitude
122 fo->psi_r_mag =
123 ctl_sqrt(ctl_add(ctl_mul(fo->psi_r.dat[0], fo->psi_r.dat[0]), ctl_mul(fo->psi_r.dat[1], fo->psi_r.dat[1])));
124
125 // Calculate electromagnetic torque
126 // Te = k_torque * (”ūr_”┴ * is_”┬ - ”ūr_”┬ * is_”┴)
127 ctrl_gt torque_term =
128 ctl_sub(ctl_mul(fo->psi_r.dat[0], i_ab_star->dat[1]), ctl_mul(fo->psi_r.dat[1], i_ab_star->dat[0]));
129 fo->torque = ctl_mul(fo->k_torque, torque_term);
130}
