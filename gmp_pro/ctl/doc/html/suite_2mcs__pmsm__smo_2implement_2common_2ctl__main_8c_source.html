74 // accelerator parameters

123#endif // BUILD_LEVEL

130 }

133 ctl_enable_output();

144void ctl_mainloop(void)
145{
146 int spd_target = gmp_base_get_system_tick() / 100;
147
148 // ctl_set_pmsm_smo_ctrl_speed(&pmsm_ctrl, float2ctrl(0.1) * spd_target - float2ctrl(1.0));
149
150#if (BUILD_LEVEL == 3)
151 // if build level == 3 switch to SMO controller
152 if (gmp_base_get_system_tick() >= 600)
153 {
154 ctl_switch_pmsm_smo_ctrl_using_smo(&pmsm_ctrl);
155 }
156#endif // BUILD_LEVEL
157
158 if (gmp_base_get_system_tick() >= 1000)
159 {
160 ctl_set_pmsm_smo_ctrl_speed(&pmsm_ctrl, float2ctrl(0.3));
161 }
162
163 if (gmp_base_get_system_tick() >= 1400)
164 {
165 ctl_set_pmsm_smo_ctrl_speed(&pmsm_ctrl, float2ctrl(0.8));
166 }
167
168 //
169 if (flag_enable_adc_calibrator)
170 {
171 if (ctl_is_adc_calibrator_cmpt(&adc_calibrator))
172 {
173 // set_adc_bias_via_channel(index_adc_calibrator, ctl_get_adc_calibrator_result(&adc_calibrator));
174 index_adc_calibrator += 1;
175 if (index_adc_calibrator > MTR_ADC_IDC)
176 flag_enable_adc_calibrator = 0;
177 }
178 }
179 return;
180}
