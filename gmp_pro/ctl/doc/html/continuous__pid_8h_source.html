114GMP_STATIC_INLINE ctrl_gt ctl_step_pid_par(ctl_pid_t* hpid, ctrl_gt input)
115{
116 // update p term
117 hpid->p_term = ctl_mul(input, hpid->kp);
118
119 // Update integrator sum with anti-windup
120 hpid->i_term = ctl_sat(hpid->i_term + ctl_mul(input, hpid->ki), hpid->integral_max, hpid->integral_min);
121
122 // update d term
123 hpid->d_term = ctl_mul((input - hpid->dn), hpid->kd);
124
125 // Output = P_term + I_term + D_term
126 hpid->out = hpid->p_term + hpid->i_term + hpid->d_term;
127
128 // Saturate final output
129 hpid->out = ctl_sat(hpid->out, hpid->out_max, hpid->out_min);
130
131 // Store current input for next derivative calculation
132 hpid->dn = input;
133
134 return hpid->out;
135}

145GMP_STATIC_INLINE ctrl_gt ctl_step_pid_ser(ctl_pid_t* hpid, ctrl_gt input)
146{
147 // update p term
148 hpid->p_term = ctl_mul(input, hpid->kp);
149
150 // Update integrator sum based on the proportional output
151 hpid->i_term = ctl_sat(hpid->i_term + ctl_mul(hpid->p_term, hpid->ki), hpid->integral_max, hpid->integral_min);
152
153 // update d term
154 hpid->d_term = ctl_mul((input - hpid->dn), hpid->kd);
155
156 // Output = P_term + I_term + D_term
157 // Note: In a true series form, Kd would also be scaled by Kp.
158 hpid->out = hpid->p_term + hpid->i_term + hpid->d_term;
159
160 // Saturate final output
161 hpid->out = ctl_sat(hpid->out, hpid->out_max, hpid->out_min);
162
163 // Store current input for next derivative calculation
164 hpid->dn = input;
165
166 return hpid->out;

233GMP_STATIC_INLINE void ctl_pid_clamping_correction(ctl_pid_t* hpid)
234{
235 // Back-calculate what the integrator SHOULD be to match the actual_out
236 ctrl_gt sn_corrected = hpid->out - hpid->p_term - hpid->d_term;
237
238 // Apply the correction to the integrator state

310GMP_STATIC_INLINE ctrl_gt ctl_step_pid_aw_par(ctl_pid_aw_t* hpid, ctrl_gt input)
311{
312 // Calculate unsaturated output
313 hpid->out_without_sat = ctl_mul(input, hpid->kp) + hpid->sn + ctl_mul((input - hpid->dn), hpid->kd);
314
315 // Saturate the output
316 hpid->out = ctl_sat(hpid->out_without_sat, hpid->out_max, hpid->out_min);
317
318 // Update integrator sum with back-calculation anti-windup
319 ctrl_gt back_calc_term = ctl_mul(hpid->out_without_sat - hpid->out, hpid->kc);
320 hpid->sn = hpid->sn + ctl_mul(input, hpid->ki) - back_calc_term;
321
322 // Store current input for next derivative calculation
323 hpid->dn = input;
324
325 return hpid->out;
326}
