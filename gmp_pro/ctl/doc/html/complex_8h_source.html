105GMP_STATIC_INLINE ctl_complex_t ctl_cdiv(ctl_complex_t a, ctl_complex_t b)
106{
107 ctl_complex_t result;
108 ctrl_gt den = b.real * b.real + b.imag * b.imag;
109
110 if (den == 0)

112 result.real = 0;
113 result.imag = 0;
114 return result;
115 }
116
117 result.real = (a.real * b.real + a.imag * b.imag) / den;
118 result.imag = (a.imag * b.real - a.real * b.imag) / den;
119 return result;
