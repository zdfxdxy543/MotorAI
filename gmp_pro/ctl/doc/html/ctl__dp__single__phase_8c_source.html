23 parameter_gt fg, parameter_gt fs)
24{
25 // Clear the SPLL structure to ensure a clean state.
26 ctl_clear_single_phase_pll(spll);
27
28 // Initialize a discrete SOGI for generating alpha-beta orthogonal signals.
29 ctl_init_discrete_sogi(&spll->sogi, 0.5, fg, fs);
30
31 // Initialize a low-pass filter for the q-axis component (Uq) of the SOGI output.
32 ctl_init_lp_filter(&spll->filter_uq, fs, fc);
33

35 ctl_init_pid_Tmode(&spll->spll_ctrl, gain, Ti, 0, fs);

60void ctl_init_single_phase_dc_pll(ctl_single_phase_dc_pll* spll, parameter_gt loop_kp, parameter_gt loop_ki,
61 parameter_gt k_sogi, parameter_gt k_dc, parameter_gt fc_uq, parameter_gt fg,
62 parameter_gt fs)
63{
64 // Clear states
65 ctl_clear_single_phase_dc_pll(spll);
66
67 // 1. Init SOGI-DC
68 // Note: k_damp is usually 1.414. k_dc is usually 0.5 to 1.0.
69 ctl_init_discrete_sogi_dc(&spll->sogi_dc, k_sogi, k_dc, fg, fs);
70
71 // 2. Init Loop Filter (PI)
72 ctl_init_pid(&spll->spll_ctrl, loop_kp, loop_ki, 0, fs);
73
74 // 3. Init Q-axis LPF
75 ctl_init_lp_filter(&spll->filter_uq, fs, fc_uq);
76
77 // 4. Init Scaling Factor
78 // Step = Fg / Fs
79 spll->freq_sf = float2ctrl(fg / fs);
80}
