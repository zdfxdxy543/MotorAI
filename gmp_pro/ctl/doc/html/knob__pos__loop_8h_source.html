82GMP_STATIC_INLINE void ctl_step_knob_pos_loop(ctl_pmsm_knob_pos_loop* knob)
83{
84 if (ctl_step_divider(&knob->div))
85 {
86 // Calculate the angular width of a single detent step.
87 ctrl_gt step_width = CTL_CTRL_CONST_1 / knob->knob_step;
88 ctrl_gt half_step_width = step_width / 2;
89
90 // Determine the index of the current detent based on the actual angle.
91 knob->output_pos = (int16_t)(knob->actual_ang / step_width);
92
93 // Set the target angle to be the center of the current detent.
94 knob->target_ang = knob->output_pos * step_width + half_step_width;
95 knob->target_pos = knob->actual_pos;
96
97 // --- P-Controller Logic ---
98 // Calculate position error.
99 int32_t delta_pos = knob->target_pos - knob->actual_pos;
100
101 // Clamp the multi-turn error to prevent excessive speed when crossing many detents at once.
102 delta_pos = (delta_pos > 2) ? 2 : delta_pos;
103 delta_pos = (delta_pos < -2) ? -2 : delta_pos;
104
105 // Combine integer and fractional parts into a single error value.
106 ctrl_gt position_error = CTL_CTRL_CONST_2_PI * (ctrl_gt)delta_pos + knob->target_ang - knob->actual_ang;
107
108 // Apply proportional gain.
109 knob->speed_ref = ctl_mul(knob->kp, position_error);
110
111 // Saturate the output to the speed limit.
112 knob->speed_ref = ctl_sat(knob->speed_ref, knob->speed_limit, -knob->speed_limit);
113 }
114}

134GMP_STATIC_INLINE void ctl_input_knob_pos_via_only_ang(ctl_pmsm_knob_pos_loop* knob, ctrl_gt actual_ang)
135{
136 ctrl_gt delta_ang = actual_ang - knob->actual_ang;
137 knob->actual_ang = actual_ang;
138
139 // Correct the multi-turn count based on angle wrap-around.
140 if (delta_ang < -CTL_CTRL_CONST_1_OVER_2)
141 {
142 knob->actual_pos += 1; // Forward wrap
143 }
144 else if (delta_ang > CTL_CTRL_CONST_1_OVER_2)
145 {
146 knob->actual_pos -= 1; // Backward wrap
147 }
148}
