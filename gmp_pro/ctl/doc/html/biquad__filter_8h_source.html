82GMP_STATIC_INLINE ctrl_gt ctl_step_biquad_filter(ctl_biquad_filter_t* obj, ctrl_gt input)
83{
84 obj->out = ctl_mul(obj->b[0], input) + ctl_mul(obj->b[1], obj->x[0]) + ctl_mul(obj->b[2], obj->x[1]) -
85 ctl_mul(obj->a[0], obj->y[0]) - ctl_mul(obj->a[1], obj->y[1]);
86
87 // Update historical inputs
88 obj->x[1] = obj->x[0];
89 obj->x[0] = input;
90
91 // Update historical outputs
92 obj->y[1] = obj->y[0];
93 obj->y[0] = obj->out;
