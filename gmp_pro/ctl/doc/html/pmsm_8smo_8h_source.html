168GMP_STATIC_INLINE ctrl_gt ctl_step_pmsm_smo(pmsm_smo_t* smo, ctrl_gt u_alpha, ctrl_gt u_beta, ctrl_gt i_alpha,
169 ctrl_gt i_beta)
170{
171 // 1. Update the PMSM observer model to get the next estimated current.
172 // This is a discrete-time model of the motor's electrical dynamics.
173 ctrl_gt delta_i_alpha = ctl_mul(smo->k1, u_alpha - smo->z.dat[0]) - ctl_mul(smo->k2, smo->i_est.dat[0]) -
174 ctl_mul(ctl_mul(smo->wr_est, smo->k3), smo->i_est.dat[1]);
175
176 ctrl_gt delta_i_beta = ctl_mul(smo->k1, u_beta - smo->z.dat[1]) - ctl_mul(smo->k2, smo->i_est.dat[1]) +
177 ctl_mul(ctl_mul(smo->wr_est, smo->k3), smo->i_est.dat[0]);
178
179 smo->i_est.dat[0] += delta_i_alpha;
180 smo->i_est.dat[1] += delta_i_beta;
181
182 // 2. Apply the sliding control law.
183 // The output 'z' is proportional to the sign of the current error.
184 // A saturation function is used as a practical replacement for the signum function.
185 smo->z.dat[0] = ctl_mul(ctl_sat(smo->i_est.dat[0] - i_alpha, float2ctrl(0.1), -float2ctrl(0.1)), smo->k_slide);
186 smo->z.dat[1] = ctl_mul(ctl_sat(smo->i_est.dat[1] - i_beta, float2ctrl(0.1), -float2ctrl(0.1)), smo->k_slide);
187
188 // 3. Low-pass filter the sliding term 'z' to extract the estimated back-EMF.
189 smo->e_est.dat[0] = ctl_step_lowpass_filter(&smo->filter_e[0], smo->z.dat[0]);
190 smo->e_est.dat[1] = ctl_step_lowpass_filter(&smo->filter_e[1], smo->z.dat[1]);
191
192 // 4. Process the back-EMF through the PLL to track angle and speed.
193 // 4a. Generate the PLL error signal (estimated q-axis component of back-EMF).
194 ctl_set_phasor_via_angle(smo->theta_est, &smo->phasor);
195 ctrl_gt e_error = -ctl_mul(smo->e_est.dat[0], smo->phasor.dat[1]) + ctl_mul(smo->e_est.dat[1], smo->phasor.dat[0]);
196
197 // 4b. PI controller generates the estimated speed command.
198 ctrl_gt pid_out = ctl_step_pid_par(&smo->pid_pll, e_error);
199
200 // 4c. Low-pass filter the speed for a smoother estimate (in rad/tick).
201 smo->wr_est = ctl_step_lowpass_filter(&smo->filter_spd, pid_out);
202
203 // 4d. Integrate the speed to get the next estimated angle.
204 smo->theta_est += ctl_mul(smo->wr_est, CTL_CTRL_CONST_1_OVER_2PI);
205 smo->theta_est = ctrl_mod_1(smo->theta_est); // Wrap angle to [0, 1.0)
206
207 // 5. Update the output interfaces.
208 // 5a. Convert speed to p.u. and apply to the velocity interface.
209 smo->spdif.speed = ctl_mul(smo->wr_est, smo->spd_sf);
210
211 // 5b. Apply phase compensation to the angle for the encoder interface.
212 // This corrects for the phase lag introduced by the back-EMF low-pass filters.
213 ctrl_gt phase_lag_comp = ctl_atan2(ctl_mul(smo->wr_est, smo->theta_compensate), CTL_CTRL_CONST_1);
214 smo->encif.elec_position = smo->theta_est + ctl_mul(phase_lag_comp, CTL_CTRL_CONST_1_OVER_2PI);
215 smo->encif.elec_position = ctrl_mod_1(smo->encif.elec_position);
216
217 return smo->encif.elec_position;
218}
