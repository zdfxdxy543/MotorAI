105void ctl_mainloop(void)
106{
107 // When the program is reach here, the following things will happen:
108 // 1. software non-block delay 500ms
109 // 2. judge if spll theta error convergence has occurred
110 // 3. then enable system
111
112 if (flag_system_enable)
113 {
114
115 // first time flag
116 // log the first time enable the system
117 if (firsttime_flag == 0)
118 {
119 tick_bias = gmp_base_get_system_tick();
120 firsttime_flag = 1;
121 }
122
123 // a delay of 100ms
124 if ((started_flag == 0) && ((gmp_base_get_system_tick() - tick_bias) > 100) && (startup_flag == 0))
125 {
126 startup_flag = 1;
127 }
128
129 // judge if PLL is close to target
130 if ((started_flag == 0) && (startup_flag == 1) && ctl_ready_mainloop())
131 {
132 ctl_enable_output();
133 started_flag = 1;
134 }
135 }
136 // if system is disabled
137 else // (flag_system_enable == 0)
138 {
139 ctl_disable_output();
140
141 // clear all flags
142 firsttime_flag = 0;
143 startup_flag = 0;
144 started_flag = 0;
145 tick_bias = 0;
146 }
147
148 return;
149}
