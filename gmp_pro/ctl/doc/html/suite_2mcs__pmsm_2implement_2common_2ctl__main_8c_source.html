188void ctl_mainloop(void)
189{
190
191 if (flag_system_enable)
192 {
193
194 // first time flag
195 // log the first time enable the system
196 if (firsttime_flag == 0)
197 {
198 tick_bias = gmp_base_get_system_tick();
199 firsttime_flag = 1;
200 }
201
202 // a delay of 100 ms
203 if ((started_flag == 0) && ((gmp_base_get_system_tick() - tick_bias) > CTRL_STARTUP_DELAY) &&
204 (startup_flag == 0))
205 {
206 startup_flag = 1;
207 }
208
209 // judge if PLL is close to target
210 if ((started_flag == 0) && (startup_flag == 1) && ctl_ready_mainloop())
211 {
212 ctl_clear_slope_f(&rg);
213
214 ctl_clear_pmsm_ctrl(&pmsm_ctrl);
215 ctl_enable_pmsm_ctrl(&pmsm_ctrl);
216
217 ctl_enable_output();
218 started_flag = 1;
219 }
220 }
221 // if system is disabled
222 else // (flag_system_enable == 0)
223 {
224 ctl_disable_output();
225
226 // clear all flags
227 firsttime_flag = 0;
228 startup_flag = 0;
229 started_flag = 0;
230 tick_bias = 0;
231 }
232
233 return;
234}
