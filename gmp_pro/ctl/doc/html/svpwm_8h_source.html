97GMP_STATIC_INLINE void ctl_svpwm_calc(ctl_svpwm_channel_t* svpwm)
98{
99 ctrl_gt Ua, Ub, Uc; // Uabc three phase parameters
100 ctrl_gt Umax, Umin, Ucom;
101
102 ctrl_gt Ualpha_tmp = -ctl_div2(svpwm->Ualpha);
103 ctrl_gt Ubeta_tmp = ctl_mul(svpwm->Ubeta, GMP_CONST_SQRT_3_OVER_2);
104
105 Ua = svpwm->Ualpha;
106 Ub = Ualpha_tmp + Ubeta_tmp;
107 Uc = Ualpha_tmp - Ubeta_tmp;
108
109 // find Vmax & Vmin
110 if (Ua > Ub)
111 {
112 Umax = Ua;
113 Umin = Ub;
114 }
115 else
116 {
117 Umax = Ub;
118 Umin = Ua;
119 }
120
121 if (Uc > Umax)
122 Umax = Uc;
123 else if (Uc < Umin)
124 Umin = Uc;
125
126 // get common mode term
127 Ucom = ctl_div2(Umax + Umin);
128
129 // get SVPWM modulation result
130 svpwm->T[0] = Ua - Ucom;
131 svpwm->T[1] = Ub - Ucom;
132 svpwm->T[2] = Uc - Ucom;
133}

146GMP_STATIC_INLINE void ctl_svpwm_calc2(ctl_svpwm_channel_t* svpwm)
147{
148 // u2s: Ualpha ,Ubeta
149 ctrl_gt X, Y, Z, T1, T2, Ta, Tb, Tc;
150 uint16_t N;
151 ctrl_gt Uabc[3] = {0};
152
153 Uabc[0] = svpwm->Ubeta;
154 Uabc[1] = ctl_mul(GMP_CONST_SQRT_3_OVER_2, svpwm->Ualpha) - ctl_div2(svpwm->Ubeta);
155 Uabc[2] = -ctl_mul(GMP_CONST_SQRT_3_OVER_2, svpwm->Ualpha) - ctl_div2(svpwm->Ubeta);
156
157 N = ((Uabc[0] > 0)) + ((Uabc[1] > 0) << 1) + ((Uabc[2] > 0) << 2);
158 X = ctl_mul(GMP_CONST_SQRT_3, svpwm->Ubeta);
159 Y = ctl_mul(GMP_CONST_3_OVER_2, svpwm->Ualpha) + ctl_mul(GMP_CONST_SQRT_3_OVER_2, svpwm->Ubeta);
160 Z = -ctl_mul(GMP_CONST_3_OVER_2, svpwm->Ualpha) + ctl_mul(GMP_CONST_SQRT_3_OVER_2, svpwm->Ubeta);
161
162 switch (N)
163 {
164 case 1:
165 T1 = Z;
166 T2 = Y;
167 break;
168 case 2:
169 T1 = Y;
170 T2 = -X;
171 break;
172 case 3:
173 T1 = -Z;
174 T2 = X;
175 break;
176 case 4:
177 T1 = -X;
178 T2 = Z;
179 break;
180 case 5:
181 T1 = X;
182 T2 = -Y;
183 break;
184 case 6:
185 T1 = -Y;
186 T2 = -Z;
187 break;
188 default:
189 T1 = 0;
190 T2 = 0;
191 break; // Default case added for safety
192 }
193
194 if ((T1 + T2) > GMP_CONST_1)
195 {
196 T1 = ctl_div(T1, (T1 + T2));
197 T2 = GMP_CONST_1 - T1;
198 }
199 Ta = ctl_div4(GMP_CONST_1 - T1 - T2);
200 Tb = Ta + ctl_div2(T1);
201 Tc = Tb + ctl_div2(T2);
202
203 Ta *= 2;
204 Tb *= 2;
205 Tc *= 2;
206
207 Ta = -Ta;
208 Tb = -Tb;
209 Tc = -Tc;
210
211 Ta += GMP_CONST_1_OVER_2;
212 Tb += GMP_CONST_1_OVER_2;
213 Tc += GMP_CONST_1_OVER_2;
214
215 switch (N)
216 {
217 case 0:
218 svpwm->T[0] = 0;
219 svpwm->T[1] = 0;
220 svpwm->T[2] = 0;
221 break;
222 case 1:
223 svpwm->T[0] = Tb;
224 svpwm->T[1] = Ta;
225 svpwm->T[2] = Tc;
226 break;
227 case 2:
228 svpwm->T[0] = Ta;
229 svpwm->T[1] = Tc;
230 svpwm->T[2] = Tb;
231 break;
232 case 3:
233 svpwm->T[0] = Ta;
234 svpwm->T[1] = Tb;
235 svpwm->T[2] = Tc;
236 break;
237 case 4:
238 svpwm->T[0] = Tc;
239 svpwm->T[1] = Tb;
240 svpwm->T[2] = Ta;
241 break;
242 case 5:
243 svpwm->T[0] = Tc;
244 svpwm->T[1] = Ta;
245 svpwm->T[2] = Tb;
246 break;
247 case 6:
248 svpwm->T[0] = Tb;
249 svpwm->T[1] = Tc;
250 svpwm->T[2] = Ta;
251 break;
252 default:
253 svpwm->T[0] = 0;
254 svpwm->T[1] = 0;
255 svpwm->T[2] = 0;
256 break;
257 }
258
259 return;

272GMP_STATIC_INLINE void ctl_svpwm_modulation(ctl_svpwm_channel_t* svpwm)
273{
274 int i = 0;
275 ctrl_gt pwm_data;
276 pwm_gt pwm_output;
277
278 for (i = 0; i < 3; ++i)
279 {
280 // Shift duty cycle from [-0.5, 0.5] to [0, 1.0]
281 pwm_data = svpwm->T[i] + float2ctrl(0.5f);
282 // Prevent data error
283 pwm_data = pwm_data < 0 ? 0 : pwm_data;
284 // Scale to PWM period
285 pwm_output = (pwm_gt)ctl_mul(pwm_data, svpwm->pwm_period);
286 // Clamp output to the maximum period value
287 svpwm->pwm_cmp[i] = pwm_output > svpwm->pwm_period ? svpwm->pwm_period : pwm_output;
288 }
289}
