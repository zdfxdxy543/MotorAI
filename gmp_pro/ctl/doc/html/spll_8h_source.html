115GMP_STATIC_INLINE void ctl_step_single_phase_pll(ctl_single_phase_pll* spll, ctrl_gt ac_input)
116{
117 // 1. Orthogonal Signal Generation using SOGI
118 // SOGI generates v_alpha (in-phase) and v_beta (quadrature, 90 deg lag)
119 ctl_step_discrete_sogi(&spll->sogi, ac_input);
120
121 // 2. Park Transform from stationary \alpha - \beta to rotating d-q frame
122 // The SOGI outputs are assigned to alpha and beta components.
123 // The negative sign is a convention to align the final locked phase.
124 spll->uab.dat[phase_alpha] = -ctl_get_discrete_sogi_ds(&spll->sogi);
125 spll->uab.dat[phase_beta] = ctl_get_discrete_sogi_qs(&spll->sogi);
126
127 // 3. Update the phasor (sin/cos) for the next iteration's Park transform.
128 ctl_set_phasor_via_angle(spll->theta, &spll->phasor);
129
130 // 4. Park Transform (AlphaBeta -> DQ)
131 // We want to lock to the Alpha axis (Input Voltage Vector).
132 // In a locked system:
133 // Vd = Valpha*cos + Vbeta*sin --> Magnitude
134 // Vq = -Valpha*sin + Vbeta*cos --> Error (to be driven to 0)
135 ctl_ct_park2(&spll->uab, &spll->phasor, &spll->udq);
136
137 // When Vq is regulated to 0, Vd represents the full vector magnitude.
138 spll->v_mag = spll->udq.dat[phase_d];
139
140 // 5. Loop Filter: PI controller drives the q-axis component (phase error) to zero.
141 ctl_step_lowpass_filter(&spll->filter_uq, spll->udq.dat[phase_q]);
142 spll->freq_error = ctl_step_pid_ser(&spll->spll_ctrl, ctl_get_lowpass_filter_result(&spll->filter_uq));
143
144 // 6. Voltage-Controlled Oscillator (VCO)
145 // The nominal frequency (1.0 p.u.) is adjusted by the error from the loop filter.
146 spll->frequency = float2ctrl(1) + spll->freq_error;
147
148 // Integrate the frequency to get the new phase angle, and warp.
149 spll->theta = ctrl_mod_1(spll->theta + ctl_mul(spll->frequency, spll->freq_sf));
150}

238GMP_STATIC_INLINE void ctl_step_single_phase_dc_pll(ctl_single_phase_dc_pll* spll, ctrl_gt ac_input)
239{
240 // 1. Orthogonal Signal Generation using SOGI-DC
241 // This step removes DC from input and generates v_alpha, v_beta
242 ctl_step_discrete_sogi_dc(&spll->sogi_dc, ac_input);
243
244 // Retrieve SOGI outputs
245 // Note: SOGI-DC Alpha is aligned with input (BandPass), Beta is lagging 90 (LowPass)
246 // Applying the same convention as standard PLL: Alpha -> -Alpha to align frames if necessary,
247 // or keep standard.
248 // Convention check: Park transform expects Vd aligned with V_alpha if theta=0.
249 // If we use standard: Vd = Valpha*cos + Vbeta*sin.
250 spll->uab.dat[phase_alpha] = -ctl_get_discrete_sogi_dc_alpha(&spll->sogi_dc);
251 spll->uab.dat[phase_beta] = ctl_get_discrete_sogi_dc_beta(&spll->sogi_dc);
252
253 // Also retrieve the estimated DC offset for diagnostics
254 spll->v_dc_est = ctl_get_discrete_sogi_dc_offset(&spll->sogi_dc);
255
256 // 2. Update Phasor (based on previous theta)
257 ctl_set_phasor_via_angle(spll->theta, &spll->phasor);
258
259 // 3. Park Transform (AlphaBeta -> DQ)
260 // Vd = Valpha*cos + Vbeta*sin --> Magnitude
261 // Vq = -Valpha*sin + Vbeta*cos --> Error
262 ctl_ct_park2(&spll->uab, &spll->phasor, &spll->udq);
263
264 // 4. Extract Output: Magnitude
265 // When locked, Vd is the amplitude
266 spll->v_mag = spll->udq.dat[phase_d];
267
268 // 5. Loop Filter chain
269 // LPF on Q-axis error (Optional, reduces harmonics)
270 ctl_step_lowpass_filter(&spll->filter_uq, spll->udq.dat[phase_q]);
271
272 // PI Controller drives filtered error to zero
273 spll->freq_error = ctl_step_pid_ser(&spll->spll_ctrl, ctl_get_lowpass_filter_result(&spll->filter_uq));
274
275 // 6. VCO (Frequency & Angle Update)
276 spll->frequency = float2ctrl(1.0f) + spll->freq_error;
277
278 // Integrate: theta += freq * (2pi * Ts_normalized)
279 spll->theta += ctl_mul(spll->frequency, spll->freq_sf);
280
281 // Wrap
282 spll->theta = ctrl_mod_1(spll->theta);
283}
