178GMP_STATIC_INLINE ctrl_gt ctl_step_boost_ctrl(boost_ctrl_t* boost)
179{
180 // Filter the raw ADC inputs
181 ctl_step_lowpass_filter(&boost->lpf_il, boost->adc_il->value);
182 ctl_step_lowpass_filter(&boost->lpf_uo, boost->adc_uo->value);
183 ctl_step_lowpass_filter(&boost->lpf_ui, boost->adc_ui->value);
184
185 if (boost->flag_enable_system)
186 {
187 // Outer voltage loop
188 if (boost->flag_enable_voltage_ctrl)
189 {
190 boost->current_set =
191 ctl_step_pid_ser(&boost->voltage_pid, boost->voltage_set - boost->lpf_uo.out) + boost->current_ff;
192 }
193
194 // Inner current loop
195 if (boost->flag_enable_current_ctrl)
196 {
197 boost->voltage_out =
198 ctl_step_pid_ser(&boost->current_pid, boost->current_set - boost->lpf_il.out) + boost->voltage_ff;
199 }
200
201#ifdef CTL_BOOST_CTRL_OUTPUT_WITHOUT_UOUT
202 // Duty cycle calculation without input voltage feedforward (for debugging)
203#if CTL_BOOST_CTRL_POSITION == UPPER_BRIDGE
204 boost->pwm_out_pu = float2ctrl(1) - boost->voltage_out;
205#elif CTL_BOOST_CTRL_POSITION == LOWER_BRIDGE
206 boost->pwm_out_pu = boost->voltage_out;
207#endif // CTL_BOOST_CTRL_POSITION
208#else // CTL_BOOST_CTRL_OUTPUT_WITHOUT_UOUT \
209 // Standard duty cycle calculation with input voltage feedforward
210 boost->vo_sat = ctl_step_saturation(&boost->modulation_saturation, boost->voltage_out);
211#if CTL_BOOST_CTRL_POSITION == UPPER_BRIDGE
212 // D = U_in / U_out_cmd
213 boost->pwm_out_pu = ctl_div(boost->lpf_ui.out, boost->vo_sat);
214#elif CTL_BOOST_CTRL_POSITION == LOWER_BRIDGE
215 // D = 1 - (U_in / U_out_cmd)
216 boost->pwm_out_pu = GMP_CONST_1 - ctl_div(boost->lpf_ui.out, boost->vo_sat);
217#endif // CTL_BOOST_CTRL_POSITION
218#endif // CTL_BOOST_CTRL_OUTPUT_WITHOUT_UOUT
219 }
220 else
221 {
222 // For safety, ensure the Boost switch is conductive by default when disabled.
223#if CTL_BOOST_CTRL_POSITION == UPPER_BRIDGE
224 boost->pwm_out_pu = float2ctrl(1);
225#elif CTL_BOOST_CTRL_POSITION == LOWER_BRIDGE
226 boost->pwm_out_pu = float2ctrl(0);
227#endif // CTL_BOOST_CTRL_POSITION
228 }
229
230 return boost->pwm_out_pu;
231}
