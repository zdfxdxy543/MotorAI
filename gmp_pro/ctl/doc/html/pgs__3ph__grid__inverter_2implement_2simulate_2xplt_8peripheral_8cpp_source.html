163// Using Windows console to simulate UART
164void at_device_flush_rx_buffer()
165{
166 uint16_t fifoLevel = 0;
167 uint16_t rxBuf[ISR_LOCAL_BUF_SIZE];
168
169 // 使用while一次性读取FIFO中的所有内容
170 while (_kbhit())
171 {
172 //_getch() 读取字符但不回显，也不等待回车
173 int ch = _getch();
174
175 // 处理特殊键 (例如方向键会产生两个码: 0/0xE0 和 键码)
176 // 这里我们简单处理，只接收普通 ASCII
177 if (ch == 0 || ch == 0xE0)
178 {
179 _getch(); // 读走无效部分
180 continue;
181 }
182
183 // 【重要】Windows控制台输入不自动回显，手动回显以便用户看到自己打的字
184 putchar(ch);
185
186 // 读取数据
187 rxBuf[fifoLevel++] = (uint16_t)ch;
188 }
189
190 // 推送给设备
191 if (fifoLevel > 0)
192 {
193 at_device_rx_isr(&at_dev, (char*)rxBuf, fifoLevel);
194 }
195}
