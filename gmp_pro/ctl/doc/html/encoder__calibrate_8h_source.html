120ec_gt ctl_task_position_encoder_offset_calibrate(position_enc_calibrate_t* obj)
121{
122 // --- Safety Check ---
123 // If an overcurrent condition is detected, abort the calibration.
124 if (obj->mc && (obj->mc->sensor.Idq.q > obj->current_limit || obj->mc->sensor.Idq.q < -obj->current_limit))
125 {
126 ctl_clear_position_encoder_calibrator(obj);
127 return GMP_EC_ERROR;
128 }
129
130 // --- State 0: Rotor Alignment and Convergence Check ---
131 if (obj->flag_stage1 == 0)
132 {
133 // Step I: Apply d-axis current to align the rotor.
134 ctl_set_motor_current_ctrl_idq_ref(obj->mc, obj->current_target, 0);
135 ctl_set_motor_current_ctrl_vdq_ff(obj->mc, 0, 0);
136
137 // Step II: Check if the encoder output has stabilized.
138 ctrl_gt current_pos = ctl_get_encoder_position(obj->pos_enc);
139 ctrl_gt pos_diff = current_pos - obj->old_position;
140 if (pos_diff < 0)
141 pos_diff = -pos_diff; // Absolute difference
142
143 if (pos_diff < obj->position_delta_target)
144 {
145 // Position is stable, shift a '0' into the convergence flag.
146 obj->flag_position_convergence = (obj->flag_position_convergence << 1) | 0x00;
147 }
148 else
149 {
150 // Position is still changing, shift a '1' into the flag.
151 obj->flag_position_convergence = (obj->flag_position_convergence << 1) | 0x01;
152 }
153
154 obj->old_position = current_pos; // Save current position for the next check.
155
156 // If the last 4 samples were stable (flag is all zeros), move to the next stage.
157 if ((obj->flag_position_convergence & 0x0F) == 0)
158 {
159 obj->switch_time = gmp_base_get_system_tick();
160 obj->flag_stage1 = 1; // Transition to Stage 1
161 }
162 return GMP_EC_RUNNING;
163 }
164 // --- State 1: Final Wait and Offset Calculation ---
165 else
166 {
167 // Step III: Wait for the specified duration to ensure rotor is fully settled.
168 if ((gmp_base_get_system_tick() - obj->switch_time) > obj->wait_time)
169 {
170 // Step IV: Calculate and save the final offset.
171 // The new offset is the sum of any existing offset and the current rotor position.
172 ctrl_gt final_pos = ctl_get_encoder_position(obj->pos_enc);
173 obj->offset = obj->pos_enc->offset + final_pos;
174
175 // Wrap the offset to the per-unit range [-1.0, 1.0).
176 while (obj->offset >= 1.0)
177 obj->offset -= 1.0;
178 while (obj->offset < 0.0)
179 obj->offset += 1.0;
180
181 // Apply the new offset to the encoder instance.
182 obj->pos_enc->offset = obj->offset;
183
184 // Stop applying current.
185 ctl_clear_position_encoder_calibrator(obj);
186
187 // Task is complete.
188 return GMP_EC_COMPLETE;
189 }
190 // If wait time has not elapsed, continue holding position.
191 return GMP_EC_RUNNING;
192 }
193}
