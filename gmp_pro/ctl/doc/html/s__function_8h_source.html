103GMP_STATIC_INLINE ctrl_gt ctl_step_s_function(ctl_s_function_t* obj, ctrl_gt input)
104{
105 // Calculate the output using the difference equation
106 ctrl_gt output = ctl_mul(obj->b0, input) + ctl_mul(obj->b1, obj->input_1) + ctl_mul(obj->b2, obj->input_2);
107 output -= ctl_mul(obj->a1, obj->output_1);
108 output -= ctl_mul(obj->a2, obj->output_2);
109
110 // Update state buffers for the next iteration
111 obj->input_2 = obj->input_1;
112 obj->input_1 = input;
113 obj->output_2 = obj->output_1;
114 obj->output_1 = output;
115
116 return output;
