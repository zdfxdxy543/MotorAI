243GMP_STATIC_INLINE void ctl_step_pmsm_mtpa_ctrl(pmsm_mtpa_controller_t* ctrl)
244{
245 ctl_vector2_t phasor;
246 ctrl_gt etheta;
247 ctrl_gt vq_limit = float2ctrl(1.0);
248
249 ctrl->isr_tick += 1;
250
251 if (ctrl->flag_enable_controller)
252 {
253 // Clark Transformation
254#if MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 3
255 ctl_ct_clarke(&ctrl->mtr_interface.iabc->value, &ctrl->iab0);
256#elif MTR_CTRL_CURRENT_MEASUREMENT_PHASES == 2
257 ctl_ct_clarke_2ph((ctl_vector2_t*)&ctrl->mtr_interface.iabc->value, (ctl_vector2_t*)&ctrl->iab0);
258#else
259#error("Wrong parameter for macro MTR_CTRL_CURRENT_MEASUREMENT_PHASES")
260#endif
261
262#if MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 3
263 ctl_ct_clarke(&ctrl->mtr_interface.uabc->value, &ctrl->uab0);
264#elif MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 2
265 ctl_ct_clarke_2ph(&ctrl->mtr_interface.uabc->value, &ctrl->uab0);
266#elif MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES == 0
267 ctl_vector3_clear(&ctrl->uab0);
268#else
269#error("Wrong parameter for macro MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES")
270#endif
271
272 // Park Transformation
273 etheta = ctl_get_mtr_elec_theta(&ctrl->mtr_interface);
274 ctl_set_phasor_via_angle(etheta, &phasor);
275 ctl_ct_park(&ctrl->iab0, &phasor, &ctrl->idq0);
276
277#if MTR_CTRL_VOLTAGE_MEASUREMENT_PHASES != 0
278 ctl_ct_park(&ctrl->uab0, &phasor, &ctrl->udq0);
279#else
280 ctl_vector3_clear(&ctrl->udq0);
281#endif
282
283 // Position Controller
284 if (ctrl->flag_enable_position_ctrl)
285 {
286 // Position control logic is not yet implemented.
287 }
288
289 // Velocity Controller
290 if (ctrl->flag_enable_velocity_ctrl)
291 {
292#if defined(PMSM_CTRL_USING_CURRENT_DISTRIBUTOR)
293 // With MTPA, speed controller outputs total current magnitude
294#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
295 ctrl->im_set =
296 ctl_step_tracking_pid(&ctrl->spd_ctrl, ctrl->speed_set, ctl_get_mtr_velocity(&ctrl->mtr_interface));
297#else // using continuous controller
298 ctrl->im_set = ctl_step_tracking_continuous_pid(&ctrl->spd_ctrl, ctrl->speed_set,
299 ctl_get_mtr_velocity(&ctrl->mtr_interface));
300#endif
301 // Distributor calculates optimal id and iq
302 ctl_step_current_distributor(&ctrl->distributor, ctrl->im_set);
303 ctrl->idq_set.dat[phase_q] = ctl_get_distributor_iq_ref(&ctrl->distributor) + ctrl->idq_ff.dat[phase_q];
304 ctrl->idq_set.dat[phase_d] = ctl_get_distributor_id_ref(&ctrl->distributor) + ctrl->idq_ff.dat[phase_d];
305#else
306 // Without MTPA, speed controller outputs only iq
307 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
308#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
309 ctrl->idq_set.dat[phase_q] = ctl_step_discrete_track_pid(&ctrl->spd_ctrl, ctrl->speed_set,
310 ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
311 ctrl->idq_ff.dat[phase_q];
312#else // using continuous controller
313 ctrl->idq_set.dat[phase_q] = ctl_step_tracking_continuous_pid(&ctrl->spd_ctrl, ctrl->speed_set,
314 ctl_get_mtr_velocity(&ctrl->mtr_interface)) +
315 ctrl->idq_ff.dat[phase_q];
316#endif // PMSM_CTRL_USING_DISCRETE_CTRL
317#endif // PMSM_CTRL_USING_CURRENT_DISTRIBUTOR
318
319#if MTR_CTRL_FEEDFORWARD_STRATEGY == 1
320 // Decoupling logic is commented out.
321 // ctl_mtr_pmsm_decouple(&ctrl->vdq_set, &ctrl->idq_set, ctrl->Ld, ctrl->Lq,
322 // ctl_get_mtr_velocity(ctrl->mtr_interface), ctrl->psi_e);
323#endif
324 }
325 else
326 {
327 ctrl->idq_set.dat[phase_d] = ctrl->idq_ff.dat[phase_d];
328 ctrl->idq_set.dat[phase_q] = ctrl->idq_ff.dat[phase_q];
329 }
330
331 // Current Controller
332 if (ctrl->flag_enable_current_ctrl)
333 {
334#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
335 ctrl->vdq_set.dat[phase_d] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_d],
336 ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
337 ctrl->vdq_ff.dat[phase_d];
338 ctrl->vdq_set.dat[phase_q] = ctl_step_discrete_pid(&ctrl->current_ctrl[phase_q],
339 ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
340 ctrl->vdq_ff.dat[phase_q];
341#else // using continuous controller
342 ctrl->vdq_set.dat[phase_d] =
343 ctl_step_pid_ser(&ctrl->current_ctrl[phase_d], ctrl->idq_set.dat[phase_d] - ctrl->idq0.dat[phase_d]) +
344 ctrl->vdq_ff.dat[phase_d];
345
346 vq_limit = ctl_sqrt(float2ctrl(1.0) - ctl_mul(ctrl->vdq_set.dat[phase_d], ctrl->vdq_set.dat[phase_d]));
347 ctl_set_pid_limit(&ctrl->current_ctrl[phase_q], vq_limit, -vq_limit);
348
349 ctrl->vdq_set.dat[phase_q] =
350 ctl_step_pid_ser(&ctrl->current_ctrl[phase_q], ctrl->idq_set.dat[phase_q] - ctrl->idq0.dat[phase_q]) +
351 ctrl->vdq_ff.dat[phase_q];
352#endif
353 ctrl->vdq_set.dat[phase_0] = 0;
354 }
355 else
356 {
357 ctrl->vdq_set.dat[phase_d] = ctrl->vdq_ff.dat[phase_d];
358 ctrl->vdq_set.dat[phase_q] = ctrl->vdq_ff.dat[phase_q];
359 ctrl->vdq_set.dat[phase_0] = 0;
360 }
361
362 // Inverse Park Transformation & Modulation
363 if (ctrl->flag_enable_modulation)
364 {
365 ctl_ct_ipark(&ctrl->vdq_set, &phasor, &ctrl->vab0_set);
366 }
367
368 // SVPWM Output Stage
369 if (ctrl->flag_enable_output)
370 {
371 ctl_ct_svpwm_calc(&ctrl->vab0_set, &ctrl->pwm_out->value);
372 }
373 else
374 {
375 ctrl->pwm_out->value.dat[phase_A] = float2ctrl(0.5);
376 ctrl->pwm_out->value.dat[phase_B] = float2ctrl(0.5);
377 ctrl->pwm_out->value.dat[phase_C] = float2ctrl(0.5);
378 }
379 }
380 else // Controller is disabled
381 {
382 ctrl->pwm_out->value.dat[phase_A] = 0;
383 ctrl->pwm_out->value.dat[phase_B] = 0;
384 ctrl->pwm_out->value.dat[phase_C] = 0;
385 ctl_clear_pmsm_mtpa_ctrl(ctrl);
386 }
387}
