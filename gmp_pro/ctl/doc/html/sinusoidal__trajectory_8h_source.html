112GMP_STATIC_INLINE ctrl_gt ctl_step_sin_planner(ctl_sin_planner_t* planner, ctrl_gt dt)
113{
114 if (!planner->is_active)
115 {
116 return planner->current_pos; // Not active, hold the last position.
117 }
118
119 // Increment the internal time
120 planner->current_time += dt;
121
122 if (planner->current_time >= planner->total_time)
123 {
124 // Trajectory finished, snap to the final position
125 planner->current_time = planner->total_time;
126 planner->is_active = 0;
127 planner->current_pos = planner->start_pos + planner->delta_pos;
128 }
129 else
130 {
131 // Calculate position using the cycloidal formula
132 ctrl_gt time_ratio = planner->current_time / planner->total_time;
133 ctrl_gt angle = 2.0f * CTL_CTRL_CONST_PI * time_ratio;
134 ctrl_gt pos_fraction = time_ratio - (1.0f / (2.0f * CTL_CTRL_CONST_PI)) * sinf(angle);
135
136 planner->current_pos = planner->start_pos + (planner->delta_pos * pos_fraction);
137 }
138
139 return planner->current_pos;
140}
