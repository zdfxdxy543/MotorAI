9void ctl_auto_tuning_mtr_current_ctrl(mtr_current_init_t* init)
10{
11 init->current_adc_fc = init->fs / 3;
12 init->voltage_adc_fc = init->fs / 3;
13
14 parameter_gt tau = 1.5f / init->fs;
15 // 3 ~ 5 is available
16 init->current_loop_bw = 1.0f / (3.0f * tau * CTL_PARAM_CONST_2PI);
17
18 // controller delay
19 parameter_gt control_delay = CTL_PARAM_CONST_2PI * init->current_loop_bw * tau;
20
21 // input filter delay
22 // Create a LPF object and calculate phase lag
23 ctl_filter_IIR1_t temp_filter;
24 ctl_init_filter_iir1_lpf(&temp_filter, init->fs, init->current_adc_fc);
25 parameter_gt filter_delay = ctl_get_filter_iir1_phase_lag(&temp_filter, init->fs, init->current_loop_bw);
26
27 // current controller phase lag
28 init->current_phase_lag = control_delay + filter_delay;
29
30 // calculate PI parameter based on band-width
31 parameter_gt lambda = 1.0f / init->current_loop_bw;
32
33 parameter_gt Td = init->mtr_Ld / init->mtr_Rs;
34 // parameter_gt Kd = 1.0f / init->mtr_Rs;
35
36 parameter_gt Tq = init->mtr_Lq / init->mtr_Rs;
37 // parameter_gt Kq = 1.0f / init->mtr_Rs;
38
39 // per unit gain: I_base / V_base
40 parameter_gt kp_scale = init->i_base / init->v_base;
41
42 // kp = Ldq * BW
43 init->kpd = init->mtr_Ld / (lambda + tau) * kp_scale;
44 init->kpq = init->mtr_Lq / (lambda + tau) * kp_scale;
45
46 init->kid = 1 / Td;
47 init->kiq = 1 / Tq;
48}

50void ctl_init_mtr_current_ctrl(mtr_current_ctrl_t* mc, mtr_current_init_t* init)
51{
52 int i;
53
54 // 1. Filter Init
55 for (i = 0; i < 3; ++i)
56 {
57 ctl_init_filter_iir1_lpf(&mc->filter_iuvw[i], init->fs, init->current_adc_fc);
58 }
59 ctl_init_filter_iir1_lpf(&mc->filter_udc, init->fs, init->voltage_adc_fc);
60
61 // 2. PID Init
62 ctl_init_pid(&mc->idq_ctrl[phase_d], init->kpd, init->kid, 0, init->fs);
63 ctl_init_pid(&mc->idq_ctrl[phase_q], init->kpq, init->kiq, 0, init->fs);
64
65 // 3. Lead Compensator Init
66 ctl_init_lead_form3(&mc->lead_compensator[phase_d], init->current_phase_lag, init->current_loop_bw, init->fs);
67 ctl_init_lead_form3(&mc->lead_compensator[phase_q], init->current_phase_lag, init->current_loop_bw, init->fs);
68
69 // 4. Decoupling Coefficient Calculation
70 // krpm, A, V
71 parameter_gt omega_base_elec = (init->spd_base * 1000.0f) * CTL_PARAM_CONST_PI / 30.0f * init->pole_pairs;
72 // Scale factor to convert (pu_speed * pu_current) -> pu_voltage
73 parameter_gt scale_fac = omega_base_elec * init->i_base / init->v_base;
74
75 mc->coef_ff_decouple[phase_d] = init->mtr_Lq * scale_fac;
76 mc->coef_ff_decouple[phase_q] = init->mtr_Ld * scale_fac;
77
78 // enable phasor calculate
79 mc->flag_enable_theta_calc = 1;
80
81 // 5. Voltage Limits Initialization
82 // 设定最大输出电压模值 (SVPWM 内切圆半径 或 过调制半径)
83 // 假设 v_base 定义为物理电压值，v_phase_limit 为物理限幅值
84 mc->max_vs_mag = (init->v_phase_limit * 1.4142f) / init->v_base;
85
86 // 设定母线电压补偿基准
87 // 用于 step 函数中的: v_scale = max_dcbus_voltage / mc->udc;
88 mc->max_dcbus_voltage = init->v_bus / init->v_base;
89
90 // dq轴实现方形限幅
91 ctl_set_pid_limit(&mc->idq_ctrl[phase_d], mc->max_vs_mag, -mc->max_vs_mag);
92 ctl_set_pid_int_limit(&mc->idq_ctrl[phase_d], mc->max_vs_mag, -mc->max_vs_mag);
93
94 ctl_set_pid_limit(&mc->idq_ctrl[phase_q], mc->max_vs_mag, -mc->max_vs_mag);
95 ctl_set_pid_int_limit(&mc->idq_ctrl[phase_q], mc->max_vs_mag, -mc->max_vs_mag);
96
97 // 6. Flags Initialization (Safe defaults)
98 mc->flag_enable_current_ctrl = 0; // 默认不使能
99 mc->flag_enable_theta_calc = 1; // 默认开启角度计算
100 mc->flag_enable_lead_compensator = 0; // 默认关闭超前补偿(需谨慎开启)
101 mc->flag_enable_decouple = 0; // 默认关闭解耦
102 mc->flag_enable_bus_compensation = 0; // 默认关闭母线补偿
103 mc->flag_enable_vdq_feedforward = 0; // 默认关闭前馈
104
105 // 7. Clear all states
106 ctl_clear_mtr_current_ctrl(mc);
107}

114void ctl_init_dpcc(ctl_dpcc_controller_t* dpcc, const ctl_dpcc_init_t* init)
115{
116 parameter_gt Ts = 1.0f / init->f_ctrl;
117
118 // Store base parameters
119 dpcc->rs = float2ctrl(init->Rs);
120 dpcc->ld = float2ctrl(init->Ld);
121 dpcc->lq = float2ctrl(init->Lq);
122 dpcc->psi_f = float2ctrl(init->psi_f);
123
124 // Pre-calculate coefficients to avoid divisions in the control loop
125 dpcc->ts_over_ld = float2ctrl(Ts / init->Ld);
126 dpcc->ts_over_lq = float2ctrl(Ts / init->Lq);
127 dpcc->ld_over_ts = float2ctrl(init->Ld / Ts);
128 dpcc->lq_over_ts = float2ctrl(init->Lq / Ts);
129
130 // Initialize all state variables to zero
131 ctl_clear_dpcc(dpcc);
132}

337void ctl_init_mpc(ctl_mpc_controller_t* mpc, const ctl_mpc_init_t* init)
338{
339 mpc->optimal_vector_index = 0; // Default to zero vector
340 mpc->Ld = (ctrl_gt)init->Ld;
341 mpc->Lq = (ctrl_gt)init->Lq;
342 mpc->psi_f = (ctrl_gt)init->psi_f;
343 mpc->Ts = 1.0f / (ctrl_gt)init->f_ctrl;
344
345 // Pre-calculate the constant parts of the discrete-time model matrices (A and B)
346 // i(k+1) = A*i(k) + B*u(k) + E
347 // A = [[1 - Ts*Rs/Ld, Ts*we*Lq/Ld], [-Ts*we*Ld/Lq, 1 - Ts*Rs/Lq]]
348 // B = [[Ts/Ld, 0], [0, Ts/Lq]]
349 // E = [0, -Ts*we*psi_f/Lq]
350
351 ctrl_gt ts_rs_over_ld = mpc->Ts * (ctrl_gt)init->Rs / mpc->Ld;
352 ctrl_gt ts_rs_over_lq = mpc->Ts * (ctrl_gt)init->Rs / mpc->Lq;
353
354 ctl_matrix2_set(&mpc->A_const, 0, 0, 1.0f - ts_rs_over_ld);
355 ctl_matrix2_set(&mpc->A_const, 0, 1, 0.0f); // Speed dependent term
356 ctl_matrix2_set(&mpc->A_const, 1, 0, 0.0f); // Speed dependent term
357 ctl_matrix2_set(&mpc->A_const, 1, 1, 1.0f - ts_rs_over_lq);
358
359 ctl_matrix2_set(&mpc->B, 0, 0, mpc->Ts / mpc->Ld);
360 ctl_matrix2_set(&mpc->B, 0, 1, 0.0f);
361 ctl_matrix2_set(&mpc->B, 1, 0, 0.0f);
362 ctl_matrix2_set(&mpc->B, 1, 1, mpc->Ts / mpc->Lq);
363}
