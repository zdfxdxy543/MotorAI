71void ctl_init()
72{
73 //
74 // stop here and wait for user start the motor controller
75 //
76 ctl_fast_disable_output();
77
78 //
79 // motor current controller init objects
80 //
81 mtr_ctrl_init.fs = CONTROLLER_FREQUENCY;
82 mtr_ctrl_init.v_base = CTRL_VOLTAGE_BASE;
83 mtr_ctrl_init.i_base = CTRL_CURRENT_BASE;
84
85 mtr_ctrl_init.v_bus = CTRL_VOLTAGE_BASE;
86 mtr_ctrl_init.v_phase_limit = MOTOR_PARAM_RATED_VOLTAGE;
87
88 mtr_ctrl_init.freq_base = MOTOR_PARAM_RATED_FREQUENCY;
89 mtr_ctrl_init.spd_base = MOTOR_PARAM_MAX_SPEED / 1000;
90 mtr_ctrl_init.pole_pairs = MOTOR_PARAM_POLE_PAIRS;
91
92 mtr_ctrl_init.mtr_Ld = MOTOR_PARAM_LS;
93 mtr_ctrl_init.mtr_Lq = MOTOR_PARAM_LS;
94 mtr_ctrl_init.mtr_Rs = MOTOR_PARAM_RS;
95
96 ctl_auto_tuning_mtr_current_ctrl(&mtr_ctrl_init);
97 ctl_init_mtr_current_ctrl(&mtr_ctrl, &mtr_ctrl_init);
98
99 //
100 // init SPWM modulator
101 //
102#if defined USING_NPC_MODULATOR
103 ctl_init_npc_modulator(&spwm, CTRL_PWM_CMP_MAX, CTRL_PWM_DEADBAND_CMP, &mtr_ctrl.iuvw, float2ctrl(0.02),
104 float2ctrl(0.005));
105#else
106 ctl_init_spwm_modulator(&spwm, CTRL_PWM_CMP_MAX, CTRL_PWM_DEADBAND_CMP, &mtr_ctrl.iuvw, float2ctrl(0.02),
107 float2ctrl(0.005));
108#endif // USING_NPC_MODULATOR
109
110 //
111 // angle signal generator
112 //
113 ctl_init_const_slope_f_pu_controller(
114 // ramp angle generator
115 &rg,
116 // target frequency (Hz), target frequency slope (Hz/s)
117 20.0f, 20.0f,
118 // rated krpm, pole pairs
119 MOTOR_PARAM_MAX_SPEED / 1000.0f, mtr_ctrl_init.pole_pairs,
120 // ISR frequency
121 CONTROLLER_FREQUENCY);
122
123 //
124 // motion controller
125 //
126 ctl_init_vel_pos_ctrl(
127 // controller object
128 &motion_ctrl,
129 // spd_kp, pos_kp, spd_ki, pos_ki
130 1.0f, 1.0f, 1.0f, 1.0f,
131 // spd_lim, spd_slope_lim, cur_lim
132 1.0f, 0.2f, 0.3f,
133 // spd_div, pos_div, controller freq
134 CTRL_SPD_DIV, CTRL_POS_DIV, CONTROLLER_FREQUENCY);
135
136 //
137 // Encoder Init
138 //
139 ctl_init_autoturn_pos_encoder(&pos_enc, mtr_ctrl_init.pole_pairs, CTRL_POS_ENC_FS);
140 ctl_set_autoturn_pos_encoder_mech_offset(&pos_enc, float2ctrl(CTRL_POS_ENC_BIAS));
141
142 ctl_init_spd_calculator(&spd_enc, &pos_enc.encif, CONTROLLER_FREQUENCY, CTRL_SPD_DIV, MOTOR_PARAM_MAX_SPEED, 20.0f);
143
144#ifdef ENABLE_SMO
145
146 //
147 // Observer Init
148 //
149 ctl_auto_tuning_pmsm_smo(&smo_init, &mtr_ctrl_init);
150 ctl_init_pmsm_smo(&smo, &smo_init);
151
152#endif // ENABLE_SMO
153
154// attach motor current controller with input port
155#if BUILD_LEVEL <= 2
156 ctl_attach_mtr_current_ctrl_port(&mtr_ctrl, &iuvw.control_port, &udc.control_port, &rg.enc, &spd_enc.encif);
157#else // BUILD_LEVEL
158 ctl_attach_mtr_current_ctrl_port(&mtr_ctrl, &iuvw.control_port, &udc.control_port, &pos_enc.encif, &spd_enc.encif);
159#endif // BUILD_LEVEL
160
161 ctl_attach_vel_pos_ctrl(&motion_ctrl, &pos_enc.encif, &spd_enc.encif);
162
163#if BUILD_LEVEL == 1
164 // Voltage open loop
165 ctl_disable_mtr_current_ctrl(&mtr_ctrl);
166 ctl_set_mtr_current_ctrl_vdq_ref(&mtr_ctrl, 0.0, 0.0);
167
168#elif BUILD_LEVEL == 2
169 // Basic current close loop, IF
170 ctl_enable_mtr_current_ctrl(&mtr_ctrl);
171 ctl_set_mtr_current_ctrl_ref(&mtr_ctrl, float2ctrl(0.1), float2ctrl(0.1));
172
173#elif BUILD_LEVEL == 3
174 // Basic current close loop, inverter
175 ctl_enable_mtr_current_ctrl(&mtr_ctrl);
176 ctl_set_mtr_current_ctrl_ref(&mtr_ctrl, float2ctrl(0.1), float2ctrl(0.1));
177
178#elif BUILD_LEVEL == 4
179 // Basic Speed close loop
180 ctl_enable_mtr_current_ctrl(&mtr_ctrl);
181 ctl_enable_velocity_ctrl(&motion_ctrl);
182 ctl_set_target_velocity(&motion_ctrl, 0.1);
183
184#endif // BUILD_LEVEL
185
186 //
187 // init and config CiA402 standard state machine
188 //
189 init_cia402_state_machine(&cia402_sm);
190 cia402_sm.minimum_transit_delay[3] = 100;
191
192#if defined SPECIFY_PC_ENVIRONMENT
193 cia402_sm.flag_enable_control_word = 0;
194 cia402_sm.current_cmd = CIA402_CMD_ENABLE_OPERATION;
195#endif // SPECIFY_PC_ENVIRONMENT
196
197 //
198 // init and config Motor Protection module
199 //
200 ctl_init_mtr_protect(&protection, CONTROLLER_FREQUENCY);
201 ctl_attach_mtr_protect_port(&protection, &mtr_ctrl.udc, (ctl_vector2_t*)&mtr_ctrl.idq0, &mtr_ctrl.idq_ref, NULL,
202 NULL);
203 ctl_set_mtr_protect_mask(&protection, MTR_PROT_DEVIATION);
204
205 //
206 // init ADC Calibrator
207 //
208 ctl_init_adc_calibrator(&adc_calibrator, 20, 0.707f, CONTROLLER_FREQUENCY);
209
210 if (flag_enable_adc_calibrator)
211 {
212 ctl_enable_adc_calibrator(&adc_calibrator);
213 }
214}

229gmp_task_status_t tsk_protect(gmp_task_t* tsk)
230{
231 GMP_UNUSED_VAR(tsk);
232
233 uint32_t error_mask = ctl_get_mtr_protect_mask(&protection);
234
235 if (mtr_ctrl.flag_enable_current_ctrl)
236 {
237 error_mask = error_mask & ~MTR_PROT_DEVIATION;
238 }
239 else
240 {
241 error_mask = error_mask | MTR_PROT_DEVIATION;
242 }
243
244 ctl_set_mtr_protect_mask(&protection, error_mask);
245
246#ifdef ENABLE_MOTOR_FAULT_PROTECTION
247 if (ctl_dispatch_mtr_protect_slow(&protection))
248 {
249 cia402_fault_request(&cia402_sm);
250 }
251#endif // ENABLE_MOTOR_FAULT_PROTECTION
252
253 return GMP_TASK_DONE;
254}

279fast_gt ctl_exec_adc_calibration(void)
280{
281 //
282 // 1. ADC Auto calibrate
283 //
284 if (flag_enable_adc_calibrator)
285 {
286 if (ctl_is_adc_calibrator_cmpt(&adc_calibrator) && ctl_is_adc_calibrator_result_valid(&adc_calibrator))
287 {
288
289 // index_adc_calibrator == 7, for Ibus
290 if (index_adc_calibrator == 7)
291 {
292 // vbus get result
293 idc.bias = idc.bias + ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), idc.gain);
294
295 // move to next position
296 index_adc_calibrator += 1;
297
298 // adc calibrate process done.
299 flag_enable_adc_calibrator = 0;
300
301 // clear INV controller
302 clear_all_controllers();
303
304 // ADC Calibrator complete here.
305 //ctl_enable_gfl_inv(&inv_ctrl);
306 }
307
308 // index_adc_calibrator == 6, for Vbus
309 else if (index_adc_calibrator == 6)
310 {
311 // vbus get result
312 //udc.bias = udc.bias + ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), udc.gain);
313
314 // move to next position
315 index_adc_calibrator += 1;
316
317 // clear calibrator
318 ctl_clear_adc_calibrator(&adc_calibrator);
319
320 // enable calibrator to next position
321 ctl_enable_adc_calibrator(&adc_calibrator);
322 }
323
324 // index_adc_calibrator == 5 ~ 3, for Vuvw
325 else if (index_adc_calibrator <= 5 && index_adc_calibrator >= 3)
326 {
327 // vuvw get result
328 uuvw.bias[index_adc_calibrator - 3] =
329 uuvw.bias[index_adc_calibrator - 3] +
330 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), uuvw.gain[index_adc_calibrator - 3]);
331
332 // move to next position
333 index_adc_calibrator += 1;
334
335 // clear calibrator
336 ctl_clear_adc_calibrator(&adc_calibrator);
337
338 // enable calibrator to next position
339 ctl_enable_adc_calibrator(&adc_calibrator);
340 }
341
342 // index_adc_calibrator == 2 ~ 0, for Iuvw
343 else if (index_adc_calibrator <= 2)
344 {
345 // iuvw get result
346 iuvw.bias[index_adc_calibrator] =
347 iuvw.bias[index_adc_calibrator] +
348 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), iuvw.gain[index_adc_calibrator]);
349
350 // move to next position
351 index_adc_calibrator += 1;
352
353 // clear calibrator
354 ctl_clear_adc_calibrator(&adc_calibrator);
355
356 // enable calibrator to next position
357 ctl_enable_adc_calibrator(&adc_calibrator);
358 }
359
360 // over-range protection
361 if (index_adc_calibrator > 13)
362 flag_enable_adc_calibrator = 0;
363 }
364
365 // ADC calibrate is not complete
366 return 0;
367 }
368
369 // skip calibrate routine
370 return 1;
371}
