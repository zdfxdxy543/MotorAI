65GMP_STATIC_INLINE ctrl_gt _nladrc_fal(ctrl_gt e, ctrl_gt alpha, ctrl_gt delta)
66{
67 if (ctl_abs(e) > delta)
68 {
69 return ctl_pow(ctl_abs(e), alpha) * _nladrc_sign(e);
70 }
71 else
72 {
73 return e / ctl_pow(delta, 1.0f - alpha);

170GMP_STATIC_INLINE ctrl_gt ctl_step_nladrc(ctl_nladrc_controller_t* adrc, ctrl_gt target, ctrl_gt feedback)
171{
172 // --- 1. Tracking Differentiator (TD) ---
173 // Calculate the target trajectory v1 and its derivative v2
174 ctrl_gt fh = _nladrc_fal(adrc->td.x1 - target, 0.5f, adrc->td.h * adrc->td.r);
175 adrc->td.x1 += adrc->td.h * adrc->td.x2;
176 adrc->td.x2 += adrc->td.h * fh;
177
178 // --- 2. Extended State Observer (ESO) ---
179 // Update state estimates using the previous control input u(k-1) and the current system output y(k)
180 ctrl_gt e = adrc->eso.z1 - feedback; // Observation error
181 ctrl_gt fe = _nladrc_fal(e, 0.5f, adrc->eso.h); // Nonlinear function
182 ctrl_gt fe1 = _nladrc_fal(e, 0.25f, adrc->eso.h);
183
184 adrc->eso.z1 += adrc->eso.h * (adrc->eso.z2 - adrc->eso.beta01 * e);
185 adrc->eso.z2 +=
186 adrc->eso.h * (adrc->eso.z3 - adrc->eso.beta02 * fe + adrc->eso.b0 * adrc->output); // u(k-1) is adrc->output
187 adrc->eso.z3 += adrc->eso.h * (-adrc->eso.beta03 * fe1);
188
189 // --- 3. Nonlinear State Error Feedback (NLSEF) ---
190 // Calculate state errors
191 ctrl_gt e1 = adrc->td.x1 - adrc->eso.z1; // Position/speed error
192 ctrl_gt e2 = adrc->td.x2 - adrc->eso.z2; // Velocity/acceleration error
193
194 // Calculate the nonlinear combination u0
195 ctrl_gt u0 = adrc->nlsef.kp * _nladrc_fal(e1, adrc->nlsef.alpha1, adrc->nlsef.delta) +
196 adrc->nlsef.kd * _nladrc_fal(e2, adrc->nlsef.alpha2, adrc->nlsef.delta);
197
198 // --- Key Point Analysis 4: Total Disturbance Compensation ---
199 // This is the essence of ADRC: subtract the total disturbance z3 estimated by the ESO from the controller output u0, then divide by the control gain b0.
200 // This operation actively cancels out all internal and external disturbances in real-time.
201 adrc->output = (u0 - adrc->eso.z3) / adrc->eso.b0;
202
203 // Output saturation
204 adrc->output = ctl_sat(adrc->output, adrc->out_max, adrc->out_min);
205
206 return adrc->output;
207}
