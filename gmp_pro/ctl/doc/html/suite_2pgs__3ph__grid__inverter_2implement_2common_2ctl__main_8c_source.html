196void ctl_mainloop(void)
197{
198 // When the program is reach here, the following things will happen:
199 // 1. software non-block delay 500ms
200 // 2. judge if spll theta error convergence has occurred
201 // 3. then enable system
202
203 if (flag_system_enable)
204 {
205
206 // first time flag
207 // log the first time enable the system
208 if (firsttime_flag == 0)
209 {
210 tick_bias = gmp_base_get_system_tick();
211 firsttime_flag = 1;
212 }
213
214 // a delay of 500ms
215 if ((started_flag == 0) && ((gmp_base_get_system_tick() - tick_bias) > 100) && (startup_flag == 0))
216 {
217 startup_flag = 1;
218 }
219
220 // judge if PLL is close to target
221 if ((started_flag == 0) && (startup_flag == 1) && ctl_ready_mainloop())
222 {
223 ctl_enable_output();
224 started_flag = 1;
225 }
226 }
227 // if system is disabled
228 else // (flag_system_enable == 0)
229 {
230 ctl_disable_output();
231
232 // clear all flags
233 firsttime_flag = 0;
234 startup_flag = 0;
235 started_flag = 0;
236 tick_bias = 0;
237 }
238
239 return;
240}

251 //if (pmsm_ctrl.flag_enable_controller)

376// This mainloop will run again and again to judge if system meets online condition,
377// when flag_system_enable is set.
378// if return 1 the system is ready to enable.
379// if return 0 the system is not ready to enable
380fast_gt ctl_ready_mainloop(void)
381{
382 ctl_clear_three_phase_inv(&inv_ctrl);
383
384#if defined SPECIFY_ENABLE_ADC_CALIBRATE
385 if (ctl_adc_calibrate())
386 {
387#endif // SPECIFY_ENABLE_ADC_CALIBRATE
388
389 if (ctl_is_three_phase_inverter_mode(&inv_ctrl))
390 return 1;
391 else
392 return (ctl_abs(ctl_get_three_phase_pll_error(&inv_ctrl)) < CTRL_SPLL_EPSILON);
393
394#if defined SPECIFY_ENABLE_ADC_CALIBRATE
395 }
396 else
397 return 0;
398#endif // SPECIFY_ENABLE_ADC_CALIBRATE
