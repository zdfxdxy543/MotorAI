101GMP_STATIC_INLINE ctrl_gt ctl_step_mppt_inc_algo(mppt_inc_algo_t* _mppt_obj)
102{
103 ctrl_gt current_voltage = _mppt_obj->adc_u->value;
104 ctrl_gt current_current = _mppt_obj->adc_i->value;
105
106 // MPPT algorithm logic
107 if (_mppt_obj->flag_enable_mppt)
108 {
109 ctrl_gt delta_voltage = current_voltage - _mppt_obj->last_voltage;
110 ctrl_gt delta_current = current_current - _mppt_obj->last_current;
111
112 // Check for dV=0 edge case
113 if (delta_voltage == 0)
114 {
115 if (delta_current > 0)
116 {
117 // Power increased, keep moving in the same direction (increase v_ref)
118 _mppt_obj->v_ref += _mppt_obj->voltage_increment;
119 }
120 else if (delta_current < 0)
121 {
122 // Power decreased, reverse direction (decrease v_ref)
123 _mppt_obj->v_ref -= _mppt_obj->voltage_increment;
124 }
125 }
126 else // Standard INC logic for dV != 0
127 {
128 // The core condition is dI/dV == -I/V at MPP.
129 // Rearranging gives I + V * (dI/dV) == 0.
130 // We check the sign of this expression to determine direction.
131 ctrl_gt inc_cond = current_current + ctl_mul(ctl_div(delta_current, delta_voltage), current_voltage);
132
133 if (inc_cond > 0)
134 {
135 // dI/dV > -I/V, left of MPP, increase voltage
136 _mppt_obj->v_ref += _mppt_obj->voltage_increment;
137 }
138 else if (inc_cond < 0)
139 {
140 // dI/dV < -I/V, right of MPP, decrease voltage
141 _mppt_obj->v_ref -= _mppt_obj->voltage_increment;
142 }
143 // if inc_cond == 0, we are at MPP, do nothing.
144 }
145
146 // Apply voltage limits
147 _mppt_obj->v_ref = ctl_sat(_mppt_obj->v_ref, _mppt_obj->max_voltage_limit, _mppt_obj->min_voltage_limit);
148 }
149
150 // Update state for the next iteration
151 _mppt_obj->last_voltage = current_voltage;
152 _mppt_obj->last_current = current_current;
153
154 // Return the calculated voltage reference
155 return _mppt_obj->v_ref;
156}
