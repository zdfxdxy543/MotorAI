101{

104
105 // Calculate numerator part: b0*x(n) + b1*x(n-1) + ...
106 output = ctl_mul(obj->num_coeffs[0], input);
107 for (i = 0; i < obj->num_order; ++i)
108 {
109 output += ctl_mul(obj->num_coeffs[i + 1], obj->input_buffer[i]);
110 }
111
112 // Calculate denominator part: -a1*y(n-1) - a2*y(n-2) - ...
113 for (i = 0; i < obj->den_order; ++i)
114 {
115 output -= ctl_mul(obj->den_coeffs[i], obj->output_buffer[i]);
116 }
117
118 // Update state buffers for the next iteration
119 // Shift past outputs
120 for (i = obj->den_order - 1; i > 0; --i)
121 {
122 obj->output_buffer[i] = obj->output_buffer[i - 1];
123 }
124 if (obj->den_order > 0)

127 }

133 }

135 {
136 obj->input_buffer[0] = input;
137 }
