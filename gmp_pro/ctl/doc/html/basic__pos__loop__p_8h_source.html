123GMP_STATIC_INLINE void ctl_step_pos_controller(ctl_pos_controller_t* pc)
124{
125 if (ctl_step_divider(&pc->div))
126 {
127 // 1. Calculate the error in full revolutions.
128 int32_t rev_error = pc->target_revs - pc->actual_revs;
129
130 // 2. Calculate the error in the fractional angle.
131 ctrl_gt angle_error = pc->target_angle - pc->actual_angle;
132
133 // Correct for angle wrap-around to ensure the shortest path is taken.
134 if (angle_error > 0.5f)
135 {
136 angle_error -= 1.0f;
137 }
138 else if (angle_error < -0.5f)
139 {
140 angle_error += 1.0f;
141 }
142
143 // 3. Combine the errors into a single total position error.
144 // The revolution error is scaled by 1.0 (representing a full turn)
145 ctrl_gt total_error = (ctrl_gt)rev_error + angle_error;
146
147 // 4. Apply the proportional gain to get the speed reference.
148 pc->speed_ref = ctl_mul(pc->kp, total_error);
149
150 // 5. Saturate the output to the defined speed limit.
151 pc->speed_ref = ctl_sat(pc->speed_ref, pc->speed_limit, -pc->speed_limit);
152 }
153}
