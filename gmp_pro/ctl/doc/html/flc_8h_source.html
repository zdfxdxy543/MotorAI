128GMP_STATIC_INLINE ctrl_gt ctl_step_flc(const flc_controller_t* flc, ctrl_gt error, ctrl_gt error_rate)
129{
130 // 1. Apply input gains to normalize the inputs
131 ctrl_gt norm_error = ctl_step_saturation(&flc->sat_e, ctl_mul(error, flc->ge));
132 ctrl_gt norm_error_rate = ctl_step_saturation(&flc->sat_ce, ctl_mul(error_rate, flc->gce));
133
134 // 3. Perform interpolation using the look-up table
135 ctrl_gt norm_output = ctl_step_interpolate_uniform_lut2d(&flc->lut, norm_error, norm_error_rate);
136
137 // 4. Apply output gain to get the final control value
138 flc->out = ctl_step_saturation(&flc->sat_u, ctl_mul(norm_output, flc->gu));
139
140 return flc->out;
141}
