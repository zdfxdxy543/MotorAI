104GMP_STATIC_INLINE void ctl_step_bldc_hall_estimator(ctl_bldc_hall_estimator_t* est, uint8_t hall_state,
105 uint32_t capture_time)
106{
107 // --- Speed Estimation ---
108 if (hall_state != est->last_hall_state && hall_state >= 1 && hall_state <= 6)
109 {
110 // A Hall state change has occurred
111 uint32_t time_diff = capture_time - est->last_capture_time;
112 est->last_capture_time = capture_time;
113 est->last_hall_state = hall_state;
114
115 if (time_diff > 0) // Avoid division by zero
116 {
117 est->sector_time_delta = (ctrl_gt)time_diff;
118 }
119
120 // Update the coarse angle based on the new state
121 est->coarse_angle_pu = HALL_STATE_TO_ANGLE_PU[hall_state];
122 }
123
124 // Calculate raw speed based on the time to cross the last sector
125 ctrl_gt raw_speed_rad_per_tick = 0.0f;
126 if (est->sector_time_delta > 0.0f)
127 {
128 raw_speed_rad_per_tick = est->sector_rad / est->sector_time_delta;
129 }
130
131 // Low-pass filter the speed estimate
132 est->estimated_speed_rad_per_tick +=
133 est->speed_filter_k * (raw_speed_rad_per_tick - est->estimated_speed_rad_per_tick);
134
135 // Update the speed output interface (in per-unit)
136 est->spdif.speed = est->estimated_speed_rad_per_tick * est->ticks_to_pu_sf;
137
138 // --- Position Interpolation ---
139 // Calculate time elapsed since the last Hall edge
140 uint32_t time_since_capture = capture_time - est->last_capture_time;
141
142 // Calculate the interpolated angle offset within the current sector
143 ctrl_gt angle_offset_rad = (ctrl_gt)time_since_capture * est->estimated_speed_rad_per_tick;
144 ctrl_gt angle_offset_pu = angle_offset_rad / (2.0f * CTL_CTRL_CONST_PI);
145
146 // Add the offset to the coarse angle to get the fine angle
147 ctrl_gt fine_elec_angle = est->coarse_angle_pu + angle_offset_pu;
148
149 // Update the position output interface
150 est->encif.elec_position = ctrl_mod_1(fine_elec_angle);
151 est->encif.position = est->encif.elec_position / est->pole_pairs; // Simplified mechanical angle
152}
153 // end of BLDC_HALL_ESTIMATOR group
