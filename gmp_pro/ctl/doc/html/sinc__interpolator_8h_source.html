97GMP_STATIC_INLINE ctrl_gt ctl_step_sinc_interpolator(ctl_sinc_interpolator_t* sinc, ctrl_gt input,
98 parameter_gt fractional_delay)
99{
100 uint32_t i;
101
102 // --- Key Point Analysis 2: Update the Circular Buffer for Input Samples ---
103 sinc->buffer[sinc->buffer_index] = input;
104
105 // --- Key Point Analysis 3: Select FIR Filter Coefficients Based on Fractional Delay ---
106 // 'fractional_delay' is mapped to an index in the pre-calculated table, selecting
107 // the best-matching set of Sinc coefficients.
108 uint32_t table_index = (uint32_t)(fractional_delay * sinc->table_size);
109 if (table_index >= sinc->table_size)
110 {
111 table_index = sinc->table_size - 1;
112 }
113 ctrl_gt* kernel = sinc->sinc_table[table_index];
114
115 // --- Key Point Analysis 4: Perform the Convolution Operation ---
116 // The output is calculated by convolving (dot product) the selected Sinc kernel
117 // (FIR coefficients) with the samples in the input buffer.
118 sinc->output = 0.0f;
119 uint32_t j = sinc->buffer_index;
120 for (i = 0; i < sinc->num_taps; i++)
121 {
122 sinc->output += kernel[i] * sinc->buffer[j];
123 if (j == 0)
124 {
125 j = sinc->num_taps - 1;
126 }
127 else
128 {
129 j--;
130 }
131 }
132
133 // Advance the circular buffer index for the next sample.
134 sinc->buffer_index++;
135 if (sinc->buffer_index >= sinc->num_taps)
136 {
137 sinc->buffer_index = 0;
138 }
139
140 return sinc->output;
141}
