299GMP_STATIC_INLINE void ctl_step_gfl_inv_ctrl(gfl_inv_ctrl_t* gfl)
300{
301 // assert critical pointer
302 gmp_base_assert(gfl);
303 gmp_base_assert(gfl->adc_iabc);
304 gmp_base_assert(gfl->adc_vabc);
305
306 gfl->isr_tick += 1;
307
308 // --- 1. Input Filtering and Coordinate Transformation ---
309 ctl_step_filter_iir1(&gfl->filter_udc, gfl->adc_udc->value);
310 ctl_step_filter_iir1(&gfl->filter_idc, gfl->adc_idc->value);
311
312 // current sensor
313#if GFL_CURRENT_SAMPLE_PHASE_MODE == 3
314 gfl->iabc.dat[phase_A] = ctl_step_filter_iir1(&gfl->filter_iabc[phase_A], gfl->adc_iabc->value.dat[phase_A]);
315 gfl->iabc.dat[phase_B] = ctl_step_filter_iir1(&gfl->filter_iabc[phase_B], gfl->adc_iabc->value.dat[phase_B]);
316 gfl->iabc.dat[phase_C] = ctl_step_filter_iir1(&gfl->filter_iabc[phase_C], gfl->adc_iabc->value.dat[phase_C]);
317
318 ctl_ct_clarke(&gfl->iabc, &gfl->iab0);
319
320#elif GFL_CURRENT_SAMPLE_PHASE_MODE == 2
321 gfl->iabc.dat[phase_A] = ctl_step_filter_iir1(&gfl->filter_iabc[phase_A], gfl->adc_iabc->value.dat[phase_A]);
322 gfl->iabc.dat[phase_B] = ctl_step_filter_iir1(&gfl->filter_iabc[phase_B], gfl->adc_iabc->value.dat[phase_B]);
323 gfl->iabc.dat[phase_C] = 0;
324
325 ctl_ct_clarke_2ph((ctl_vector2_t*)&gfl->iabc, (ctl_vector2_t*)&gfl->iab0);
326 gfl->iab0.dat[phase_0] = 0;
327
328#endif // GFL_CURRENT_SAMPLE_PHASE_MODE
329
330 // voltage sensor
331#if GFL_VOLTAGE_SAMPLE_PHASE_MODE == 3
332 gfl->vabc.dat[phase_A] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_A], gfl->adc_vabc->value.dat[phase_A]);
333 gfl->vabc.dat[phase_B] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_B], gfl->adc_vabc->value.dat[phase_B]);
334 gfl->vabc.dat[phase_C] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_C], gfl->adc_vabc->value.dat[phase_C]);
335
336 ctl_ct_clarke(&gfl->vabc, &gfl->vab0);
337
338#elif GFL_VOLTAGE_SAMPLE_PHASE_MODE == 2
339 gfl->vabc.dat[phase_A] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_A], gfl->adc_vabc->value.dat[phase_A]);
340 gfl->vabc.dat[phase_B] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_B], gfl->adc_vabc->value.dat[phase_B]);
341 gfl->vabc.dat[phase_C] = 0;
342
343 ctl_ct_clarke_2ph((ctl_vector2_t*)&gfl->vabc, (ctl_vector2_t*)&gfl->vab0);
344 gfl->vab0.dat[phase_0] = 0;
345
346#elif GFL_VOLTAGE_SAMPLE_PHASE_MODE == 1
347 gfl->vabc.dat[phase_UAB] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_UAB], gfl->adc_vabc->value.dat[phase_UAB]);
348 gfl->vabc.dat[phase_UBC] = ctl_step_filter_iir1(&gfl->filter_uabc[phase_UBC], gfl->adc_vabc->value.dat[phase_UBC]);
349 gfl->vabc.dat[phase_0] = 0;
350 ctl_ct_clarke_from_line((ctl_vector2_t*)&gfl->vabc, (ctl_vector2_t*)&gfl->vab0);
351 gfl->vab0.dat[phase_0] = 0;
352
353#endif // GFL_VOLTAGE_SAMPLE_PHASE_MODE
354
355 // --- 2. Grid Synchronization (PLL) ---
356 if (gfl->flag_enable_pll)
357 {
358#ifdef USING_DSOGI_PLL
359 ctl_step_dsogi_pll(&gfl->pll, gfl->vab0.dat[phase_alpha], gfl->vab0.dat[phase_beta]);
360#else
361 ctl_step_pll_3ph(&gfl->pll, gfl->vab0.dat[phase_alpha], gfl->vab0.dat[phase_beta]);
362#endif // USING_DSOGI_PLL
363 }
364
365 // --- 3. Main Control Logic ---
366 if (gfl->flag_enable_system)
367 {
368 // --- 3a. Angle and Phasor Generation ---
369 if (gfl->flag_enable_offgrid)
370 {
371 gfl->angle = ctl_step_ramp_generator(&gfl->rg);
372 ctl_set_phasor_via_angle(gfl->angle, &gfl->phasor);
373 }
374 // using external phasor
375 else if (gfl->flag_enable_external_phasor)
376 {
377 gmp_base_assert(gfl->phasor_ext);
378 ctl_vector2_copy(&gfl->phasor, gfl->phasor_ext);
379 }
380 // using PLL phasor
381 else
382 {
383#ifdef USING_DSOGI_PLL
384 gfl->angle = gfl->pll.srf_pll.theta;
385 ctl_vector2_copy(&gfl->phasor, &gfl->pll.srf_pll.phasor);
386#else
387 gfl->angle = gfl->pll.theta;
388 ctl_vector2_copy(&gfl->phasor, &gfl->pll.phasor);
389#endif // USING_DSOGI_PLL
390 }
391
392 // --- 3b. Park Transformation ---
393 ctl_ct_park2((ctl_vector2_t*)&gfl->iab0, &gfl->phasor, &gfl->idq);
394 ctl_ct_park2((ctl_vector2_t*)&gfl->vab0, &gfl->phasor, &gfl->vdq);
395
396 // --- 3c. current controller ---
397 // if current controller is disabled the vdq_out would output directly.
398 if (gfl->flag_enable_current_ctrl)
399 {
400 gfl->vdq_out.dat[phase_d] =
401 ctl_step_pid_ser(&gfl->pid_idq[phase_d], gfl->idq_set.dat[phase_d] - gfl->idq.dat[phase_d]);
402 gfl->vdq_out.dat[phase_q] =
403 ctl_step_pid_ser(&gfl->pid_idq[phase_q], gfl->idq_set.dat[phase_q] - gfl->idq.dat[phase_q]);
404
405 // decouple
406 if (gfl->flag_enable_decouple)
407 {
408 gfl->vdq_ff_decouple.dat[phase_d] = gfl->coef_ff_decouple * gfl->idq.dat[phase_q];
409 gfl->vdq_ff_decouple.dat[phase_q] = -gfl->coef_ff_decouple * gfl->idq.dat[phase_d];
410 }
411 else
412 {
413 gfl->vdq_ff_decouple.dat[phase_d] = 0;
414 gfl->vdq_ff_decouple.dat[phase_q] = 0;
415 }
416
417 // active damping
418 if (gfl->flag_enable_active_damping)
419 {
420 // ĽĆËăÎ˘·ÖÁż (´ú±íµçČÝµçÁ÷ trend)
421 ctrl_gt diff_d = gfl->vdq.dat[phase_d] - gfl->vdq_last.dat[phase_d];
422 ctrl_gt diff_q = gfl->vdq.dat[phase_q] - gfl->vdq_last.dat[phase_q];
423
424 // damping filter
425 gfl->vdq_ff_damping.dat[phase_d] =
426 ctl_step_biquad_filter(&gfl->filter_damping[phase_d], gfl->coef_ff_damping * diff_d);
427 gfl->vdq_ff_damping.dat[phase_q] =
428 ctl_step_biquad_filter(&gfl->filter_damping[phase_q], gfl->coef_ff_damping * diff_q);
429
430 ctl_vector2_copy(&gfl->vdq_last, &gfl->vdq);
431 }
432 else
433 {
434 gfl->vdq_ff_damping.dat[phase_d] = 0;
435 gfl->vdq_ff_damping.dat[phase_q] = 0;
436 }
437
438 // mix all vdq up
439 gfl->vdq_out.dat[phase_d] += gfl->vdq_ff_decouple.dat[phase_d] + gfl->vdq_ff_damping.dat[phase_d];
440 gfl->vdq_out.dat[phase_q] += gfl->vdq_ff_decouple.dat[phase_q] + gfl->vdq_ff_damping.dat[phase_q];
441 }
442
443 // --- 3d. lead compensator ---
444 if (gfl->flag_enable_lead_compensator)
445 {
446 gfl->vdq_out_comp.dat[phase_d] = ctl_step_lead(&gfl->lead_compensator[phase_d], gfl->vdq_out.dat[phase_d]);
447 gfl->vdq_out_comp.dat[phase_q] = ctl_step_lead(&gfl->lead_compensator[phase_q], gfl->vdq_out.dat[phase_q]);
448 }
449 else
450 {
451 ctl_vector2_copy(&gfl->vdq_out_comp, &gfl->vdq_out);
452 }
453
454 // --- 3e. output inverse park Transformation ---
455 ctl_ct_ipark2(&gfl->vdq_out_comp, &gfl->phasor, &gfl->vab_pos);
456
457 gfl->vab0_out.dat[phase_alpha] = gfl->vab_pos.dat[phase_alpha] + gfl->vab0_ff_external.dat[phase_alpha];
458 gfl->vab0_out.dat[phase_beta] = gfl->vab_pos.dat[phase_beta] + gfl->vab0_ff_external.dat[phase_beta];
459 gfl->vab0_out.dat[phase_0] = gfl->vab0_ff_external.dat[phase_0];
460 }
461 else
462 {
463 gfl->vab0_out.dat[phase_alpha] = 0;
464 gfl->vab0_out.dat[phase_beta] = 0;
465 gfl->vab0_out.dat[phase_0] = 0;
466 }
467}

714GMP_STATIC_INLINE void ctl_step_gfl_pq(gfl_pq_ctrl_t* pq)
715{
716 // Safety check for pointers
717 if (!pq->vdq_meas || !pq->idq_meas)
718 return;
719
720 // Local variables for readability
721 ctrl_gt vd = pq->vdq_meas->dat[phase_d];
722 ctrl_gt vq = pq->vdq_meas->dat[phase_q];
723 ctrl_gt id = pq->idq_meas->dat[phase_d];
724 ctrl_gt iq = pq->idq_meas->dat[phase_q];
725
726 // -----------------------------------------------------------
727 // 1. Calculate Instantaneous Power (Per-Unit assumption)
728 // P = vd*id + vq*iq
729 // Q = vq*id - vd*iq (Standard convention, verify with your grid standard)
730 // -----------------------------------------------------------
731 // Note: If Vq is strictly regulated to 0 by PLL, P ~= Vd*Id, Q ~= -Vd*Iq
732
733 pq->pq_meas.dat[0] = ctl_mul(vd, id) + ctl_mul(vq, iq); // Active Power P
734 pq->pq_meas.dat[1] = ctl_mul(vq, id) - ctl_mul(vd, iq); // Reactive Power Q
735
736 // -----------------------------------------------------------
737 // 2. Main Control Loop
738 // -----------------------------------------------------------
739 if (pq->flag_enable)
740 {
741 // --- Active Power Control (P -> Id) ---
742 // Error = Setpoint - Measure
743 ctrl_gt p_err = pq->pq_set.dat[0] - pq->pq_meas.dat[0];
744
745 // PID Output is Id reference
746 // Source convention: P>0 means discharging to grid.
747 pq->idq_set_out.dat[phase_d] = ctl_step_pid_ser(&pq->pid_p, p_err);
748
749 // --- Reactive Power Control (Q -> Iq) ---
750 // Error = Setpoint - Measure
751 ctrl_gt q_err = pq->pq_set.dat[1] - pq->pq_meas.dat[1];
752
753 // PID Output is Iq reference
754 // Source convention Q>0 means discharging inductive reactive power to grid.
755 pq->idq_set_out.dat[phase_q] = ctl_step_pid_ser(&pq->pid_q, q_err);
756
757 // -----------------------------------------------------------
758 // 3. Current Limiting (Circular Saturation)
759 // Prevent the reference from exceeding converter capability.
760 // -----------------------------------------------------------
761 ctrl_gt id_ref = pq->idq_set_out.dat[phase_d];
762 ctrl_gt iq_ref = pq->idq_set_out.dat[phase_q];
763
764 ctrl_gt i_mag_sq = ctl_mul(id_ref, id_ref) + ctl_mul(iq_ref, iq_ref);
765 ctrl_gt i2_limit = pq->max_i2_mag;
766
767 if (i_mag_sq > i2_limit)
768 {
769 // Simple scaling to keep vector direction but limit magnitude
770 ctrl_gt scaler = ctl_sqrt(ctl_div(i2_limit, i_mag_sq));
771 pq->idq_set_out.dat[phase_d] *= ctl_mul(scaler, pq->idq_set_out.dat[phase_d]);
772 pq->idq_set_out.dat[phase_q] *= ctl_mul(scaler, pq->idq_set_out.dat[phase_q]);
773 }
774 }
775 else
776 {
777 // If disabled, reset integrators and zero output
778 pq->idq_set_out.dat[phase_d] = 0;
779 pq->idq_set_out.dat[phase_q] = 0;
780 }
781}
