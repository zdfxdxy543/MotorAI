139GMP_STATIC_INLINE ctrl_gt ctl_step_pll_3ph(srf_pll_t* pll, ctrl_gt alpha, ctrl_gt beta)
140{
141 // Store input variables for monitoring.
142 ctl_vector2_t vdq_temp;
143 pll->e_ab.dat[phase_alpha] = alpha;
144 pll->e_ab.dat[phase_beta] = beta;
145
146 // 1. Generate the phasor {sin(theta), cos(theta)} from the current angle estimate.
147 ctl_set_phasor_via_angle(pll->theta, &pll->phasor);
148
149 // 2. Perform Park transform to get the q-axis voltage (error signal).
150 // Vq = -Valpha * sin(theta) + Vbeta * cos(theta)
151 ctl_ct_park2(&pll->e_ab, &pll->phasor, &vdq_temp);
152 pll->e_error = vdq_temp.dat[phase_q];
153 pll->v_mag = vdq_temp.dat[phase_d];
154
155 // 3. Step the PI controller with the error signal to get the frequency deviation.
156 ctl_step_pid_par(&pll->pid_pll, pll->e_error);
157
158 // 4. Update the estimated frequency (nominal frequency + deviation).
159 pll->freq_pu = float2ctrl(1.0) + pll->pid_pll.out;
160
161 // 5. Integrate the frequency to update the angle (VCO - Voltage Controlled Oscillator).
162 pll->theta += ctl_mul(pll->freq_pu, pll->freq_sf);
163 pll->theta = ctrl_mod_1(pll->theta); // Wrap angle between 0 and 1.0.
164
165 return pll->theta;
166}

283GMP_STATIC_INLINE ctrl_gt ctl_step_ddsrf_pll(ddsrf_pll_t* pll, ctrl_gt alpha, ctrl_gt beta)
284{
285 // Store inputs
286 pll->e_alpha = alpha;
287 pll->e_beta = beta;
288
289 // Prepare vector for optimized park functions
290 ctl_vector2_t v_ab_2;
291 v_ab_2.dat[phase_alpha] = alpha;
292 v_ab_2.dat[phase_beta] = beta;
293
294 // 1. Update Phasor (Theta -> Sin, Cos)
295 ctl_set_phasor_via_angle(pll->theta, &pll->phasor);
296
297 // 2. Calculate 2*Theta Phasor for Decoupling
298#if CTL_DDSRF_USE_DOUBLE_ANGLE_FORMULA == 1
299 // cos(2x) = cos^2(x) - sin^2(x)
300 // sin(2x) = 2*sin(x)*cos(x)
301 // Note: Using ctl_mul for fixed-point safety
302 ctrl_gt cos_sq = ctl_mul(pll->phasor.dat[phasor_cos], pll->phasor.dat[phasor_cos]);
303 ctrl_gt sin_sq = ctl_mul(pll->phasor.dat[phasor_sin], pll->phasor.dat[phasor_sin]);
304
305 pll->phasor_2w.dat[phasor_cos] = cos_sq - sin_sq;
306 pll->phasor_2w.dat[phasor_sin] = ctl_mul2(ctl_mul(sin_theta, cos_theta));
307
308#else
309 // calculate directly.
310 ctl_set_phasor_via_angle(ctl_mul2(pll->theta), &pll->phasor_2w);
311
312#endif //CTL_DDSRF_USE_DOUBLE_ANGLE_FORMULA
313
314 ctrl_gt cos_2theta = pll->phasor_2w.dat[phasor_cos];
315 ctrl_gt sin_2theta = pll->phasor_2w.dat[phasor_sin];
316
317 // 3. Raw Transformation (Coupled)
318 // Positive Sequence Frame (+omega)
319 ctl_ct_park2(&v_ab_2, &pll->phasor, &pll->v_pos_raw);
320
321 // Negative Sequence Frame (-omega)
322 ctl_ct_park2_neg(&v_ab_2, &pll->phasor, &pll->v_neg_raw);
323
324 // 4. Decoupling Network Calculation
325 // Use the *filtered DC values* from the previous step to cancel the *current raw ripple*
326
327 // --- Decouple Positive Sequence ---
328 // Remove V_neg component which appears as V_neg_dc * e^(-j2theta)
329 // d_pos* = d_pos_raw - (d_neg_dc * cos2t + q_neg_dc * sin2t)
330 // q_pos* = q_pos_raw - (q_neg_dc * cos2t - d_neg_dc * sin2t)
331
332 ctrl_gt term_p1 = ctl_mul(pll->v_neg_dc.dat[phase_d], cos_2theta);
333 ctrl_gt term_p2 = ctl_mul(pll->v_neg_dc.dat[phase_q], sin_2theta);
334
335 ctrl_gt term_p3 = ctl_mul(pll->v_neg_dc.dat[phase_q], cos_2theta);
336 ctrl_gt term_p4 = ctl_mul(pll->v_neg_dc.dat[phase_d], sin_2theta);
337
338 ctrl_gt v_pos_d_decoupled = pll->v_pos_raw.dat[phase_d] - (term_p1 + term_p2);
339 ctrl_gt v_pos_q_decoupled = pll->v_pos_raw.dat[phase_q] - (term_p3 - term_p4);
340
341 // --- Decouple Negative Sequence ---
342 // Remove V_pos component which appears as V_pos_dc * e^(+j2theta)
343 // d_neg* = d_neg_raw - (d_pos_dc * cos2t - q_pos_dc * sin2t)
344 // q_neg* = q_neg_raw - (q_pos_dc * cos2t + d_pos_dc * sin2t)
345
346 ctrl_gt term_n1 = ctl_mul(pll->v_pos_dc.dat[phase_d], cos_2theta);
347 ctrl_gt term_n2 = ctl_mul(pll->v_pos_dc.dat[phase_q], sin_2theta);
348
349 ctrl_gt term_n3 = ctl_mul(pll->v_pos_dc.dat[phase_q], cos_2theta);
350 ctrl_gt term_n4 = ctl_mul(pll->v_pos_dc.dat[phase_d], sin_2theta);
351
352 ctrl_gt v_neg_d_decoupled = pll->v_neg_raw.dat[phase_d] - (term_n1 - term_n2);
353 ctrl_gt v_neg_q_decoupled = pll->v_neg_raw.dat[phase_q] - (term_n3 + term_n4);
354
355 // 5. Low Pass Filtering (Extract DC)
356 pll->v_pos_dc.dat[phase_d] = ctl_step_filter_iir1(&pll->lpf_pos_d, v_pos_d_decoupled);
357 pll->v_pos_dc.dat[phase_q] = ctl_step_filter_iir1(&pll->lpf_pos_q, v_pos_q_decoupled);
358
359 pll->v_neg_dc.dat[phase_d] = ctl_step_filter_iir1(&pll->lpf_neg_d, v_neg_d_decoupled);
360 pll->v_neg_dc.dat[phase_q] = ctl_step_filter_iir1(&pll->lpf_neg_q, v_neg_q_decoupled);
361
362 // 6. Loop Control (PI on Positive Q-Axis)
363 // Error signal is the decoupled positive sequence q-axis voltage
364 pll->e_error = pll->v_pos_dc.dat[phase_q];
365
366 ctl_step_pid_par(&pll->pid_pll, pll->e_error);
367
368 // 7. Update Frequency and Angle
369 pll->freq_pu = float2ctrl(1.0f) + pll->pid_pll.out;
370
371 // theta = theta + freq * Ts_gain
372 pll->theta += ctl_mul(pll->freq_pu, pll->freq_sf);
373 pll->theta = ctrl_mod_1(pll->theta); // Wrap [0, 1.0]
374
375 // 8. Update Output Magnitudes
376 // Magnitudes are simply the D-component in SRF if Q is regulated to 0 (for Pos seq)
377 // For Neg seq, magnitude is sqrt(d^2 + q^2).
378 // Here we simply output the DC components, user can calculate sqrt if needed.
379 // Or approximate mag = d for pos seq.
380 pll->v_pos_mag = pll->v_pos_dc.dat[phase_d];
381
382 // For negative sequence, we might want the vector magnitude
383 // If not computationally expensive:
384 // pll->v_neg_mag = sqrt(d*d + q*q);
385 // For now, assign d component or implement explicit mag calc based on demand.
386 // Simple Approximation:
387 pll->v_neg_mag = pll->v_neg_dc.dat[phase_d]; // Placeholder if neg q is small, else needs sqrt
388
389 return pll->theta;
390}

456GMP_STATIC_INLINE ctrl_gt ctl_step_dsogi_pll(dsogi_pll_t* dsogi, ctrl_gt alpha, ctrl_gt beta)
457{
458 ctrl_gt v_alpha_p, v_alpha_q; // Alpha axis: In-phase & Quadrature
459 ctrl_gt v_beta_p, v_beta_q; // Beta axis: In-phase & Quadrature
460
461 // ---------------------------------------------------------
462 // Step 1: Run SOGI Filters
463 // ---------------------------------------------------------
464 ctl_step_discrete_sogi(&dsogi->sogi_alpha, alpha);
465 ctl_step_discrete_sogi(&dsogi->sogi_beta, beta);
466
467 // Retrieve filtered signals
468 // p = prime (in-phase/direct), q = quadrature (90 deg lag)
469 v_alpha_p = ctl_get_discrete_sogi_ds(&dsogi->sogi_alpha);
470 v_alpha_q = ctl_get_discrete_sogi_qs(&dsogi->sogi_alpha);
471
472 v_beta_p = ctl_get_discrete_sogi_ds(&dsogi->sogi_beta);
473 v_beta_q = ctl_get_discrete_sogi_qs(&dsogi->sogi_beta);
474
475 // ---------------------------------------------------------
476 // Step 2: Positive Sequence Calculation (PNSC)
477 // Formula:
478 // v_alpha+ = 0.5 * (v_alpha' - q * v_beta')
479 // v_beta+ = 0.5 * (q * v_alpha' + v_beta')
480 // ---------------------------------------------------------
481
482 // v_pos_alpha = 0.5 * (v_alpha_p - v_beta_q)
483 dsogi->v_pos_seq.dat[phase_alpha] = ctl_div2(v_alpha_p - v_beta_q);
484
485 // v_pos_beta = 0.5 * (v_alpha_q + v_beta_p)
486 dsogi->v_pos_seq.dat[phase_beta] = ctl_div2(v_alpha_q + v_beta_p);
487
488 // ---------------------------------------------------------
489 // Step 3: Run SRF-PLL with Positive Sequence Voltage
490 // ---------------------------------------------------------
491 return ctl_step_pll_3ph(&dsogi->srf_pll, dsogi->v_pos_seq.dat[phase_alpha], dsogi->v_pos_seq.dat[phase_beta]);
492}
