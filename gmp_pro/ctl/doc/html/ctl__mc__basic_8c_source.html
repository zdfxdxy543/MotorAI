193void ctl_init_const_slope_f_pu_controller(ctl_slope_f_pu_controller* ctrl, parameter_gt frequency,
194 parameter_gt freq_slope, parameter_gt rated_krpm, parameter_gt pole_pairs,
195 parameter_gt isr_freq)
196{
197 // 1. Calculate Base Frequency (Rated Electrical Hz)
198 // f_base = (RPM * Poles) / 60
199 // rated_krpm is in 1000 RPM
200 parameter_gt base_freq_hz = (rated_krpm * 1000.0f * pole_pairs) / 60.0f;
201
202 // Prevent divide by zero if user inputs bad data
203 if (base_freq_hz < 0.1f)
204 base_freq_hz = 1.0f;
205
206 // 2. Clear Positions
207 ctrl->enc.elec_position = 0;
208 ctrl->enc.position = 0;
209
210 // 3. Init Ramp Generator
211 // The slope will be updated in the step function, init with 0.
212 // Range is [0, 1) for electrical angle.
213 ctl_init_ramp_generator_via_freq(&ctrl->rg, isr_freq, 0, 1, 0);
214
215 // 4. Calculate Conversion Ratio
216 // This ratio converts "1.0 pu frequency" into "step size per ISR tick"
217 // step = (f_base / f_isr)
218 ctrl->ratio_freq_pu_to_step = float2ctrl(base_freq_hz / isr_freq);
219
220 // 5. Initialize Target Frequency in PU
221 // target_pu = target_hz / base_hz
222 ctrl->target_freq_pu = float2ctrl(frequency / base_freq_hz);
223
224 // 6. Initialize Slope Limiter
225 // The limiter needs to limit the change of PU per Tick.
226 // Max Change (Hz/s) = freq_slope
227 // Max Change (PU/s) = freq_slope / base_freq_hz
228 ctrl_gt slope_limit_per_tick = float2ctrl(freq_slope / base_freq_hz);
229
230 ctl_init_slope_limiter(&ctrl->freq_slope, slope_limit_per_tick, -slope_limit_per_tick, isr_freq);
231
232 // Set initial output of limiter to current target (assuming instant start if needed, or 0)
233 // Usually start from 0 for soft start
234 ctrl->freq_slope.out = 0;
235 ctrl->current_freq_pu = 0;
236}

262 ctrl->target_voltage = 0;
263
264#if !defined CTRL_GT_IS_FIXED
265 ctrl->v_over_f = float2ctrl(voltage_over_frequency * isr_freq);
266#elif defined CTRL_GT_IS_FLOAT
267 ctrl->v_over_f = float2ctrl(voltage_over_frequency * isr_freq / (2 ^ GLOBAL_Q));

270#endif // CTRL_GT_IS_XXX
271 ctrl->v_bias = voltage_bias;
272
273 ctl_init_slope_limiter(&ctrl->freq_slope, float2ctrl(freq_slope / isr_freq), -float2ctrl(freq_slope / isr_freq),
274 isr_freq);
275
276 ctl_init_saturation(&ctrl->volt_sat, voltage_bound, -voltage_bound);

296void ctl_init_mtr_protect(ctl_mtr_protect_t* prot, parameter_gt fs)
297{
298 // Default safe values (User should overwrite these)
299 prot->limit_ov_pu = float2ctrl(1.2f);
300 prot->limit_uv_pu = float2ctrl(0.5f);
301 prot->limit_oc_sq_pu = float2ctrl(1.5f * 1.5f); // 1.5x Overload
302 prot->limit_dev_sq_pu = float2ctrl(0.5f * 0.5f); // 0.5pu Deviation
303 prot->limit_mtr_ot = 1000; // Raw ADC value
304 prot->limit_inv_ot = 1000;
305
306 // Default Filtering
307 prot->limit_cnt_ov = (uint16_t)(fs / 2000); // Very Fast
308 prot->limit_cnt_oc = (uint16_t)(fs / 2000); // Very Fast
309 prot->limit_cnt_dev = (uint16_t)(fs / 5); // Very Slow (e.g., 200ms @ 10kHz)
310
311 prot->limit_cnt_uv = (uint16_t)(fs / 10); // Slow
312 prot->limit_cnt_mtr_ot = (uint16_t)fs;
313 prot->limit_cnt_inv_ot = (uint16_t)fs;
314
315 // Ensure minimum count is 5 to prevent false triggering on noise
316 if (prot->limit_cnt_ov <= 5)
317 prot->limit_cnt_ov = 5;
318 if (prot->limit_cnt_oc <= 5)
319 prot->limit_cnt_oc = 5;
320
321 // Enable all protections by default (Mask = 0)
322 prot->error_mask.all = 0;
323
324 ctl_clear_mtr_protect(prot);
325}

342fast_gt ctl_dispatch_mtr_protect_slow(ctl_mtr_protect_t* prot)
343{
344 // 1. Latch Check
345 if ((prot->error_code.all & (~prot->error_mask.all)) != MTR_PROT_NONE)
346 return 1;
347
348 // 2. Check Under Voltage
349 ctrl_gt u_dc = *(prot->ptr_udc);
350 if (ctl_mtr_protect_debounce(ctl_is_less(u_dc, prot->limit_uv_pu), &prot->cnt_uv, prot->limit_cnt_uv))
351 {
352 prot->error_code.bit.under_voltage = 1;
353
354 if (!prot->error_mask.bit.under_voltage)
355 {
356 return 1;
357 }
358 }
359
360 // 3. Check Motor Over Temp
361 if (prot->ptr_mtr_temp)
362 {
363 // Assuming ptr_mtr_temp->value holds the data
364 ctrl_gt temp = prot->ptr_mtr_temp->value;
365 if (ctl_mtr_protect_debounce(ctl_is_greater_ot(temp, prot->limit_mtr_ot), &prot->cnt_mtr_ot,
366 prot->limit_cnt_mtr_ot))
367 {
368 prot->error_code.bit.mtr_over_temp = 1;
369
370 if (!prot->error_mask.bit.mtr_over_temp)
371 {
372 return 1;
373 }
374 }
375 }
376
377 // 4. Check Inverter Over Temp
378 if (prot->ptr_inv_temp)
379 {
380 ctrl_gt temp = prot->ptr_inv_temp->value; // Fixed: was reading mtr_temp
381 if (ctl_mtr_protect_debounce(ctl_is_greater_ot(temp, prot->limit_inv_ot), &prot->cnt_inv_ot,
382 prot->limit_cnt_inv_ot))
383 {
384 prot->error_code.bit.inv_over_temp = 1;
385
386 if (!prot->error_mask.bit.inv_over_temp)
387 {
388 return 1;
389 }
390 }
391 }
392
393 return 0;
394}
