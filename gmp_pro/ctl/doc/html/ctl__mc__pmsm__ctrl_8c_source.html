27void ctl_init_pmsm_controller(pmsm_controller_t* ctrl, pmsm_controller_init_t* init)
28{
29#ifdef PMSM_CTRL_USING_DISCRETE_CTRL
30 // --- Initialize discrete PID controllers ---
31 // d-axis current controller
32 ctl_init_discrete_pid(&ctrl->current_ctrl[phase_d], init->current_pid_gain, init->current_Ti, init->current_Td,
33 init->fs);
34 ctl_set_discrete_pid_limit(&ctrl->current_ctrl[phase_d], init->voltage_limit_max, init->voltage_limit_min);
35
36 // q-axis current controller
37 ctl_init_discrete_pid(&ctrl->current_ctrl[phase_q], init->current_pid_gain, init->current_Ti, init->current_Td,
38 init->fs);
39 ctl_set_discrete_pid_limit(&ctrl->current_ctrl[phase_q], init->voltage_limit_max, init->voltage_limit_min);
40
41 // Velocity controller
42 ctl_init_tracking_pid(&ctrl->spd_ctrl, init->spd_pid_gain, init->spd_Ti, init->spd_Td,
43 init->current_limit_max, init->current_limit_min, init->acc_limit_max,
44 init->acc_limit_min, init->spd_ctrl_div, init->fs);
45#else // using continuous controller
46 // --- Initialize continuous PID controllers ---
47 // d-axis current controller
48 ctl_init_pid_Tmode(&ctrl->current_ctrl[phase_d], init->current_pid_gain, init->current_Ti, init->current_Td,
49 init->fs);
50 ctl_set_pid_limit(&ctrl->current_ctrl[phase_d], init->voltage_limit_max, init->voltage_limit_min);
51
52 // q-axis current controller
53 ctl_init_pid_Tmode(&ctrl->current_ctrl[phase_q], init->current_pid_gain, init->current_Ti, init->current_Td,
54 init->fs);
55 ctl_set_pid_limit(&ctrl->current_ctrl[phase_q], init->voltage_limit_max, init->voltage_limit_min);
56
57 // Velocity controller
58 ctl_init_tracking_continuous_pid(&ctrl->spd_ctrl, init->spd_pid_gain, init->spd_Ti, init->spd_Td,
59 init->current_limit_max, init->current_limit_min, init->acc_limit_max,
60 init->acc_limit_min, init->spd_ctrl_div, init->fs);
61#endif // PMSM_CTRL_USING_DISCRETE_CTRL
62
63 // --- Clear intermediate variables ---
64 ctl_vector3_clear(&ctrl->iab0);
65 ctl_vector3_clear(&ctrl->uab0);
66 ctl_vector3_clear(&ctrl->idq0);
67 ctl_vector3_clear(&ctrl->udq0);
68
69 // --- Clear feed-forward parameters ---
70 ctl_vector2_clear(&ctrl->idq_ff);
71 ctl_vector2_clear(&ctrl->vdq_ff);
72
73 // --- Clear set-point parameters ---
74 ctl_vector3_clear(&ctrl->vab0_set);
75 ctl_vector3_clear(&ctrl->vdq_set);
76 ctl_vector2_clear(&ctrl->idq_set);
77 ctrl->speed_set = 0;
78 ctrl->pos_set = 0;
79 ctrl->revolution_set = 0;
80
81 // --- Set initial state ---
82 // Disable controller and set to a safe default mode
83 ctl_disable_pmsm_ctrl(ctrl);
84 ctl_pmsm_ctrl_valphabeta_mode(ctrl);
85}
