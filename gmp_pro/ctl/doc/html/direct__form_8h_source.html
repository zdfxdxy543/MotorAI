347GMP_STATIC_INLINE ctrl_gt ctl_step_df23(ctl_df23_t* df, ctrl_gt input)
348{
349 // Calculate output u(k)
350 df->output = ctl_mul(df->b0, input) + df->d1;
351
352 // Calculate new state variables
353 ctrl_gt d1_new = ctl_mul(df->b1, input) - ctl_mul(df->a1, df->output) + df->d2;
354 ctrl_gt d2_new = ctl_mul(df->b2, input) - ctl_mul(df->a2, df->output) + df->d3;
355 ctrl_gt d3_new = ctl_mul(df->b3, input) - ctl_mul(df->a3, df->output);
356
357 // Update state variables
358 df->d1 = d1_new;
359 df->d2 = d2_new;
360 df->d3 = d3_new;
361
362 return df->output;
363}
