63void ctl_init()
64{
65 //
66 // stop here and wait for user start the motor controller
67 //
68 ctl_fast_disable_output();
69
70 //
71 // GFL inverter init objects
72 //
73 gfl_init.fs = CONTROLLER_FREQUENCY;
74 gfl_init.v_base = CTRL_VOLTAGE_BASE;
75 gfl_init.v_grid = 0.33f;
76 gfl_init.i_base = CTRL_CURRENT_BASE;
77 gfl_init.freq_base = 50.0f;
78
79 gfl_init.grid_filter_L = 1.5e-3f;
80 gfl_init.grid_filter_C = 5.0e-6f;
81
82 ctl_auto_tuning_gfl_inv(&gfl_init);
83 ctl_init_gfl_inv(&inv_ctrl, &gfl_init);
84
85 ctl_auto_tuning_neg_inv(&gfl_neg_init, &gfl_init);
86 ctl_init_neg_inv(&neg_current_ctrl, &gfl_neg_init);
87 ctl_attach_neg_inv_to_gfl(&neg_current_ctrl, &inv_ctrl);
88
89 //
90 // init SPWM modulator
91 //
92#if defined USING_NPC_MODULATOR
93 ctl_init_npc_modulator(&spwm, CTRL_PWM_CMP_MAX, CTRL_PWM_DEADBAND_CMP, &inv_ctrl.adc_iabc->value, float2ctrl(0.02),
94 float2ctrl(0.005));
95#else
96 ctl_init_spwm_modulator(&spwm, CTRL_PWM_CMP_MAX, CTRL_PWM_DEADBAND_CMP, &inv_ctrl.adc_iabc->value, float2ctrl(0.02),
97 float2ctrl(0.005));
98#endif // USING_NPC_MODULATOR
99
100 //
101 // Power controller
102 //
103 ctl_init_gfl_pq(&pq_ctrl, 0.75f, 0.001f, 0.75f, 0.001f, 1.0f, CONTROLLER_FREQUENCY);
104 ctl_attach_gfl_pq_to_core(&pq_ctrl, &inv_ctrl);
105
106#if BUILD_LEVEL == 1
107 // Voltage open loop, inverter
108 ctl_set_gfl_inv_openloop_mode(&inv_ctrl);
109 ctl_set_gfl_inv_voltage_openloop(&inv_ctrl, float2ctrl(0.6), float2ctrl(0.6));
110
111#elif BUILD_LEVEL == 2
112 // Basic current close loop, inverter
113 ctl_set_gfl_inv_current_mode(&inv_ctrl);
114 ctl_set_gfl_inv_current(&inv_ctrl, float2ctrl(0.1), float2ctrl(0.1));
115
116#elif BUILD_LEVEL == 3
117 // Basic current close loop, inverter
118 ctl_set_gfl_inv_current_mode(&inv_ctrl);
119 ctl_set_gfl_inv_current(&inv_ctrl, float2ctrl(0.1), float2ctrl(0));
120
121 ctl_enable_neg_current_inv(&neg_current_ctrl);
122
123 ctl_enable_gfl_inv_pll(&inv_ctrl);
124 ctl_set_gfl_inv_grid_connect(&inv_ctrl);
125
126#elif BUILD_LEVEL == 4
127 // current close loop with feed forward, inverter
128 ctl_set_gfl_inv_current_mode(&inv_ctrl);
129 ctl_set_gfl_inv_current(&inv_ctrl, float2ctrl(0.6), float2ctrl(0.6));
130
131 ctl_enable_gfl_inv_decouple(&inv_ctrl);
132 ctl_enable_gfl_inv_active_damp(&inv_ctrl);
133 ctl_enable_gfl_inv_lead_compensator(&inv_ctrl);
134
135#endif // BUILD_LEVEL
136
137 //
138 // init and config CiA402 standard state machine
139 //
140 init_cia402_state_machine(&cia402_sm);
141 cia402_sm.minimum_transit_delay[3] = 100;
142
143#if defined SPECIFY_PC_ENVIRONMENT
144 cia402_sm.flag_enable_control_word = 0;
145 cia402_sm.current_cmd = CIA402_CMD_ENABLE_OPERATION;
146#endif // SPECIFY_PC_ENVIRONMENT
147
148#if BUILD_LEVEL >= 3
149
150 // NOTICE:
151 // if grid connect is request disable switch delay from CIA402_SM_SWITCH_ON_DISABLED to CIA402_SM_SWITCHED_ON
152 // or a longer judgment time can lead to failure to connect to the grid.
153 cia402_sm.minimum_transit_delay[CIA402_SM_READY_TO_SWITCH_ON] = 0;
154 cia402_sm.minimum_transit_delay[CIA402_SM_SWITCHED_ON] = 0;
155
156#endif // BUILD_LEVEL
157
158 //
159 // init ADC Calibrator
160 //
161 ctl_init_adc_calibrator(&adc_calibrator, 20, 0.707f, CONTROLLER_FREQUENCY);
162
163 if (flag_enable_adc_calibrator)
164 {
165 ctl_enable_adc_calibrator(&adc_calibrator);
166 }
167
168#if 1
169
170 ctl_init_triangle_wave_generator(&tri_wave, CONTROLLER_FREQUENCY, 0.2f, 0.8f, 0.1f);
171
172 //
173 // Ž┬├µšŌę╗ūķ▓╬╩²┤¾Ė┼0.7pu┐¬╩╝š±Ą┤
174 //
175
176 inv_ctrl.pid_idq[phase_d].kp /= 4.0f;

235fast_gt ctl_exec_adc_calibration(void)
236{
237 //
238 // 1. ADC Auto calibrate
239 //
240 if (flag_enable_adc_calibrator)
241 {
242 if (ctl_is_adc_calibrator_cmpt(&adc_calibrator) && ctl_is_adc_calibrator_result_valid(&adc_calibrator))
243 {
244
245 // index_adc_calibrator == 13, for Ibus
246 if (index_adc_calibrator == 13)
247 {
248 // vbus get result
249 idc.bias = idc.bias + ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), idc.gain);
250
251 // move to next position
252 index_adc_calibrator += 1;
253
254 // adc calibrate process done.
255 flag_enable_adc_calibrator = 0;
256
257 // clear INV controller
258 ctl_clear_gfl_inv_with_PLL(&inv_ctrl);
259
260 // ADC Calibrator complete here.
261 //ctl_enable_gfl_inv(&inv_ctrl);
262 }
263
264 // index_adc_calibrator == 12, for Vbus
265 else if (index_adc_calibrator == 12)
266 {
267 // vbus get result
268 //udc.bias = udc.bias + ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), udc.gain);
269
270 // move to next position
271 index_adc_calibrator += 1;
272
273 // clear calibrator
274 ctl_clear_adc_calibrator(&adc_calibrator);
275
276 // enable calibrator to next position
277 ctl_enable_adc_calibrator(&adc_calibrator);
278 }
279
280 // index_adc_calibrator == 11 ~ 9, for Vuvw
281 else if (index_adc_calibrator <= 11 && index_adc_calibrator >= 9)
282 {
283 // vuvw get result
284 uuvw.bias[index_adc_calibrator - 9] =
285 uuvw.bias[index_adc_calibrator - 9] +
286 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), uuvw.gain[index_adc_calibrator - 9]);
287
288 // move to next position
289 index_adc_calibrator += 1;
290
291 // clear calibrator
292 ctl_clear_adc_calibrator(&adc_calibrator);
293
294 // enable calibrator to next position
295 ctl_enable_adc_calibrator(&adc_calibrator);
296 }
297
298 // index_adc_calibrator == 8 ~ 6, for Vabc
299 else if (index_adc_calibrator <= 8 && index_adc_calibrator >= 6)
300 {
301 // vabc get result
302 vabc.bias[index_adc_calibrator - 6] =
303 vabc.bias[index_adc_calibrator - 6] +
304 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), vabc.gain[index_adc_calibrator - 6]);
305
306 // move to next position
307 index_adc_calibrator += 1;
308
309 // clear calibrator
310 ctl_clear_adc_calibrator(&adc_calibrator);
311
312 // enable calibrator to next position
313 ctl_enable_adc_calibrator(&adc_calibrator);
314 }
315
316 // index_adc_calibrator == 5 ~ 3, for Iabc
317 else if (index_adc_calibrator <= 5 && index_adc_calibrator >= 3)
318 {
319
320 // iabc get result
321 iabc.bias[index_adc_calibrator - 3] =
322 iabc.bias[index_adc_calibrator - 3] +
323 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), iabc.gain[index_adc_calibrator - 3]);
324
325 // move to next position
326 index_adc_calibrator += 1;
327
328 // clear calibrator
329 ctl_clear_adc_calibrator(&adc_calibrator);
330
331 // enable calibrator to next position
332 ctl_enable_adc_calibrator(&adc_calibrator);
333 }
334
335 // index_adc_calibrator == 2 ~ 0, for Iuvw
336 else if (index_adc_calibrator <= 2)
337 {
338 // iuvw get result
339 iuvw.bias[index_adc_calibrator] =
340 iuvw.bias[index_adc_calibrator] +
341 ctl_div(ctl_get_adc_calibrator_result(&adc_calibrator), iuvw.gain[index_adc_calibrator]);
342
343 // move to next position
344 index_adc_calibrator += 1;
345
346 // clear calibrator
347 ctl_clear_adc_calibrator(&adc_calibrator);
348
349 // enable calibrator to next position
350 ctl_enable_adc_calibrator(&adc_calibrator);
351 }
352
353 // over-range protection
354 if (index_adc_calibrator > 13)
355 flag_enable_adc_calibrator = 0;
356 }
357
358 // ADC calibrate is not complete
359 return 0;
360 }
361
362 // skip calibrate routine
363 return 1;
