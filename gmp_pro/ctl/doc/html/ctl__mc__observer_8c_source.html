118void ctl_init_acm_smo(ctl_acm_smo_t* smo, const ctl_acm_smo_init_t* init)
119{
120 // Clear all states
121 smo->encif.position = 0;
122 //ctl_vector2_clear(&smo->encif.position);
123 smo->spdif.speed = 0.0f;
124 ctl_vector2_clear(&smo->i_s_est);
125 ctl_vector2_clear(&smo->psi_r_est);
126 ctl_vector2_clear(&smo->z);
127 smo->omega_e_est = 0.0f;
128 smo->theta_est = 0.0f;
129
130 // Initialize controllers
131 ctl_init_pid_Tmode(&smo->pid_pll, init->pll_kp, init->pll_ki, 0, init->f_ctrl);
132 ctl_init_lp_filter(&smo->filter_spd, init->f_ctrl, init->speed_lpf_fc);
133
134 // Store parameters
135 smo->k_slide = init->k_slide;
136 smo->pole_pairs = (ctrl_gt)init->pole_pairs;
137 smo->Ts = 1.0f / init->f_ctrl;
138
139 // Pre-calculate model coefficients for optimization
140 parameter_gt sigma = 1.0f - (init->Lm * init->Lm) / (init->Ls * init->Lr);
141 parameter_gt Tr = init->Lr / init->Rr;
142
143 smo->c1 = float2ctrl(1.0f / (sigma * init->Ls));
144 smo->c2 = float2ctrl((init->Rs / (sigma * init->Ls)) + ((1.0f - sigma) / (sigma * Tr)));
145 smo->c3 = float2ctrl(init->Lm / (sigma * init->Ls * Tr));
146 smo->c4 = float2ctrl(init->Lm / (sigma * init->Ls));
147 smo->c5 = float2ctrl(init->Lm / Tr);
148
149 // Calculate speed scaling factor
150 ctrl_gt base_speed_rad_s = (ctrl_gt)init->speed_base_rpm * CTL_PARAM_CONST_2PI / 60.0f;
151 smo->speed_pu_sf = 1.0f / base_speed_rad_s;
152}

171void ctl_init_acm_mpc(ctl_acm_mpc_controller_t* mpc, const ctl_acm_mpc_init_t* init)
172{
173 mpc->optimal_vector_index = 0;
174 ctl_vector2_clear(&mpc->psi_r_est);
175 mpc->flux_ref_sq = 0;
176 mpc->lambda_flux = float2ctrl(50.0f); // Default weighting factor, should be tuned.
177 mpc->Ts = float2ctrl(1.0f / (ctrl_gt)init->f_ctrl);
178 mpc->pole_pairs = (ctrl_gt)init->pole_pairs;
179
180 // Pre-calculate coefficients for the discrete-time model to optimize the step function.
181 // Based on Euler discretization of the continuous-time ACM model in the stationary frame.
182 parameter_gt sigma = 1.0f - (init->Lm * init->Lm) / (init->Ls * init->Lr);
183 parameter_gt Tr = init->Lr / init->Rr;
184 parameter_gt sigma_ls = sigma * init->Ls;
185
186 mpc->c_iss = float2ctrl(1.0f - mpc->Ts * (init->Rs / sigma_ls + (1.0f - sigma) / (sigma * Tr)));
187 mpc->c_is_u = float2ctrl(mpc->Ts / sigma_ls);
188 mpc->c_is_pr = float2ctrl(mpc->Ts * init->Lm / (sigma_ls * init->Lr * Tr));
189 mpc->c_is_pr_w = float2ctrl(mpc->Ts * init->Lm / (sigma_ls * init->Lr));
190 mpc->c_pr_is = float2ctrl(mpc->Ts * init->Lm / Tr);
191 mpc->c_pr_pr = float2ctrl(1.0f - mpc->Ts / Tr);
192}

209void ctl_init_bldc_hall_estimator(ctl_bldc_hall_estimator_t* est, uint16_t pole_pairs, parameter_gt timer_freq_hz,
210 parameter_gt speed_base_rpm, parameter_gt speed_filter_fc_hz,
211 parameter_gt isr_freq_hz)
212{
213 //ctl_vector2_clear(&est->encif.position);
214 est->encif.position = 0;
215 //ctl_vector2_clear(&est->spdif.speed);
216 est->spdif.speed = 0;
217 est->last_hall_state = 0;
218 est->last_capture_time = 0;
219 est->sector_time_delta = 0;
220 est->estimated_speed_rad_per_tick = 0;
221 est->coarse_angle_pu = 0;
222
223 est->pole_pairs = (ctrl_gt)pole_pairs;
224 est->sector_rad = CTL_PARAM_CONST_PI / 3.0f;
225

227 ctrl_gt wc = 2.0f * CTL_PARAM_CONST_PI * (ctrl_gt)speed_filter_fc_hz;
