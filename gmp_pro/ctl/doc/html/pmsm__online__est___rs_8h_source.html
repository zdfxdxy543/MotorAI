94GMP_STATIC_INLINE parameter_gt ctl_step_online_rs_est(ctl_online_rs_est_t* est, parameter_gt ud, parameter_gt id,
95 parameter_gt iq, parameter_gt omega_e)
96{
97 // --- 1. 计算电流微分项 (did/dt) ---
98 // 使用后向差分近似: d_id/dt ≈ (id[k] - id[k-1]) / Ts
99 parameter_gt did_dt = (id - est->id_prev) / est->Ts;
100
101 // --- 2. 获取所需的电机参数 ---
102 parameter_gt Ld = est->pmsm_params->Ld;
103 parameter_gt Lq = est->pmsm_params->Lq;
104
105 // --- 3. 构建参考模型和可调模型 ---
106 // 参考模型输出 (Reference Model Output): ud_ref = ud - Ld*did/dt + ωe*Lq*iq
107 parameter_gt ud_ref = ud - Ld * did_dt + omega_e * Lq * iq;
108
109 // 可调模型输出 (Adjustable Model Output): ud_adj = Rs_hat * id
110 parameter_gt ud_adj = est->Rs_estimated * id;
111
112 // --- 4. 计算误差 ---
113 parameter_gt error = ud_ref - ud_adj;
114
115 // --- 5. 更新自适应机制 (PI控制器) ---
116 // 为了保证稳定性，通常使用 id 作为误差的权重
117 // 但为简化，在低速或id变化不大的情况下，可直接使用error
118 // 一个折中的方法是使用id的符号或幅值进行限幅
119 if (fabsf(id) > 1e-3) // 仅在d轴电流不为零时更新，避免不稳定
120 {
121 est->Rs_estimated = ctl_step_pid_ser(&est->adapt_pi, error);
122 }
123
124 // --- 6. 更新状态变量 ---
125 est->id_prev = id;
126
127 return est->Rs_estimated;
128}
