123GMP_STATIC_INLINE ctrl_gt ctl_step_repetitive_controller(ctl_repetitive_controller_t* rc, ctrl_gt compensated_error)
124{
125 // 1. Get the past output u_out(k-N) from the circular buffer
126 ctrl_gt past_output = rc->state_buffer[rc->buffer_index];
127
128 // 2. Calculate the new output: u_out(k) = q * u_out(k-N) + u_in(k)
129 rc->output = ctl_mul(past_output, rc->q_filter_coeff) + compensated_error;
130
131 // 3. Saturate the output
132 rc->output = ctl_sat(rc->output, rc->out_max, rc->out_min);
133
134 // 4. Update the buffer with the new output for the next cycle
135 rc->state_buffer[rc->buffer_index] = rc->output;
136
137 // 5. Advance the buffer index for the next sample
138 rc->buffer_index++;
139 if (rc->buffer_index >= rc->period_samples)
140 {
141 rc->buffer_index = 0;
142 }
143
144 return rc->output;
145}
