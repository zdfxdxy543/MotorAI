141 struct

143 // --- 低 8 位 (Byte 0) ---
144
145 // Bit 0: 准备好合闸 (Ready to Switch On)
146 // 1 = 系统已完成预充，无故障，等待 Switch On 命令
147 uint16_t ready_to_switch_on : 1;
148
149 // Bit 1: 已合闸 (Switched On)
150 // 1 = 强电电路已接通 (继电器闭合)
151 uint16_t switched_on : 1;
152
153 // Bit 2: 运行允许 (Operation Enabled)
154 // 1 = PWM 已输出，闭环控制正在运行
155 uint16_t operation_enabled : 1;
156
157 // Bit 3: 故障 (Fault)
158 // 1 = 发生故障
159 uint16_t fault : 1;
160
161 // Bit 4: 电压允许 (Voltage Enabled)
162 // 1 = 直流母线电压/主电源已施加
163 uint16_t voltage_enabled : 1;
164
165 // Bit 5: 快速停机 (Quick Stop)
166 // 注意逻辑反向:
167 // 1 = 正常 (Drive is NOT performing quick stop)
168 // 0 = 正在急停 (Drive is reacting to a Quick Stop request)
169 uint16_t quick_stop : 1;
170
171 // Bit 6: 合闸禁止 (Switch On Disabled)
172 // 1 = 系统处于初始化完成或故障复位后的待机状态，禁止直接合闸
173 uint16_t switch_on_disabled : 1;
174
175 // Bit 7: 警告 (Warning)
176 // 1 = 有警告参数超限，但不需要停机
177 uint16_t warning : 1;
178
179 // --- 高 8 位 (Byte 1) ---
180
181 // Bit 8: 厂商自定义 (Manufacturer specific)
182 uint16_t manufacturer_8 : 1;
183
184 // Bit 9: 远程控制 (Remote)
185 // 1 = 控制权在 CANopen/EtherCAT 总线 (响应 0x6040)
186 // 0 = 本地控制 (忽略总线控制字)
187 uint16_t remote : 1;
188
189 // Bit 10: 目标到达 (Target Reached)
190 // 1 = 轴已停止或到达目标位置/速度
191 // 在 Homing 模式下表示回零完成
192 uint16_t target_reached : 1;
193
194 // Bit 11: 内部限制有效 (Internal limit active)
195 // 1 = 内部电流、速度或位置软限位已被触发
196 uint16_t internal_limit_active : 1;
197
198 // Bits 12-13: 模式相关 (Operation mode specific)
199 // 例如在 CSP 模式下通常为 0，在 Homing 模式下指示状态
200 uint16_t oms_12 : 1;
201 uint16_t oms_13 : 1;
202
203 // Bit 14: 厂商自定义 (这里您定义为回零完成标志)
204 // 1 = Home has completed
205 uint16_t mfg_home_completed : 1;
206
207 // Bit 15: 厂商自定义
208 uint16_t manufacturer_15 : 1;
209
210 } bits;
