140void ctl_mainloop(void)
141{
142 // When the program is reach here, the following things will happen:
143 // 1. software non-block delay 500ms
144 // 2. judge if spll theta error convergence has occurred
145 // 3. then enable system
146
147 if (flag_system_enable)
148 {
149
150 // first time flag
151 // log the first time enable the system
152 if (firsttime_flag == 0)
153 {
154 tick_bias = gmp_base_get_system_tick();
155 firsttime_flag = 1;
156 }
157
158 // a delay of 100ms
159 if ((started_flag == 0) && ((gmp_base_get_system_tick() - tick_bias) > CTRL_STARTUP_DELAY) &&
160 (startup_flag == 0))
161 {
162 startup_flag = 1;
163 }
164
165 // judge if PLL is close to target
166 if ((started_flag == 0) && (startup_flag == 1) && ctl_ready_mainloop())
167 {
168 ctl_enable_output();
169 started_flag = 1;
170 }
171
172#if BUILD_LEVEL >= 4
173
174 if (gmp_base_get_system_tick() - tick_bias > 300)
175 {
176 ctl_enable_mppt_PnO_algo(&mppt);
177 }
178
179#endif // BUILD_LEVEL
180 }
181 // if system is disabled
182 else // (flag_system_enable == 0)
183 {
184 ctl_disable_output();
185
186 // clear all flags
187 firsttime_flag = 0;
188 startup_flag = 0;
189 started_flag = 0;
190 tick_bias = 0;
191 }
192
193 return;
194}
