43GMP_STATIC_INLINE
44void ctl_input_callback(void)
45{
46 // update system tick
47 gmp_step_system_tick();
48
49 // copy ADC data to raw buffer
50 // NOTICE use Result base not adc base.
51 udc_raw = ADC_readResult(MOTOR_VBUS_RESULT_BASE, MOTOR_VBUS);

53 uabc_raw[phase_U] = ADC_readResult(MOTOR_VU_RESULT_BASE, MOTOR_VU);
54 uabc_raw[phase_V] = ADC_readResult(MOTOR_VV_RESULT_BASE, MOTOR_VV);
55 uabc_raw[phase_W] = ADC_readResult(MOTOR_VW_RESULT_BASE, MOTOR_VW);
56
57 iabc_raw[phase_U] = ADC_readResult(MOTOR_IU_RESULT_BASE, MOTOR_IU);
58 iabc_raw[phase_V] = ADC_readResult(MOTOR_IV_RESULT_BASE, MOTOR_IV);
59 iabc_raw[phase_W] = ADC_readResult(MOTOR_IW_RESULT_BASE, MOTOR_IW);
60
61 // invoke ADC p.u. routine
62 ctl_step_tri_ptr_adc_channel(&iabc);
63 ctl_step_tri_ptr_adc_channel(&uabc);
64 ctl_step_ptr_adc_channel(&idc);
65 ctl_step_ptr_adc_channel(&udc);
66
67#ifdef PMSM_CTRL_USING_QEP_ENCODER
68 // Step auto turn pos encoder
69 ctl_step_autoturn_pos_encoder(&pos_enc, EQEP_getPosition(EQEP_Encoder_BASE));
70#else // PMSM_CTRL_USING_QEP_ENCODER
