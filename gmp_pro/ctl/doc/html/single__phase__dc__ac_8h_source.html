225GMP_STATIC_INLINE ctrl_gt ctl_step_sinv(sinv_ctrl_t* sinv)
226{
227 // --- 1. ADC Input Filtering ---
228 ctl_step_lowpass_filter(&sinv->lpf_idc, sinv->adc_idc->value);
229 ctl_step_lowpass_filter(&sinv->lpf_udc, sinv->adc_udc->value);
230 ctl_step_lowpass_filter(&sinv->lpf_il, sinv->adc_il->value);
231 ctl_step_lowpass_filter(&sinv->lpf_ugrid, sinv->adc_ugrid->value);
232 ctl_step_lowpass_filter(&sinv->lpf_igrid, sinv->adc_igrid->value);
233
234 // --- 2. Grid Synchronization (PLL) ---
235 if (sinv->flag_enable_spll)
236 {
237 ctl_step_single_phase_pll(&sinv->spll, ctl_get_lowpass_filter_result(&sinv->lpf_ugrid));
238 }
239
240 // --- 3. AC Measurement ---
241 if (sinv->flag_enable_ac_measurement)
242 {
243#ifdef CTL_SINV_CTRL_ENABLE_SINE_ANALYZER
244 sinv->vg_rms =
245 ctl_step_sine_analyzer(&sinv->ac_voltage_measure, ctl_get_lowpass_filter_result(&sinv->lpf_ugrid));
246 sinv->ig_rms =
247 ctl_step_sine_analyzer(&sinv->ac_current_measure, ctl_get_lowpass_filter_result(&sinv->lpf_igrid));
248#endif
249 }
250
251 if (sinv->flag_enable_system)
252 {
253 // --- 4. Angle Reference Generation ---
254 if (sinv->flag_enable_ramp)
255 {
256 ctl_step_ramp_generator(&sinv->rg);
257 }
258
259 // Select ramp generator as angle source
260 if (sinv->flag_angle_freerun)
261 {
262 sinv->target_phase = ctl_mul(ctl_sin(ctl_get_ramp_generator_output(&sinv->rg)), sinv->pf_set) +
263 ctl_mul(ctl_cos(ctl_get_ramp_generator_output(&sinv->rg)),
264 ctl_sqrt(float2ctrl(1) - ctl_mul(sinv->pf_set, sinv->pf_set)));
265 }
266 // Select SPLL as angle source
267 else
268 {
269 sinv->target_phase = ctl_mul(sinv->spll.phasor.dat[phasor_sin], sinv->pf_set) +
270 ctl_mul(sinv->spll.phasor.dat[phasor_cos],
271 ctl_sqrt(float2ctrl(1) - ctl_mul(sinv->pf_set, sinv->pf_set)));
272 }
273
274 // --- 5. Outer Voltage Loop ---
275 if (sinv->flag_enable_voltage_ctrl)
276 {
277 if (sinv->flag_rectifier_mode)
278 { // Rectifier (AC->DC)
279 sinv->ig_set =
280 -ctl_step_pid_ser(&sinv->voltage_pid, sinv->v_set - ctl_get_lowpass_filter_result(&sinv->lpf_udc));
281 }
282 else
283 { // Inverter (DC->AC)
284 sinv->ig_set = ctl_step_pid_ser(&sinv->voltage_pid, sinv->v_set - sinv->vg_rms);
285 }
286 }
287
288 // --- 6. Inner Current Loop ---
289 if (sinv->flag_enable_current_ctrl)
290 {
291 sinv->ig_ref = ctl_mul(sinv->target_phase, sinv->ig_set);
292 ctrl_gt current_error = sinv->ig_ref - ctl_get_lowpass_filter_result(&sinv->lpf_igrid);
293
294 sinv->modulation = ctl_step_qpr_controller(&sinv->sinv_qpr_base, current_error);
295
296 if (sinv->flag_enable_harmonic_ctrl)
297 {
298 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_3, -current_error);
299 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_5, -current_error);
300 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_7, -current_error);
301 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_9, -current_error);
302 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_11, -current_error);
303 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_13, -current_error);
304 sinv->modulation += ctl_step_qr_controller(&sinv->sinv_qr_15, -current_error);
305 }
306 }
307 else
308 { // Open-loop current
309 sinv->ig_ref = ctl_mul(sinv->target_phase, sinv->ig_set);
310 sinv->modulation = sinv->ig_ref;
311 }
312
313 // --- 7. Modulation Stage ---
314#ifdef CTL_SINV_CTRL_UNIPOLAR_MODULATION
315 // Unipolar SPWM
316 sinv->sinv_pwm_pu[PHASE_L] = sinv->modulation;
317 sinv->sinv_pwm_pu[PHASE_N] = -sinv->modulation;
318#else
319 // Bipolar SPWM
320 sinv->sinv_pwm_pu[PHASE_L] = ctl_div2(ctl_add(sinv->modulation, float2ctrl(1.0f)));
321 sinv->sinv_pwm_pu[PHASE_N] = ctl_div2(ctl_sub(float2ctrl(1.0f), sinv->modulation));
322#endif
323 }
324 else
325 {
326 // Disable output if system is not enabled
327 sinv->sinv_pwm_pu[PHASE_L] = 0;
328 sinv->sinv_pwm_pu[PHASE_N] = 0;
329 }
330
331 return sinv->modulation;
332}
