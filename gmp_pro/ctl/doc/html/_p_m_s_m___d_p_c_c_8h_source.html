121GMP_STATIC_INLINE void ctl_step_dpcc(ctl_dpcc_controller_t* dpcc, const ctl_vector2_t* idq_ref,
122 const ctl_vector2_t* idq_fbk, ctrl_gt omega_e)
123{
124 ctrl_gt id_k = idq_fbk->dat[0];
125 ctrl_gt iq_k = idq_fbk->dat[1];
126
127 // --- 1. One-step Current Prediction ---
128 // Predict the current at the end of this step, i(k+1), using the voltage applied in the previous step, u(k-1).
129 // This compensates for the one-step digital control delay.
130 ctrl_gt id_pred = id_k + ctl_mul(dpcc->ts_over_ld, dpcc->udq_last.dat[0] - ctl_mul(dpcc->rs, id_k) +
131 ctl_mul(omega_e, ctl_mul(dpcc->lq, iq_k)));
132
133 ctrl_gt iq_pred = iq_k + ctl_mul(dpcc->ts_over_lq, dpcc->udq_last.dat[1] - ctl_mul(dpcc->rs, iq_k) -
134 ctl_mul(omega_e, (ctl_mul(dpcc->ld, id_k) + dpcc->psi_f)));
135
136 dpcc->idq_predicted.dat[0] = id_pred;
137 dpcc->idq_predicted.dat[1] = iq_pred;
138
139 // --- 2. Dead-beat Voltage Calculation ---
140 // Calculate the voltage u(k) required to make the current at the next step, i(k+2), equal to the reference i_ref(k+1).
141 // We assume i_ref(k+1) is the same as the current i_ref(k).
142 dpcc->udq_out.dat[0] = ctl_mul(dpcc->ld_over_ts, (idq_ref->dat[0] - id_pred)) + ctl_mul(dpcc->rs, id_pred) -
143 ctl_mul(omega_e, ctl_mul(dpcc->lq, iq_pred));
144
145 dpcc->udq_out.dat[1] = ctl_mul(dpcc->lq_over_ts, (idq_ref->dat[1] - iq_pred)) + ctl_mul(dpcc->rs, iq_pred) +
146 ctl_mul(omega_e, (ctl_mul(dpcc->ld, id_pred) + dpcc->psi_f));
147
148 // --- 3. Store the calculated voltage for the next prediction step ---
149 ctl_vector2_copy(&dpcc->udq_last, &dpcc->udq_out);
150}
