107GMP_STATIC_INLINE ctrl_gt ctl_step_sine_analyzer(sine_analyzer_t* sine, ctrl_gt wave_input)
108{
109 sine->wave_norm = ctl_abs(wave_input);
110 sine->curr_sign = (wave_input > sine->threshold) ? 1 : 0;
111 sine->sample_index++;
112 sine->sumup += sine->wave_norm;
113 sine->sumup_sqr += (sine->wave_norm * sine->wave_norm);
114 sine->zero_cross = 0;
115
116 // Check for a positive-going zero-crossing, which marks the end of a full cycle.
117 if (sine->prev_sign != sine->curr_sign && sine->curr_sign == 1)
118 {
119 // Check if the cycle duration is within the expected valid range.
120 if (sine->sample_index > sine->sample_min)
121 {
122 sine->zero_cross = 1;
123
124 // Pre-calculate inverse values for efficiency.
125 sine->inv_index = float2ctrl(1.0f / sine->sample_index);
126 sine->inv_index_sqrt = ctl_sqrt(sine->inv_index);
127
128 // Calculate average, RMS, and frequency.
129 sine->wave_avg = ctl_mul(sine->inv_index, sine->sumup);
130 sine->wave_rms = ctl_mul(ctl_sqrt(sine->sumup_sqr), sine->inv_index_sqrt);
131 sine->ac_freq_pu = ctl_mul(sine->freq_sf, sine->inv_index);
132
133 // Reset accumulators for the next cycle.
134 sine->sample_index = 0;
135 sine->sumup = 0;
136 sine->sumup_sqr = 0;
137 }
138 // Handle cases where a zero-crossing happens too quickly (likely due to noise/jitter).
139 else
140 {
141 if (sine->jitter_counter <= 30)
142 {
143 sine->jitter_counter += 1;
144 }
145 sine->sample_index = 0;
146 }
147 }
148
149 // Protection: If no valid wave is detected for too long, reset the analyzer.
150 if (sine->sample_index > sine->sample_max || sine->jitter_counter > 20)
151 {
152 ctl_sine_clear_analyzer(sine);
153 }
154
155 // Update the sign for the next iteration.
156 sine->prev_sign = sine->curr_sign;
157
158 return sine->wave_rms;
159}
