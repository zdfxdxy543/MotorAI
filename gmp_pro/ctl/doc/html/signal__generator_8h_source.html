200GMP_STATIC_INLINE ctrl_gt ctl_step_square_wave_generator(ctl_square_wave_generator_t* sq)
201{
202 // Accumulate phase
203 sq->phase += sq->phase_step;
204
205 // Check for phase wrap-around
206 if (sq->phase >= (2.0f * CTL_PARAM_CONST_PI))
207 {
208 sq->phase -= (2.0f * CTL_PARAM_CONST_PI);
209 }
210
211 // Determine output based on phase
212 // Output is high for the first half of the cycle (0 to PI)
213 // and low for the second half (PI to 2*PI).
214 if (sq->phase < CTL_PARAM_CONST_PI)
215 {
216 sq->output = sq->high_level;
217 }
218 else
219 {
220 sq->output = sq->low_level;
221 }
222
223 return sq->output;
224}

258 // Update output with the current slope
259 tri->output += tri->slope;

261 // Check for peak reversal

263 {
264 tri->output = tri->pos_peak; // Clamp to peak

269 tri->output = tri->neg_peak; // Clamp to peak
270 tri->slope = -tri->slope; // Reverse direction
