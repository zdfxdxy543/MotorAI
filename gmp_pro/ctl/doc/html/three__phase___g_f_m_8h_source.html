148// init gfm init parameters by GFL init struct.
149void ctl_auto_tuning_gfm_droop_ctrl(gfm_droop_init_t* gfm_init, gfl_inv_ctrl_init_t* gfl)
150{
151 gmp_base_assert(gfm_init);
152 gmp_base_assert(gfl_init);
153
154 // 1. Copy Basic System Params from GFL
155 gfm_init->fs = gfl_init->fs;
156 gfm_init->freq_base = gfl_init->freq_base;
157 gfm_init->v_base = gfl_init->v_base;
158 gfm_init->i_base = gfl_init->i_base;
159
160 // Calculate S_base (VA)
161 // S_base = V_base * I_base
162 // Let's assume P.U. consistency:
163 gfm_init->s_base = gfl_init->v_base * gfl_init->i_base;
164
165 // 2. Filter Capacitor
166 gfm_init->cf_capacitance = gfl_init->grid_filter_C;
167
168 // 3. Tuning Voltage Loop
169 // Bandwidth: Typically 1/5 to 1/10 of Current Loop Bandwidth
170 gfm_init->voltage_loop_bw = gfl_init->current_loop_bw / 5.0f;
171
172 // Kp Calculation: C * 2*pi*BW
173 // Physical Kp = C * 2*pi*BW
174 // P.U. Kp = Physical_Kp * (V_base / I_base)
175 // (Input is Voltage err, Output is Current ref)
176 parameter_gt kp_phys = gfm_init->cf_capacitance * CTL_PARAM_CONST_2PI * gfm_init->voltage_loop_bw;
177 gfm_init->kp_voltage = kp_phys * (gfm_init->v_base / gfm_init->i_base);
178
179 // Ki Calculation: Place zero to cancel pole or set suitable stiffness
180 // Typically zero at BW/10 or BW/5
181 gfm_init->ki_voltage = gfm_init->kp_voltage * CTL_PARAM_CONST_2PI * (gfm_init->voltage_loop_bw / 5.0f);
182
183 // 4. Default Droop Settings
184 gfm_init->droop_p_percent = 0.01f; // 1% freq drop at rated load
185 gfm_init->droop_q_percent = 0.05f; // 5% volt drop at rated load
186 gfm_init->power_lpf_fc = 2.0f; // 2Hz Power Filter (Simulated Inertia)
187
188 // 5. Default Virtual Impedance (Low Voltage Microgrid usually Resistive)
189 gfm_init->rv_virtual = 0.0f;
190 gfm_init->lv_virtual = 0.0f;
191
192 gfm_init->current_limit = 1.2f; // 1.2 p.u. current limit
193}

195void ctl_init_gfm_droop(gfm_droop_ctrl_t* gfm, const gfm_droop_init_t* init)
196{
197 gmp_base_assert(gfm);
198 gmp_base_assert(init);
199
200 // Set Setpoints Defaults
201 gfm->f_nom = init->freq_base; // 50Hz
202 gfm->v_nom = 1.0f; // 1.0 p.u. voltage
203 gfm->pq_set.dat[0] = 0.0f; // 0 Power setpoint
204 gfm->pq_set.dat[1] = 0.0f;
205
206 gfm->Ts = 1.0f / init->fs;
207 gfm->omega_nom = CTL_PARAM_CONST_2PI * init->freq_base;
208
209 // 1. Calculate Droop Coeffs
210 // Kp_droop = (Delta_f_max) / S_base
211 // Delta_f_max = f_nom * droop_percent
212 // If working in P.U., S_base = 1.0.
213 // Let's assume P/Q inputs to struct are in P.U.
214 gfm->coef_droop[0] = (init->droop_p_percent);
215 gfm->coef_droop[1] = (init->droop_q_percent);
216
217 // 2. Virtual Impedance Coeffs
218 // V_drop = I * Z.
219 // Need to convert physical Ohms/Henrys to P.U. impedance?
220 // Or if I is P.U., Z should be P.U.
221 // Z_base = V_base / I_base.
222 parameter_gt z_base = init->v_base / init->i_base;
223 gfm->coef_vir_res = init->rv_virtual / z_base;
224 gfm->coef_vir_ind = (init->lv_virtual * gfm->omega_nom) / z_base;
225
226 // 3. Feedforward Coeff
227 // omega * C * V_base / I_base (to convert V p.u. to I p.u.)
228 // V_pu * V_base = V_volts. Current = w * C * V_volts. I_pu = Current / I_base.
229 // Coeff = w * C * V_base / I_base = w * C * Z_base.
230 gfm->coef_vdq_ff = gfm->omega_nom * init->cf_capacitance * z_base;
231
232 // 4. Init Controllers
233 ctl_init_filter_iir1_lpf(&gfm->lpf_pq[0], init->fs, init->power_lpf_fc);
234 ctl_init_filter_iir1_lpf(&gfm->lpf_pq[1], init->fs, init->power_lpf_fc);
235
236 ctl_init_pid(&gfm->pid_vdq[phase_d], init->kp_voltage, init->ki_voltage, 0, init->fs);
237 ctl_init_pid(&gfm->pid_vdq[phase_q], init->kp_voltage, init->ki_voltage, 0, init->fs);
238
239 ctl_init_saturation(&gfm->sat_id, -init->current_limit, init->current_limit);
240 ctl_init_saturation(&gfm->sat_iq, -init->current_limit, init->current_limit);
241
242 // Clear States
243 ctl_clear_gfm_droop_ctrl(gfm);
244
245 // Flags
246 gfm->flag_enable_system = 0;
247 gfm->flag_enable_droop = 1;
248 gfm->flag_enable_voltage_loop = 1;
249 gfm->flag_enable_voltage_ff = 1;
250 gfm->flag_enable_virtual_imp = 0; // Default off
251}

256GMP_STATIC_INLINE void ctl_step_gfm_droop(gfm_droop_ctrl_t* gfm)
257{
258 // Assert inputs
259 gmp_base_assert(gfm->vdq_meas);
260 gmp_base_assert(gfm->idq_meas);
261 gmp_base_assert(gfm->pll_angle);
262
263 // --- 1. Power Calculation ---
264 // P = 1.5 * (Vd*Id + Vq*Iq)
265 // Q = 1.5 * (Vq*Id - Vd*Iq)
266 // Assuming Amplitude Invariant transformation coefficient inside the calc or pre-scaled inputs.
267 // Here we use standard P.U. or SI calculation.
268 // If inputs are P.U., 1.5 factor is usually absorbed in base definition or not needed if P_base = V_base*I_base.
269 // Let's assume standard math:
270 ctrl_gt vd = gfm->vdq_meas->dat[phase_d];
271 ctrl_gt vq = gfm->vdq_meas->dat[phase_q];
272 ctrl_gt id = gfm->idq_meas->dat[phase_d];
273 ctrl_gt iq = gfm->idq_meas->dat[phase_q];
274
275 // Simple Power Calc (Adjust 1.5 factor based on your Park Transform convention)
276 // Using 1.0 here assuming P.U. system consistency.
277 gfm->pq_inst.dat[0] = ctl_mul(1.5f, ctl_mul(vd, id) + ctl_mul(vq, iq));
278 gfm->pq_inst.dat[1] = ctl_mul(1.5f, ctl_mul(vq, id) - ctl_mul(vd, iq));
279
280 // Filter Power
281 gfm->pq_avg.dat[0] = ctl_step_filter_iir1(&gfm->lpf_pq[0], gfm->pq_inst.dat[0]);
282 gfm->pq_avg.dat[1] = ctl_step_filter_iir1(&gfm->lpf_pq[1], gfm->pq_inst.dat[1]);
283
284 // --- 2. Droop Control ---
285 if (gfm->flag_enable_droop)
286 {
287 // f = f0 - kp * (P - P0)
288 gfm->droop_freq_ref = gfm->f_nom - ctl_mul(gfm->coef_droop[0], (gfm->pq_avg.dat[0] - gfm->pq_set.dat[0]));
289
290 // V = V0 - kq * (Q - Q0)
291 gfm->droop_v_mag_ref = gfm->v_nom - ctl_mul(gfm->coef_droop[1], (gfm->pq_avg.dat[1] - gfm->pq_set.dat[1]));
292 }
293 else
294 {
295 gfm->droop_freq_ref = gfm->f_nom;
296 gfm->droop_v_mag_ref = gfm->v_nom;
297 }
298
299 // --- 3. Angle Generation & Pre-sync Logic ---
300 if (gfm->flag_enable_system)
301 {
302 // GFM Mode: Integrate frequency
303 ctrl_gt omega = CTL_PARAM_CONST_2PI * gfm->droop_freq_ref;
304 gfm->theta_out += omega * gfm->Ts;
305
306 // Wrap Angle
307 if (gfm->theta_out > CTL_PARAM_CONST_2PI)
308 gfm->theta_out -= CTL_PARAM_CONST_2PI;
309 else if (gfm->theta_out < 0.0f)
310 gfm->theta_out += CTL_PARAM_CONST_2PI;
311 }
312 else
313 {
314 // Pre-Sync / Standby Mode: Track PLL
315 // Reset Integrator state to match Grid
316 gfm->theta_out = *gfm->pll_angle;
317
318 // Optional: Pre-load P/Q filters or PID integrators to avoid bumps if needed
319 // For simplest voltage start, we just track angle.
320 }
321
322 // Update Phasor (Shared with Park Transforms)
323 ctl_set_phasor_via_angle(gfm->theta_out, &gfm->phasor_out);
324
325 // --- 4. Virtual Impedance ---
326 // V_virt_d = R*id - wL*iq
327 // V_virt_q = R*iq + wL*id
328 if (gfm->flag_enable_virtual_imp)
329 {
330 gfm->vdq_vir_imp.dat[phase_d] = ctl_mul(gfm->coef_vir_res, id) - ctl_mul(gfm->coef_vir_ind, iq);
331 gfm->vdq_vir_imp.dat[phase_q] = ctl_mul(gfm->coef_vir_res, iq) + ctl_mul(gfm->coef_vir_ind, id);
332 }
333 else
334 {
335 gfm->vdq_vir_imp.dat[phase_d] = 0.0f;
336 gfm->vdq_vir_imp.dat[phase_q] = 0.0f;
337 }
338
339 // Calculate Reference Voltage (aligned to D-axis)
340 // Vd_ref = V_droop - V_vir_d
341 // Vq_ref = 0 - V_vir_q
342 gfm->vdq_virt_ref.dat[phase_d] = gfm->droop_v_mag_ref - gfm->vdq_vir_imp.dat[phase_d];
343 gfm->vdq_virt_ref.dat[phase_q] = -gfm->vdq_vir_imp.dat[phase_q];
344
345 // --- 5. Voltage Control Loop ---
346 if (gfm->flag_enable_voltage_loop)
347 {
348 // Error
349 ctrl_gt err_vd = gfm->vdq_virt_ref.dat[phase_d] - vd;
350 ctrl_gt err_vq = gfm->vdq_virt_ref.dat[phase_q] - vq;
351
352 // PID Calculation
353 ctrl_gt id_ref = ctl_step_pid_ser(&gfm->pid_vdq[phase_d], err_vd);
354 ctrl_gt iq_ref = ctl_step_pid_ser(&gfm->pid_vdq[phase_q], err_vq);
355
356 // Feedforward Decoupling
357 // Id_ff = - w*C*Vq + Id_load (Measured Id is load current)
358 // Iq_ff = + w*C*Vd + Iq_load
359 if (gfm->flag_enable_voltage_ff)
360 {
361 gfm->vdq_ff_decouple.dat[phase_d] = -ctl_mul(gfm->coef_vdq_ff, vq); // -wC*vq
362 gfm->vdq_ff_decouple.dat[phase_q] = ctl_mul(gfm->coef_vdq_ff, vd); // +wC*vd
363
364 // Add decoupling + Feedforward Load Current (id, iq)
365 id_ref += gfm->vdq_ff_decouple.dat[phase_d] + id;
366 iq_ref += gfm->vdq_ff_decouple.dat[phase_q] + iq;
367 }
368
369 // Saturation (Current Limit)
370 gfm->idq_ref_out.dat[phase_d] = ctl_step_saturation(&gfm->sat_id, id_ref);
371 gfm->idq_ref_out.dat[phase_q] = ctl_step_saturation(&gfm->sat_iq, iq_ref);
372 }
373 else
374 {
375 // Loop Disabled: Output 0 or hold
376 gfm->idq_ref_out.dat[phase_d] = 0.0f;
377 gfm->idq_ref_out.dat[phase_q] = 0.0f;
378 }
379}

385GMP_STATIC_INLINE fast_gt ctl_if_gfm_voltage_ready(gfm_droop_ctrl_t* gfm)
386{
387 // In Pre-Sync (System disabled), Theta is tracking PLL.
388 // We just need to check if the Grid Voltage Amplitude matches our Target V_nom.
389
390 // Calculate Grid Voltage Magnitude
391 ctrl_gt vd = gfm->vdq_meas->dat[phase_d];
392 ctrl_gt vq = gfm->vdq_meas->dat[phase_q];
393 ctrl_gt v_mag_meas = ctl_sqrt(ctl_mul(vd, vd) + ctl_mul(vq, vq));
394
395 // Check difference
396 ctrl_gt err = ctl_abs(v_mag_meas - gfm->v_nom);
397
398 // Threshold: e.g., 5% of V_nom.
399 // This threshold could be a configurable parameter, hardcoded here for simplicity or added to struct.
400 ctrl_gt threshold = gfm->v_nom * 0.05f;
401
402 return (err < threshold) ? 1 : 0;
403}
