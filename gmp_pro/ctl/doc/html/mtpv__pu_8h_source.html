85GMP_STATIC_INLINE void ctl_step_mtpv_pu(ctl_mtpv_pu_controller_t* mtpv, ctrl_gt iq_cmd_pu, ctrl_gt u_max_pu,
86 ctrl_gt omega_e_pu)
87{
88 ctrl_gt we_sq = omega_e_pu * omega_e_pu;
89 ctrl_gt u_max_sq = u_max_pu * u_max_pu;
90
91 // --- Calculate terms for the quadratic equation of Id: A*id^2 + B*id + C = 0 ---
92 ctrl_gt A = we_sq * mtpv->ld_sq_pu + mtpv->rs_sq_pu;
93 ctrl_gt B = 2.0f * we_sq * mtpv->ld_pu * mtpv->psi_f_pu;
94 ctrl_gt C = we_sq * mtpv->psi_f_pu * mtpv->psi_f_pu + we_sq * mtpv->lq_sq_pu * iq_cmd_pu * iq_cmd_pu +
95 mtpv->rs_sq_pu * iq_cmd_pu * iq_cmd_pu + 2.0f * mtpv->rs_pu * omega_e_pu * mtpv->psi_f_pu * iq_cmd_pu -
96 u_max_sq;
97
98 // --- Solve the quadratic equation for Id ---
99 ctrl_gt discriminant = B * B - 4.0f * A * C;
100
101 if (discriminant >= 0.0f)
102 {
103 // A valid solution exists. We choose the negative Id for field weakening.
104 // The solution with the smaller magnitude (closer to zero) is typically desired.
105 mtpv->id_ref_pu = (-B + sqrtf(discriminant)) / (2.0f * A);
106 }
107 else
108 {
109 // No real solution, indicates extreme condition where the voltage ellipse
110 // cannot be reached even with maximum field weakening.
111 // Clamp Id to the characteristic current, which is the theoretical maximum field-weakening current.
112 mtpv->id_ref_pu = -mtpv->psi_f_pu / mtpv->ld_pu;
113 }
114
115 // The q-axis current is the original command from the speed loop.
116 // In a full implementation, Iq might also be adjusted to stay within the overall current limit circle.
117 mtpv->iq_ref_pu = iq_cmd_pu;
118}
