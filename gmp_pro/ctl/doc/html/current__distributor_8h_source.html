91GMP_STATIC_INLINE void ctl_step_current_distributor(ctl_current_distributor_t* dist, ctrl_gt im_cmd)
92{
93 ctrl_gt alpha;
94 ctrl_gt im_mag = fabsf(im_cmd);
95
96 // 1. Determine the current angle 'alpha' based on the selected mode.
97 if (dist->mode == DIST_MODE_LUT_LINEAR)
98 {
99 // Use the generic 1D interpolation function from surf_search.h
100 alpha = ctl_step_interpolate_lut1d(&dist->im_axis_lut, dist->alpha_values, im_mag);
101 }
102 else // DIST_MODE_CONST_ALPHA
103 {
104 alpha = dist->const_alpha;
105 }
106
107 // 2. Calculate Id and Iq based on the total current magnitude and the angle alpha.
108 ctrl_gt id_mag = im_mag * cosf(alpha);
109 ctrl_gt iq_mag = im_mag * sinf(alpha);
110
111 // 3. Apply the original sign of the command to the torque-producing component (Iq).
112 // The sign of Id is determined by the LUT (typically negative for MTPA).
113 dist->id_ref = id_mag;
114 dist->iq_ref = (im_cmd >= 0.0f) ? iq_mag : -iq_mag;
115}
