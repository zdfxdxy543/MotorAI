116void ctl_init_foldback_protection(std_foldback_protection_t* obj, parameter_gt voltage_f_cut,
117 parameter_gt current_f_cut, parameter_gt v_rated, parameter_gt i_max,
118 parameter_gt i_short, parameter_gt fs)

125 obj->current_max = i_max;
126 obj->current_short_circuit = i_short;
127
128 // The initial output should be the maximum allowed current
129 obj->current_limit_output = i_max;

159static void _ctl_init_biquad_differentiator(ctl_biquad_filter_t* obj, parameter_gt fs, parameter_gt L, parameter_gt fc)
160{
161 parameter_gt wc = CTL_PARAM_CONST_2PI * fc;
162 parameter_gt K = L * wc; // Gain factor
163
164 // Tustin Transform: s = 2*fs * (1-z^-1)/(1+z^-1)
165 // H(z) = K * [2fs(1-z^-1)/(1+z^-1)] / [2fs(1-z^-1)/(1+z^-1) + wc]
166 // Simplifying...
167
168 parameter_gt k_tustin = 2.0f * fs;
169 parameter_gt D0 = k_tustin + wc;
170
171 // Coefficients for standard Direct Form I Biquad
172 // y[n] = b0*x[n] + b1*x[n-1] + ... - a1*y[n-1] ...
173
174 obj->b[0] = (K * k_tustin) / D0;
175 obj->b[1] = -(K * k_tustin) / D0; // Note the negative sign
176 obj->b[2] = 0.0f;
177
178 // Denominator a1, a2
179 // Denom(z) = (k + wc) + (wc - k)z^-1
180 // a1 = (wc - k) / (k + wc)
181 obj->a[0] = (wc - k_tustin) / D0;
182 obj->a[1] = 0.0f;
183
184 ctl_clear_biquad_filter(obj);
185}

187void ctl_init_vir_imp(vir_imp_t* imp, const parameter_gt R_vir, const parameter_gt L_vir, parameter_gt fs)
188{
189 gmp_base_assert(imp);
190
191 // 1. Resistive Gain
192 imp->gain_R = R_vir;
193
194 // 2. Inductive Differentiator
195 if (fabsf(L_vir) > 1e-12f) // If L is non-zero
196 {
197 _ctl_init_biquad_differentiator(&imp->diff_filter, fs, L_vir, fs / 10);
198 }
199 else
200 {
201 // Zero inductance -> Pass 0
202 // Set all coeffs to 0
203 ctl_clear_biquad_filter(&imp->diff_filter);
204 imp->diff_filter.b[0] = 0;
205 imp->diff_filter.b[1] = 0;
206 imp->diff_filter.b[2] = 0;
207 imp->diff_filter.a[0] = 0;
208 imp->diff_filter.a[1] = 0;
209 }
210
211 imp->out = 0.0f;
