65GMP_STATIC_INLINE int32_t ctl_search_lut1d_index(const ctl_lut1d_t* lut, ctrl_gt target)
66{
67 // Handle out-of-bounds cases
68 if (target < lut->axis[0])
69 {
70 return -1;
71 }
72 if (target >= lut->axis[lut->size - 1])
73 {
74 return lut->size - 2;
75 }
76
77 // Binary search
78 uint32_t left = 0;
79 uint32_t right = lut->size - 1;
80 while (left <= right)
81 {
82 uint32_t mid = left + (right - left) / 2;
83 if (lut->axis[mid] > target)
84 {
85 right = mid - 1;
86 }
87 else
88 {
89 left = mid + 1;
90 }
91 }
92 return right;
93}

102GMP_STATIC_INLINE ctrl_gt ctl_step_interpolate_lut1d(const ctl_lut1d_t* lut, const ctrl_gt* values, ctrl_gt target)
103{
104 int32_t index = ctl_search_lut1d_index(lut, target);
105
106 // Handle boundary conditions
107 if (index < 0)
108 {
109 return values[0]; // Target is below the range
110 }
111 if ((uint32_t)index >= lut->size - 1)
112 {
113 return values[lut->size - 1]; // Target is above the range
114 }
115
116 // Linear interpolation
117 ctrl_gt x0 = lut->axis[index];
118 ctrl_gt y0 = values[index];
119 ctrl_gt x1 = lut->axis[index + 1];
120 ctrl_gt y1 = values[index + 1];
121
122 ctrl_gt weight = ctl_div(target - x0, x1 - x0);
123 return y0 + ctl_mul(weight, (y1 - y0));
124}

159GMP_STATIC_INLINE ctrl_gt ctl_step_interpolate_lut2d(const ctl_lut2d_t* lut, ctrl_gt target1, ctrl_gt target2)
160{
161 // Find lower-bound indices for both axes
162 int32_t idx1 = ctl_search_lut1d_index(&lut->dim1_axis, target1);
163 int32_t idx2 = ctl_search_lut1d_index(&lut->dim2_axis, target2);
164
165 // Clamp indices to valid range
166 idx1 = (idx1 < 0) ? 0 : idx1;
167 idx2 = (idx2 < 0) ? 0 : idx2;
168 idx1 = (idx1 > (int32_t)lut->dim1_axis.size - 2) ? (lut->dim1_axis.size - 2) : idx1;
169 idx2 = (idx2 > (int32_t)lut->dim2_axis.size - 2) ? (lut->dim2_axis.size - 2) : idx2;
170
171 // Get corner points of the interpolation cell
172 ctrl_gt p00 = lut->surface[idx1][idx2];
173 ctrl_gt p10 = lut->surface[idx1 + 1][idx2];
174 ctrl_gt p01 = lut->surface[idx1][idx2 + 1];
175 ctrl_gt p11 = lut->surface[idx1 + 1][idx2 + 1];
176
177 // Calculate interpolation weights
178 ctrl_gt w1 =
179 ctl_div(target1 - lut->dim1_axis.axis[idx1], lut->dim1_axis.axis[idx1 + 1] - lut->dim1_axis.axis[idx1]);
180 ctrl_gt w2 =
181 ctl_div(target2 - lut->dim2_axis.axis[idx2], lut->dim2_axis.axis[idx2 + 1] - lut->dim2_axis.axis[idx2]);
182
183 // Bilinear interpolation
184 ctrl_gt r1 = p00 + ctl_mul(w1, p10 - p00);
185 ctrl_gt r2 = p01 + ctl_mul(w1, p11 - p01);
186 return r1 + ctl_mul(w2, r2 - r1);
187}

230GMP_STATIC_INLINE ctrl_gt ctl_step_interpolate_uniform_lut2d(const ctl_uniform_lut2d_t* lut, ctrl_gt x, ctrl_gt y)
231{
232 // Calculate floating point indices
233 ctrl_gt x_fidx = ctl_mul(x - lut->x_min, lut->x_step_inv);
234 ctrl_gt y_fidx = ctl_mul(y - lut->y_min, lut->y_step_inv);
235
236 // Clamp indices
237 x_fidx = ctl_sat(x_fidx, (ctrl_gt)(lut->x_size - 1), 0);
238 y_fidx = ctl_sat(y_fidx, (ctrl_gt)(lut->y_size - 1), 0);
239
240 // Get integer indices (lower bound)
241 uint32_t x_idx = (uint32_t)x_fidx;
242 uint32_t y_idx = (uint32_t)y_fidx;
243
244 // Get interpolation weights (fractional part)
245 ctrl_gt wx = x_fidx - (ctrl_gt)x_idx;
246 ctrl_gt wy = y_fidx - (ctrl_gt)y_idx;
247
248 // Get corner points
249 ctrl_gt p00 = lut->surface[x_idx][y_idx];
250 ctrl_gt p10 = lut->surface[x_idx + 1][y_idx];
251 ctrl_gt p01 = lut->surface[x_idx][y_idx + 1];
252 ctrl_gt p11 = lut->surface[x_idx + 1][y_idx + 1];
253
254 // Bilinear interpolation
255 ctrl_gt r1 = p00 + ctl_mul(wx, p10 - p00);
256 ctrl_gt r2 = p01 + ctl_mul(wx, p11 - p01);
257 return r1 + ctl_mul(wy, r2 - r1);
