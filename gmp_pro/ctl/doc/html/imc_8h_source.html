134GMP_STATIC_INLINE ctrl_gt ctl_step_imc(ctl_imc_controller_t* imc, ctrl_gt r, ctrl_gt y_p)
135{
136 // 1. Get the delayed input for the plant model from the buffer
137 uint16_t read_idx =
138 (imc->delay_buffer_idx + IMC_MAX_DEAD_TIME_SAMPLES - imc->dead_time_samples) % IMC_MAX_DEAD_TIME_SAMPLES;
139 ctrl_gt u_delayed = imc->u_delay_buffer[read_idx];
140
141 // 2. Update the internal plant model
142 // y_m(k) = a_p_d * y_m(k-1) + b_p_d * u(k-d-1)
143 imc->y_m = imc->a_p_d * imc->y_m + imc->b_p_d * u_delayed;
144
145 // 3. Calculate the model error (disturbance estimate)
146 ctrl_gt model_error = r - (y_p - imc->y_m);
147
148 // 4. Pass the error through the Q controller
149 // u_q(k) = a_q_d*u_q(k-1) + b0_q_d*e(k) + b1_q_d*e(k-1)
150 ctrl_gt u_q = imc->a_q_d * imc->q_out_1 + imc->b0_q_d * model_error + imc->b1_q_d * imc->q_in_1;
151
152 // Update Q controller state for next iteration
153 imc->q_in_1 = model_error;
154 imc->q_out_1 = u_q;
155
156 // The output of Q is the new control signal
157 imc->u_out = u_q;
158
159 // 5. Store the new control output in the delay buffer
160 imc->u_delay_buffer[imc->delay_buffer_idx] = imc->u_out;
161 imc->delay_buffer_idx = (imc->delay_buffer_idx + 1) % IMC_MAX_DEAD_TIME_SAMPLES;
162
163 return imc->u_out;
164}
