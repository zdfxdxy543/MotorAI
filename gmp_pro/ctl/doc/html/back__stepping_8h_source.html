114GMP_STATIC_INLINE ctrl_gt ctl_step_backstepping(ctl_backstepping_controller_t* bc, ctrl_gt y_ref, ctrl_gt y_ref_dot,
115 ctrl_gt y_actual, ctrl_gt disturbance_est)
116{
117 // 1. Calculate the tracking error (z1)
118 ctrl_gt error = y_ref - y_actual;
119
120 // 2. Calculate the control law based on the backstepping design
121 // u = (tau_p * (y_ref_dot + k1*error) + y_actual - disturbance_est) / K_p
122
123 // Term 1: Dynamics and tracking term
124 ctrl_gt term1 = ctl_mul(bc->tau_p, (y_ref_dot + ctl_mul(bc->k1, error)));
125
126 // Term 2: State feedback and disturbance rejection
127 ctrl_gt term2 = y_actual - disturbance_est;
128
129 // 3. Sum terms and divide by plant gain
130 if (fabsf(bc->K_p) > 1e-9f) // Avoid division by zero
131 {
132 bc->u_out = (term1 + term2) / bc->K_p;
133 }
134 else
135 {
136 bc->u_out = 0.0f;
137 }
138
139 return bc->u_out;
140}
